58  CentralBodyData( const std::vector< std::string >& centralBodies,
59  const std::vector< std::string >& bodiesToIntegrate,
60  const std::map< std::string,
61  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >&
62  bodyStateFunctions,
63  const std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) >
64  globalFrameOriginBarycentricStateFunction,
65  const std::string globalFrameOrigin ):
66  centralBodies_( centralBodies ), globalFrameOriginBarycentricStateFunction_( globalFrameOriginBarycentricStateFunction )
67  {
68  // Check consistency of input.
69  if( centralBodies.size( ) != bodiesToIntegrate.size( ) )
70  {
71  throw std::runtime_error(
72  "Error in CentralBodyData, number of central bodies not equal to number of bodies to integrate " );
73  }
74 
75 
76  bodyOriginType_.resize( bodiesToIntegrate.size( ) );
77 
78  // Iterate over all integrated bodies to set body origin, its type and associated data.
79  for( unsigned int i = 0; i < bodiesToIntegrate.size( ); i++ )
80  {
81  // Check if central body is inertial.
82  if( centralBodies.at( i ) == globalFrameOrigin )
83  {
84  bodyOriginType_.at( i ) = global_frame_origin;
85  }
86  else
87  {
88  // Check if central body of current integrated body is also integrated.
89  int centralBodyIndex = -1;
90  for( unsigned int j = 0; j < bodiesToIntegrate.size( ); j++ )
91  {
92  // If there is a match between central body and an integrated body, store body
93  // index.
94  if( centralBodies.at( i ) == bodiesToIntegrate[ j ] )
95  {
96  centralBodyIndex = j;
97  if( i == j )
98  {
99  throw std::runtime_error( "Error, body " + bodiesToIntegrate[ j ] +
100  " cannot be its own central body" );
101  }
102  }
103  }
104 
105  // If no integrated central body foundl set body origin as being from an ephemeris.
106  if( centralBodyIndex == -1 )
107  {
108  bodyOriginType_.at( i ) = from_ephemeris;
109  centralBodiesFromEphemerides_[ i ]
110  = bodyStateFunctions.at( centralBodies.at( i ) );
111  }
112  // Else, set body origin as being from another integrated body, set indices of both
113  // bodies in centralBodiesFromIntegration_
114  else
115  {
116  bodyOriginType_.at( i ) = from_integration;
117  centralBodiesFromIntegration_[ i ] = centralBodyIndex;
118  }
119  }
120  }
121 
122  std::vector< int > numericalBodies_;
123 
124  // Set initial update order of bodies.
125  updateOrder_.resize( bodiesToIntegrate.size( ) );
126  int currentUpdateIndex = 0;
127  for( unsigned int i = 0; i < bodyOriginType_.size( ); i++ )
128  {
129  // If body origin is not from another integrated body, order in update is irrelevant,
130  // set at current index of vector.
131  if( bodyOriginType_.at( i ) == global_frame_origin || bodyOriginType_.at( i ) == from_ephemeris )
132  {
133  updateOrder_[ currentUpdateIndex ] = i;
134  currentUpdateIndex++;
135  }
136  // Else, store index of body having another integrated body as central body
137  else if( bodyOriginType_.at( i ) == from_integration )
138  {
139  numericalBodies_.push_back( i );
140  }
141  }
142 
143  // Switch body order if there are dependencies between bodies,
144  int bodyToMove;
145  for( unsigned int i = 0; i < numericalBodies_.size( ); i++ )
146  {
147  for( unsigned int j = 0; j < i; j++ )
148  {
149  if( centralBodies[ numericalBodies_[ j ] ]
150  == bodiesToIntegrate[ numericalBodies_.at( i ) ] )
151  {
152  // Move central body to index before integrated body.
153  bodyToMove = numericalBodies_.at( i );
154  numericalBodies_.erase( numericalBodies_.begin( ) + i );
155  numericalBodies_.insert( numericalBodies_.begin( ) + j, bodyToMove );
156  break;
157  }
158  }
159  }
160 
161  // Add propagated bodies to list.
162  for( unsigned int i = 0; i < numericalBodies_.size( ); i++ )
163  {
164  updateOrder_[ currentUpdateIndex ] = numericalBodies_.at( i );
165  currentUpdateIndex++;
166  }
167 
168  localInternalState_.resize( 6 * bodiesToIntegrate.size( ), 1 );
169  }

186  void getReferenceFrameOriginInertialStates(
187  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& internalState, const TimeType time,
188  std::vector< Eigen::Matrix< StateScalarType, 6, 1 > >& referenceFrameOriginStates,
189  const bool areInputStateLocal = true )
190  {
191  localInternalState_ = internalState;
192  if( referenceFrameOriginStates.size( ) != updateOrder_.size( ) )
193  {
194  referenceFrameOriginStates.resize( updateOrder_.size( ) );
195  }
196 
197  // Update state in correct order.
198  for( unsigned int i = 0; i < updateOrder_.size( ); i++ )
199  {
200  getSingleReferenceFrameOriginInertialState(
201  localInternalState_, time, updateOrder_.at( i ),
202  referenceFrameOriginStates.at( updateOrder_.at( i ) ));
203 
204  // Modify current input state to global frame if input is local (in propagation frame).
205  if( areInputStateLocal )
206  {
207  localInternalState_.segment( 6 * updateOrder_.at( i ), 6 ) +=
208  std::move( referenceFrameOriginStates.at( updateOrder_.at( i ) ) );
209  }
210  }

280  const TimeType time,

298  std::to_string( bodyOriginType_[ bodyIndex ] ) );
299  break;
