82  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
83  getDerivativeFunctionWrtStateOfIntegratedBody(
84  const std::pair< std::string, std::string >& stateReferencePoint,
85  const propagators::IntegratedStateType integratedStateType )
86  {
87  // Initialize to empty function; 0 parameter size.
88  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
89  partialFunction = std::make_pair( std::function< void( Eigen::Block< Eigen::MatrixXd > ) >( ), 0 );
90 
91  // Check if state dependency exists
92  switch( integratedStateType )
93  {
94  case propagators::rotational_state:
95  {
96  // Check if reference id is consistent.
97  if( stateReferencePoint.second != "" )
98  {
99  throw std::runtime_error( "Error when getting state derivative partial torque model, cannot have reference point on body for dynamics" );
100  }
101  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
102  else if( stateReferencePoint.first == bodyUndergoingTorque_ )
103  {
104  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtRotationalStateOfAcceleratedBody, this, std::placeholders::_1 ), 7 );
105  }
106  else if( stateReferencePoint.first == bodyExertingTorque_ )
107  {
108  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtRotationalStateOfAcceleratingBody, this, std::placeholders::_1 ), 7 );
109  }
110  else if( isTorquePartialWrtAdditionalBodyNonNull( stateReferencePoint.first ) )
111  {
112  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtRotationalStateOfAdditionalBody,
113  this, std::placeholders::_1, stateReferencePoint.first ), 3 );
114  }
115  break;
116  }
117  case propagators::translational_state:
118  {
119  // Check if reference id is consistent.
120  if( stateReferencePoint.second != "" )
121  {
122  throw std::runtime_error( "Error when getting torque partial, cannot have reference point on body for body mass" );
123  }
124  else if( isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType ) )
125  {
126  partialFunction = std::make_pair( std::bind( &TorquePartial::wrtNonRotationalStateOfAdditionalBody,
127  this, std::placeholders::_1, stateReferencePoint, integratedStateType ), 1 );
128  }
129  }
130  case propagators::custom_state:
131  {
132  break;
133  }
134  default:
135  std::string errorMessage =
136  "Error when getting state derivative partial torque model, dynamics type " +
137  std::to_string( integratedStateType ) + "not recognized" ;
138  throw std::runtime_error( errorMessage );
139  break;
140  }
141 
142 
143  return partialFunction;
144  }
