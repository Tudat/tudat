684 template< typename IndependentVariableType, typename DependentVariableType,
685  typename IndependentVariableStepType = IndependentVariableType >
686 std::shared_ptr< numerical_integrators::NumericalIntegrator< IndependentVariableType, DependentVariableType,
687 DependentVariableType, IndependentVariableStepType > > createIntegrator(
688  std::function< DependentVariableType(
689  const IndependentVariableType, const DependentVariableType& ) > stateDerivativeFunction,
690  const DependentVariableType initialState,
691  std::shared_ptr< IntegratorSettings< IndependentVariableType > > integratorSettings )
692 {
693  // Declare eventual output
694  std::shared_ptr< NumericalIntegrator
695  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > > integrator;
696 
697  // Retrieve requested type of integrator
698  switch( integratorSettings->integratorType_ )
699  {
700  case euler:
701  {
702  // Create Euler integrator
703  integrator = std::make_shared< EulerIntegrator
704  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
705  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
706  break;
707  }
708  case rungeKutta4:
709  {
710  // Create Runge-Kutta 4 integrator
711  integrator = std::make_shared< RungeKutta4Integrator
712  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
713  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
714  break;
715  }
716  case rungeKuttaVariableStepSize:
717  {
718  // Cast integrator
719  std::shared_ptr< RungeKuttaVariableStepSizeBaseSettings< IndependentVariableType > >
720  variableStepIntegratorSettings = std::dynamic_pointer_cast< RungeKuttaVariableStepSizeBaseSettings<
721  IndependentVariableType > >( integratorSettings );
722 
723  if( std::fabs( static_cast<double>(variableStepIntegratorSettings->initialTimeStep_) ) <
724  std::fabs( static_cast<double>(variableStepIntegratorSettings->minimumStepSize_) ) )
725  {
726  throw std::runtime_error( "Error when making RK variable step-size integrator: initial step size is smaller than minimum step" );
727  }
728 
729 
730  if( std::fabs( static_cast<double>(variableStepIntegratorSettings->initialTimeStep_) ) >
731  std::fabs( static_cast<double>(variableStepIntegratorSettings->maximumStepSize_) ) )
732  {
733  throw std::runtime_error( "Error when making RK variable step-size integrator: initial step size is larger than maximum step" );
734  }
735 
736  // Check input consistency
737  if ( variableStepIntegratorSettings == nullptr )
738  {
739  throw std::runtime_error( "Error, type of integrator settings (rungeKuttaVariableStepSize) not compatible with "
740  "selected integrator (derived class of IntegratorSettings must be "
741  "RungeKuttaVariableStepSizeBaseSettings for this type)." );
742  }
743 
744  // Get requested RK coefficients
745  RungeKuttaCoefficients coefficients = RungeKuttaCoefficients::get( variableStepIntegratorSettings->coefficientSet_ );
746 
747  // Check which constructor is being used
748  if ( variableStepIntegratorSettings->areTolerancesDefinedAsScalar_ )
749  {
750  // Settings with scalar tolerances
751  std::shared_ptr< RungeKuttaVariableStepSizeSettingsScalarTolerances< IndependentVariableType > >
752  scalarTolerancesIntegratorSettings = std::dynamic_pointer_cast<
753  RungeKuttaVariableStepSizeSettingsScalarTolerances< IndependentVariableType > >( variableStepIntegratorSettings );
754 
755  // Check input consistency
756  if ( scalarTolerancesIntegratorSettings == nullptr )
757  {
758  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. Input class must be of "
759  "RungeKuttaVariableStepSizeSettingsScalarTolerances type." );
760  }
761 
762  // Create Runge-Kutta integrator with scalar tolerances
763  integrator = std::make_shared< RungeKuttaVariableStepSizeIntegrator
764  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
765  ( coefficients, stateDerivativeFunction, integratorSettings->initialTime_, initialState,
766  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->minimumStepSize_ ),
767  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->maximumStepSize_ ),
768  static_cast< typename DependentVariableType::Scalar >( scalarTolerancesIntegratorSettings->relativeErrorTolerance_ ),
769  static_cast< typename DependentVariableType::Scalar >( scalarTolerancesIntegratorSettings->absoluteErrorTolerance_ ),
770  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->safetyFactorForNextStepSize_ ),
771  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
772  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
773  }
774  else
775  {
776  // Settings with vector tolerances
777  std::shared_ptr< RungeKuttaVariableStepSizeSettingsVectorTolerances< IndependentVariableType > >
778  vectorTolerancesIntegratorSettings = std::dynamic_pointer_cast<
779  RungeKuttaVariableStepSizeSettingsVectorTolerances< IndependentVariableType > >(
780  variableStepIntegratorSettings );
781 
782  // Check input consistency
783  if ( vectorTolerancesIntegratorSettings == nullptr )
784  {
785  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. Input class must be of "
786  "RungeKuttaVariableStepSizeSettingsVectorTolerances type with suitable independent variable type." );
787  }
788 
789  // Check that sizes of tolerances and initial state match
790  auto relativeErrorTolerance = vectorTolerancesIntegratorSettings->relativeErrorTolerance_.template cast<
791  typename DependentVariableType::Scalar >( );
792  auto absoluteErrorTolerance = vectorTolerancesIntegratorSettings->absoluteErrorTolerance_.template cast<
793  typename DependentVariableType::Scalar >( );
794  if ( ( relativeErrorTolerance.rows( ) != initialState.rows( ) ) ||
795  ( relativeErrorTolerance.cols( ) != initialState.cols( ) ) ||
796  ( absoluteErrorTolerance.rows( ) != initialState.rows( ) ) ||
797  ( absoluteErrorTolerance.cols( ) != initialState.cols( ) ) )
798  {
799  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. The sizes of the "
800  "relative and absolute tolerance vectors do not match the size of the initial state. "
801  "This could be the case if you are propagating more than just one state, e.g., translational "
802  "and/or rotational dynamics and mass, or more than one body." );
803  }
804 
805  // Create Runge-Kutta integrator with vector tolerances
806  integrator = std::make_shared< RungeKuttaVariableStepSizeIntegrator
807  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
808  ( coefficients, stateDerivativeFunction, integratorSettings->initialTime_, initialState,
809  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->minimumStepSize_ ),
810  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->maximumStepSize_ ),
811  relativeErrorTolerance, absoluteErrorTolerance,
812  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->safetyFactorForNextStepSize_ ),
813  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
814  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
815  }
816  break;
817  }
818  case bulirschStoer:
819  {
820  // Check input consistency
821  std::shared_ptr< BulirschStoerIntegratorSettings< IndependentVariableType > > bulirschStoerIntegratorSettings =
822  std::dynamic_pointer_cast< BulirschStoerIntegratorSettings< IndependentVariableType > >(
823  integratorSettings );
824 
825
