101  receiverState, transmitterState, time, isTimeAtReception );
102 
103  // Compute spherical relative position
104  Eigen::Matrix< ObservationScalarType, 3, 1 > sphericalRelativeCoordinates =
105  coordinate_conversions::convertCartesianToSpherical< ObservationScalarType >(
106  transmitterState.segment( 0, 3 ) - receiverState.segment( 0, 3 ) ).
107  template cast< ObservationScalarType >( );
108 
109  // Set link end times and states.
110  linkEndTimes.clear( );
111  linkEndStates.clear( );
112  linkEndStates.push_back( transmitterState.template cast< double >( ) );
113  linkEndStates.push_back( receiverState.template cast< double >( ) );
114 
115  if( isTimeAtReception )
116  {
117  linkEndTimes.push_back( static_cast< double >( time - lightTime ) );
118  linkEndTimes.push_back( static_cast< double >( time ) );

120  else
