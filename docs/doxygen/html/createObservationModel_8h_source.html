637 template< typename ObservationScalarType = double, typename TimeType = double >
638 std::shared_ptr< DopplerProperTimeRateInterface > createOneWayDopplerProperTimeCalculator(
639  std::shared_ptr< DopplerProperTimeRateSettings > properTimeRateSettings,
640  const LinkEnds& linkEnds,
641  const simulation_setup::SystemOfBodies &bodies,
642  const LinkEndType linkEndForCalculator )
643 {
644  std::shared_ptr< DopplerProperTimeRateInterface > properTimeRateInterface;
645 
646  // Check tyope of proper time rate model
647  switch( properTimeRateSettings->dopplerProperTimeRateType_ )
648  {
649  case direct_first_order_doppler_proper_time_rate:
650  {
651  // Check input consistency
652  std::shared_ptr< DirectFirstOrderDopplerProperTimeRateSettings > directFirstOrderDopplerProperTimeRateSettings =
653  std::dynamic_pointer_cast< DirectFirstOrderDopplerProperTimeRateSettings >( properTimeRateSettings );
654  if( directFirstOrderDopplerProperTimeRateSettings == nullptr )
655  {
656  throw std::runtime_error( "Error when making DopplerProperTimeRateInterface, input type (direct_first_order_doppler_proper_time_rate) is inconsistent" );
657  }
658  else if( linkEnds.count( linkEndForCalculator ) == 0 )
659  {
660  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not find link end " +
661  std::to_string( linkEndForCalculator );
662  throw std::runtime_error( errorMessage );
663  }
664  else
665  {
666  if( bodies.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->getGravityFieldModel( ) == nullptr )
667  {
668  throw std::runtime_error( "Error when making DirectFirstOrderDopplerProperTimeRateInterface, no gravity field found for " +
669  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ );
670  }
671  else
672  {
673  // Retrieve gravitational parameter
674  std::function< double( ) > gravitationalParameterFunction =
675  std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
676  bodies.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->
677  getGravityFieldModel( ) );
678 
679  // Create calculation object.
680  LinkEndId referencePointId =
681  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, "" );
682  if( ( linkEnds.at( receiver ) != referencePointId ) && ( linkEnds.at( transmitter ) != referencePointId ) )
683  {
684  properTimeRateInterface = std::make_shared<
685  DirectFirstOrderDopplerProperTimeRateInterface >(
686  linkEndForCalculator, gravitationalParameterFunction,
687  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, unidentified_link_end,
688  getLinkEndCompleteEphemerisFunction< double, double >(
689  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, ""), bodies ) );
690  }
691  else
692  {
693  throw std::runtime_error(
694  "Error, proper time reference point as link end not yet implemented for DopplerProperTimeRateInterface creation" );
695  }
696  }
697  }
698  break;
699  }
700  default:
701  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not recognize type " +
702  std::to_string( properTimeRateSettings->dopplerProperTimeRateType_ );
703  throw std::runtime_error( errorMessage );
704  }
705  return properTimeRateInterface;
706 }

752 template< int ObservationSize = 1 >
753 std::shared_ptr< ObservationBias< ObservationSize > > createObservationBiasCalculator(
754  const LinkEnds linkEnds,
755  const ObservableType observableType,
756  const std::shared_ptr< ObservationBiasSettings > biasSettings,
757  const simulation_setup::SystemOfBodies &bodies )
758 {
759  std::shared_ptr< ObservationBias< ObservationSize > > observationBias;
760  switch( biasSettings->observationBiasType_ )
761  {
762  case constant_absolute_bias:
763  {
764  // Check input consistency
765  std::shared_ptr< ConstantObservationBiasSettings > constantBiasSettings = std::dynamic_pointer_cast<
766  ConstantObservationBiasSettings >( biasSettings );
767  if( constantBiasSettings == nullptr )
768  {
769  throw std::runtime_error( "Error when making constant observation bias, settings are inconsistent" );
770  }
771 
772  if( !constantBiasSettings->useAbsoluteBias_ )
773  {
774  throw std::runtime_error( "Error when making constant observation bias, class settings are inconsistent" );
775  }
776 
777  // Check if size of bias is consistent with requested observable size
778  if( constantBiasSettings->observationBias_.rows( ) != ObservationSize )
779  {
780  throw std::runtime_error( "Error when making constant observation bias, bias size is inconsistent" );
781  }
782  observationBias = std::make_shared< ConstantObservationBias< ObservationSize > >(
783  constantBiasSettings->observationBias_ );
784  break;
785  }
786  case arc_wise_constant_absolute_bias:
787  {
788  // Check input consistency
789  std::shared_ptr< ArcWiseConstantObservationBiasSettings > arcwiseBiasSettings = std::dynamic_pointer_cast<
790  ArcWiseConstantObservationBiasSettings >( biasSettings );
791  if( arcwiseBiasSettings == nullptr )
792  {
793  throw std::runtime_error( "Error when making arc-wise observation bias, settings are inconsistent" );
794  }
