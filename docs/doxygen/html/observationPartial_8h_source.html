481 template< int ObservationSize >
482 std::shared_ptr< ObservationPartial< ObservationSize > > createObservationPartialWrtLinkProperty(
483  const observation_models::LinkEnds& linkEnds,
484  const observation_models::ObservableType observableType,
485  const std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > parameterToEstimate,
486  const bool useBiasPartials = true )
487 {
488  std::shared_ptr< ObservationPartial< ObservationSize > > observationPartial;
489 
490  // Check parameter type
491  switch( parameterToEstimate->getParameterName( ).first )
492  {
493  case estimatable_parameters::constant_additive_observation_bias:
494  {
495  if( useBiasPartials )
496  {
497  // Check input consistency
498  std::shared_ptr< estimatable_parameters::ConstantObservationBiasParameter > constantBias =
499  std::dynamic_pointer_cast< estimatable_parameters::ConstantObservationBiasParameter >(
500  parameterToEstimate );
501  if( constantBias == nullptr )
502  {
503  throw std::runtime_error( "Error when making partial w.r.t. observation bias, type is inconsistent" );
504  }
505  else
506  {
507  // Check dependency between parameter and link properties.
508  if( linkEnds == constantBias->getLinkEnds( ) && observableType == constantBias->getObservableType( ) )
509  {
510  observationPartial = std::make_shared< ObservationPartialWrtConstantAbsoluteBias< ObservationSize > >(
511  observableType, linkEnds );
512  }
513  }
514  }
515  break;
516  }
517  case estimatable_parameters::arcwise_constant_additive_observation_bias:
518  {
519  if( useBiasPartials )
520  {
521  // Check input consistency
522  std::shared_ptr< estimatable_parameters::ArcWiseObservationBiasParameter > arcwiseBias =
523  std::dynamic_pointer_cast< estimatable_parameters::ArcWiseObservationBiasParameter >(
524  parameterToEstimate );
525  if( arcwiseBias == nullptr )
526  {
527  throw std::runtime_error( "Error when making partial w.r.t. arcwise observation bias, type is inconsistent" );
528  }
529  else
530  {
531  // Check dependency between parameter and link properties.
532  if( linkEnds == arcwiseBias->getLinkEnds( ) && observableType == arcwiseBias->getObservableType( ) )
533  {
534  observationPartial = std::make_shared< ObservationPartialWrtArcWiseAbsoluteBias< ObservationSize > >(
535  observableType, linkEnds,
536  arcwiseBias->getLookupScheme( ),
537  arcwiseBias->getLinkEndIndex( ),
538  arcwiseBias->getArcStartTimes( ).size( ) );
539  }
540  }
541  }
542  break;
543  }
544  case estimatable_parameters::constant_relative_observation_bias:
545  {
546  if( useBiasPartials )
