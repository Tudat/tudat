236  void update( const double currentTime = 0.0 )
237  {
238  if( !( currentTime_ == currentTime ) )
239  {
240  // Compute helper quantities.
241  Eigen::Vector3d rangeVector = ( acceleratedBodyState_( ) - sourceBodyState_( ) );
242  double range = rangeVector.norm( );
243  double rangeInverse = 1.0 / ( range );
244 
245  // Compute position partial.
246  currentPartialWrtPosition_ =
247  ( radiationPressureCoefficientFunction_( ) * areaFunction_( ) * radiationPressureFunction_( ) /
248  acceleratedBodyMassFunction_( ) ) * ( Eigen::Matrix3d::Identity( ) * rangeInverse - 3.0 *
249  rangeVector * rangeVector.transpose( ) * rangeInverse / (
250  range * range ) );
251  currentTime_ = currentTime;
252  }
253  }
