74 template< typename StateScalarType >
75 Eigen::Matrix< StateScalarType, 6, 1 > transformStateToFrameFromRotations(
76  const Eigen::Matrix< StateScalarType, 6, 1 >& stateInBaseFrame,
77  const Eigen::Quaterniond& rotationToFrame,
78  const Eigen::Matrix3d& rotationMatrixToFrameDerivative )
79 {
80  Eigen::Matrix< StateScalarType, 6, 1 >stateInTargetFrame;
81  stateInTargetFrame.segment( 0, 3 ) =
82  rotationToFrame.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 0, 3 );
83  stateInTargetFrame.segment( 3, 3 ) =
84  rotationMatrixToFrameDerivative.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 0, 3 ) +
85  rotationToFrame.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 3, 3 );
86  return stateInTargetFrame;
87 }
