276 int getSingleArcParameterSetSize(
277  std::shared_ptr< EstimatableParameterSet< InitialStateParameterType > > estimatableParameterSet )
278 {
279  int totalParameterSetSize = estimatableParameterSet->getEstimatedParameterSetSize( );
280  std::vector< std::shared_ptr< EstimatableParameter< Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > >
281  initialStateParameters = estimatableParameterSet->getEstimatedInitialStateParameters( );
282 
283  for( unsigned int i = 0; i < initialStateParameters.size( ); i++ )
284  {
285  if( initialStateParameters.at( i )->getParameterName( ).first == arc_wise_initial_body_state )
286  {
287  totalParameterSetSize -=
288  ( std::dynamic_pointer_cast< ArcWiseInitialTranslationalStateParameter< InitialStateParameterType > >(
289  initialStateParameters.at( i ) )->getNumberOfStateArcs( ) - 1 ) * 6;
290  }
291  else if( ( initialStateParameters.at( i )->getParameterName( ).first != initial_body_state ) &&
292  ( initialStateParameters.at( i )->getParameterName( ).first != initial_rotational_body_state ))
293  {
294  throw std::runtime_error(
295  "Error when getting single arc paramater vector, did not recognize initial state parameter " +
296  std::to_string( initialStateParameters.at( i )->getParameterName( ).first ) );
297  }
298  }
299  return totalParameterSetSize;

325 template< typename InitialStateParameterType >
326 std::vector< double > getMultiArcStateEstimationArcStartTimes(
327  const std::shared_ptr< EstimatableParameterSet< InitialStateParameterType > > estimatableParameters,
328  const bool throwErrorOnSingleArcDynamics = true )
329 
330 {
331  // Retrieve initial dynamical parameters.
332  std::vector< std::shared_ptr< EstimatableParameter<
333  Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
334  estimatableParameters->getEstimatedInitialStateParameters( );
335 
336  std::vector< double > arcStartTimes;
337 
338  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
339  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
340  {
341  if( initialDynamicalParameters.at( i )->getParameterName( ).first == arc_wise_initial_body_state )
342  {
343  std::shared_ptr< ArcWiseInitialTranslationalStateParameter< InitialStateParameterType > > arcWiseStateParameter =
344  std::dynamic_pointer_cast< ArcWiseInitialTranslationalStateParameter< InitialStateParameterType > >(
345  initialDynamicalParameters.at( i ) );
346  if( arcWiseStateParameter == nullptr )
347  {
348  throw std::runtime_error( "Error when getting arc times from estimated parameters, parameter is inconsistent" );
349  }
350 
351  if( arcStartTimes.size( ) == 0 )
352  {
353  arcStartTimes = arcWiseStateParameter->getArcStartTimes( );
354  }
355  else
356  {
357  // Check arc consistency
358  std::vector< double > currentArcStartTimes = arcWiseStateParameter->getArcStartTimes( );
359  if( currentArcStartTimes.size( ) != arcStartTimes.size( ) )
360  {
361  throw std::runtime_error(
362  "Error when getting arc start times for parameters, number of arcs is inconsistent" );
363  }
364  else
365  {
366  for( unsigned int j = 0; j < arcStartTimes.size( ); j++ )
367  {
368  if( arcStartTimes.at( j ) != currentArcStartTimes.at( j ) )
369  {
370  throw std::runtime_error(
371  "Error when getting arc start times for parameters, arc times are inconsistent" );
372  }
373  }
374  }
375  }
376 
377  }
378  else
379  {
380  if( throwErrorOnSingleArcDynamics )
381  {
382  throw std::runtime_error( "Error when getting arc times from estimated parameters, soingle arc dynamics found" );
383  }
384  }
385  }
386 
387  return arcStartTimes;
388 }
