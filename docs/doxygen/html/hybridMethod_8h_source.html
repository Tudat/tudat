31  HybridMethod(
32  const Eigen::Vector6d& stateAtDeparture,
33  const Eigen::Vector6d& stateAtArrival,
34  const double maximumThrust,
35  const double specificImpulse,
36  const double timeOfFlight,
37  std::shared_ptr< simulation_setup::OptimisationSettings > optimisationSettings,
38  const std::pair< double, double > initialAndFinalMEEcostatesBounds = std::make_pair( - 10.0, 10.0 ) ):
39  LowThrustLeg( stateAtDeparture, stateAtArrival, timeOfFlight ),
40  maximumThrust_( maximumThrust ),
41  specificImpulse_( specificImpulse ),
42  optimisationSettings_( optimisationSettings ),
43  initialAndFinalMEEcostatesBounds_( initialAndFinalMEEcostatesBounds )
44  {
45 
46  // Store initial spacecraft mass.
47  initialSpacecraftMass_ = bodies_[ bodyToPropagate_ ]->getBodyMass();
48 
49  // Convert the thrust model proposed as initial guess into simplified thrust model adapted to the hybrid method.
50  if ( optimisationSettings_->initialGuessThrustModel_.first.size( ) != 0 )
51  {
52  if ( optimisationSettings_->initialGuessThrustModel_.first.size( ) != 10 )
53  {
54  throw std::runtime_error( "Error when providing an initial guess for hybrid method, vector size"
55  "unconsistent with the 5 initial and 5 final MEE costate values which are required." );
56  }
57  else
58  {
59  initialGuessThrustModel_.first = optimisationSettings_->initialGuessThrustModel_.first;
60  }
61  }
62  else
63  {
64  initialGuessThrustModel_.first = std::vector< double>( );
65  }
66  initialGuessThrustModel_.second = optimisationSettings_->initialGuessThrustModel_.second;
67 
68  // Perform optimisation
69  std::pair< std::vector< double >, std::vector< double > > bestIndividual = performOptimisation( );
70  championFitness_ = bestIndividual.first;
71  championDesignVariables_ = bestIndividual.second;
72 
73 
74  // Transform vector of design variables into 3D vector of throttles.
75  Eigen::VectorXd initialCostates; initialCostates.resize( 5 );
76  Eigen::VectorXd finalCostates; finalCostates.resize( 5 );
77  for ( unsigned int i = 0 ; i < 5 ; i++ )
78  {
79  initialCostates[ i ] = championDesignVariables_[ i ];
80  finalCostates[ i ] = championDesignVariables_[ i + 5 ];
81  }
82 
83  bodies_[ bodyToPropagate_ ]->setConstantBodyMass( initialSpacecraftMass_ );
84 
85  // Create Sims-Flanagan leg from the best optimisation individual.
86  hybridMethodModel_ = std::make_shared< HybridMethodModel >(
87  stateAtDeparture_, stateAtArrival_, initialCostates, finalCostates, maximumThrust_, specificImpulse_, timeOfFlight_ );
88 
89  }
