75  template< typename StateScalarType = double, typename TimeType = double >
76  std::shared_ptr< Ephemeris > getEphemeris(
77  const std::string& origin, const std::string& body )
78  {
79  typedef Eigen::Matrix< StateScalarType, 6, 1 > StateType;
80  std::shared_ptr< Ephemeris > ephemerisBetweenFrames;
81 
82  // If requested 'body' is global base frame, return constant zero ephemeris.
83  if( body == origin )
84  {
85  //NOTE: Should generalize to long double state type.
86  ephemerisBetweenFrames = std::make_shared< ConstantEphemeris >(
87  [ = ]( ){ return Eigen::Vector6d::Zero( ); }, origin, "ECLIPJ2000" );
88  }
89  else
90  {
91  // Set frames between which ephemeris must be created.
92  std::vector< std::string > framesToCheck;
93  framesToCheck.push_back( origin );
94  framesToCheck.push_back( body );
95 
96  // Find nearest common frame between frames.
97  std::pair< std::string, int > nearestCommonFrame = getNearestCommonFrame( framesToCheck );
98 
99  // Initialize list of ephemeris functions for composite ephemeris creation
100  std::map< int, std::pair< std::function< StateType( const TimeType& ) >, bool > >
101  totalEphemerisList;
102 
103  // If body is nearest common frame, get set of ephemeris and set to subtract them when
104  // making composite ephemeris.
105  std::vector< std::shared_ptr< Ephemeris > > ephemerisList;
106  if( nearestCommonFrame.first == body )
107  {
108  ephemerisList = getDirectEphemerisFromLowerToUpperFrame( body, origin );
109  for( unsigned int i = 0; i < ephemerisList.size( ); i++ )
110  {
111  totalEphemerisList[ i ] = std::make_pair(
112  std::bind( &Ephemeris::getTemplatedStateFromEphemeris
113  < StateScalarType, TimeType >,
114  ephemerisList[ i ], std::placeholders::_1 ), false );
115  }
116  }
117  // If origin is nearest common frame, get set of ephemeris and set to add them when
118  // making composite ephemeris.
119  else if( nearestCommonFrame.first == origin )
120  {
121  ephemerisList = getDirectEphemerisFromLowerToUpperFrame( origin, body );
122  for( unsigned int i = 0; i < ephemerisList.size( ); i++ )
123  {
124  totalEphemerisList[ i ] = std::make_pair(
125  std::bind( &Ephemeris::getTemplatedStateFromEphemeris
126  < StateScalarType, TimeType >,
127  ephemerisList[ i ], std::placeholders::_1 ), true );
128  }
129  }
130  // If nearest common frame is neither input, create link from both to nearest common frame.
131  else
132  {
133  // Get set of ephemeris from nearest common frame to body and set to add them when
134  // making composite ephemeris.
135  ephemerisList = getDirectEphemerisFromLowerToUpperFrame( nearestCommonFrame.first, body );
136  for( unsigned int i = 0; i < ephemerisList.size( ); i++ )
137  {
138  totalEphemerisList[ i ] = std::make_pair(
139  std::bind( &Ephemeris::getTemplatedStateFromEphemeris
140  < StateScalarType, TimeType >,
141  ephemerisList[ i ], std::placeholders::_1 ), true );
142  }
143  int firstListSize = ephemerisList.size( );
144 
145  // Get set of ephemeris from nearest common frame to origin and set to subtract them
146  // when making composite ephemeris.
147  ephemerisList = getDirectEphemerisFromLowerToUpperFrame( nearestCommonFrame.first, origin );
148  for( unsigned int i = 0; i < ephemerisList.size( ); i++ )
149  {
150  totalEphemerisList[ i + firstListSize ] = std::make_pair(
151  std::bind( &Ephemeris::getTemplatedStateFromEphemeris
152  < StateScalarType, TimeType >,
153  ephemerisList[ i ], std::placeholders::_1 ), false );
154  }
155  }
156 
157  // Create composite ephemeris
158  ephemerisBetweenFrames = std::make_shared< CompositeEphemeris< TimeType, StateScalarType > >(
159  totalEphemerisList,
160  std::map< int, std::function< StateType( const TimeType, const StateType& ) > >( ),
161  origin );
162  }
163 
164  return ephemerisBetweenFrames;
165  }

252 std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >
253 getTranslationFunctionsFromIntegrationFrameToEphemerisFrame(
254  const std::vector< std::string >& centralBodies,
255  const std::vector< std::string >& bodiesToIntegrate,
256  const std::shared_ptr< ephemerides::ReferenceFrameManager > frameManager )
257 {
258  std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >
259  translationFunctionMap;
260 
261  // Check consistency of input.
262  if( centralBodies.size( ) != bodiesToIntegrate.size( ) )
263  {
264  throw std::runtime_error(
265  "Error when making translation functrions from integration to ephemeris frame, input vector sizes inconsistent" );
266  }
267  else
268  {
269  for( unsigned int i = 0; i < centralBodies.size( ); i++ )
270  {
271  // If ephemeris and integration origin are not equal, provide translation function.
272  if( centralBodies.at( i ) != frameManager->getBaseFrameNameOfBody( bodiesToIntegrate.at( i ) ) )
273  {
274  translationFunctionMap[ bodiesToIntegrate.at( i ) ] = std::bind(
275  &ephemerides::Ephemeris::getTemplatedStateFromEphemeris< StateScalarType, TimeType >,
276  frameManager->getEphemeris< StateScalarType, TimeType >(
277  centralBodies.at( i ),
278  frameManager->getBaseFrameNameOfBody( bodiesToIntegrate.at( i ) ) ), std::placeholders::_1 );
279  }
280  }
281  }
282 
283  return translationFunctionMap;
284 }
