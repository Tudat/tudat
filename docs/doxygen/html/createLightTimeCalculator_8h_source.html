67 std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType& ) > getLinkEndCompleteEphemerisFunction(
68  const std::shared_ptr< simulation_setup::Body > bodyWithLinkEnd,

70 {
71  typedef Eigen::Matrix< StateScalarType, 6, 1 > StateType;
72 
73  std::function< StateType( const TimeType& ) > linkEndCompleteEphemerisFunction;
74 
75  // Checking transmitter if a reference point is to be used
76  if( linkEndId.second != "" )
77  {
78  if( bodyWithLinkEnd->getGroundStationMap( ).count( linkEndId.second ) == 0 )
79  {
80  std::string errorMessage = "Error when making ephemeris function for " + linkEndId.first + ", " +
81  linkEndId.second + ", station not found.";
82  throw std::runtime_error( errorMessage );
83  }
84 
85  // Retrieve function to calculate state of transmitter S/C
86  linkEndCompleteEphemerisFunction =
87  std::bind( &ephemerides::Ephemeris::getTemplatedStateFromEphemeris< StateScalarType,TimeType >,
88  createReferencePointEphemeris< TimeType, StateScalarType >(
89  bodyWithLinkEnd, bodyWithLinkEnd->getRotationalEphemeris( ),
90  std::bind( &ground_stations::GroundStation::getStateInPlanetFixedFrame
91  < StateScalarType, TimeType >,
92  bodyWithLinkEnd->getGroundStation( linkEndId.second ), std::placeholders::_1 ) ), std::placeholders::_1 );
93 
94  }
95  // Else, create state function for center of mass
96  else
97  {
98  // Create function to calculate state of transmitting ground station.
99  linkEndCompleteEphemerisFunction =
100  std::bind( &simulation_setup::Body::getStateInBaseFrameFromEphemeris< StateScalarType, TimeType >,
101  bodyWithLinkEnd, std::placeholders::_1 );
102  }
