41 template< typename ObservationScalarType = double >
42 Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > getConcatenatedWeightsVector(
43  const typename std::map< observation_models::ObservableType, std::map<
44  observation_models::LinkEnds, Eigen::VectorXd > >& weightsData )
45 {
46  typedef std::map< observation_models::ObservableType, std::map< observation_models::LinkEnds, Eigen::VectorXd > >
47  WeightsDataStructure;
48 
49  // Get total required size of weights vector
50  int totalNumberOfObservations = 0;
51  for( typename WeightsDataStructure::const_iterator observablesIterator =
52  weightsData.begin( ); observablesIterator != weightsData.end( ); observablesIterator++ )
53  {
54  for( typename std::map< observation_models::LinkEnds, Eigen::VectorXd >::const_iterator dataIterator =
55  observablesIterator->second.begin( ); dataIterator != observablesIterator->second.end( ); dataIterator++ )
56  {
57  totalNumberOfObservations += dataIterator->second.rows( );
58  }
59  }
60  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > concatenatedWeights = Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >::Zero( totalNumberOfObservations, 1 );
61 
62  // Iterate over all observations and concatenate the weight vectors.
63  int currentIndex = 0;
64  for( typename WeightsDataStructure::const_iterator observablesIterator =
65  weightsData.begin( ); observablesIterator != weightsData.end( ); observablesIterator++ )
66  {
67  for( typename std::map< observation_models::LinkEnds, Eigen::VectorXd >::const_iterator dataIterator =
68  observablesIterator->second.begin( ); dataIterator != observablesIterator->second.end( ); dataIterator++ )
69  {
70  concatenatedWeights.segment( currentIndex, dataIterator->second.rows( ) ) =
71  dataIterator->second.template cast< ObservationScalarType >( );
72  currentIndex += dataIterator->second.rows( );
73  }
74  }
75 
76  return concatenatedWeights;
77 }

265  static std::pair< std::map< observation_models::ObservableType, int >, int > getNumberOfObservationsPerObservable(
266  const PodInputType& observationsAndTimes )
267  {
268  // Initialize counters.
269  std::map< observation_models::ObservableType, int > numberOfObservations;
270  int totalNumberOfObservations = 0;
271 
272  // Iterate over all observabel types.
273  for( typename PodInputType::const_iterator observablesIterator = observationsAndTimes.begin( );
274  observablesIterator != observationsAndTimes.end( ); observablesIterator++ )
275  {
276  // Initialize number of observations for current observable
277  numberOfObservations[ observablesIterator->first ] = 0;
278 
279  // Iterate over all link ends.
280  for( typename SingleObservablePodInputType::const_iterator dataIterator = observablesIterator->second.begin( );
281  dataIterator != observablesIterator->second.end( ); dataIterator++ )
282  {
283  // Add number of observations with given link ends.
284  numberOfObservations[ observablesIterator->first ] += dataIterator->second.first.size( );
285  }
286 
287  // Add to total number of observations.
288  totalNumberOfObservations += numberOfObservations[ observablesIterator->first ];

291  return std::make_pair( numberOfObservations, totalNumberOfObservations );

305  void calculateObservationMatrixAndResiduals(
306  const PodInputType& observationsAndTimes, const int parameterVectorSize, const int totalObservationSize,
307  std::pair< Eigen::VectorXd, Eigen::MatrixXd >& residualsAndPartials )
308  {
309  // Initialize return data.
310  residualsAndPartials.second = Eigen::MatrixXd::Zero( totalObservationSize, parameterVectorSize );
311  residualsAndPartials.first = Eigen::VectorXd::Zero( totalObservationSize );
312 
313  // Declare variable denoting current index in vector of all observations.
314  int startIndex = 0;
315 
316  // Iterate over all observable types in observationsAndTimes
317  std::vector< TimeType > simulationInputTime;
318  for( typename PodInputType::const_iterator observablesIterator = observationsAndTimes.begin( );
319  observablesIterator != observationsAndTimes.end( ); observablesIterator++ )
320  {
321  int observableStartIndex = startIndex;
322 
323  // Iterate over all link ends for current observable type in observationsAndTimes
324  for( typename SingleObservablePodInputType::const_iterator dataIterator = observablesIterator->second.begin( );
325  dataIterator != observablesIterator->second.end( ); dataIterator++ )
326  {
327  simulationInputTime.clear( );
328  simulationInputTime = dataIterator->second.second.first;
329 
330  // Compute estimated ranges and range partials from current parameter estimate.
331  std::pair< ObservationVectorType, Eigen::MatrixXd > observationsWithPartials;
332  observationsWithPartials = observationManagers_[ observablesIterator->first ]->
333  computeObservationsWithPartials(
334  simulationInputTime, dataIterator->first, dataIterator->second.second.second );
335 
336  // Compute residuals for current link ends and observabel type.
337  residualsAndPartials.first.segment( startIndex, dataIterator->second.first.size( ) ) =
338  ( dataIterator->second.first - observationsWithPartials.first ).template cast< double >( );
339 
340  // Set current observation partials in matrix of all partials
341  residualsAndPartials.second.block( startIndex, 0, dataIterator->second.first.size( ), parameterVectorSize ) =
342  observationsWithPartials.second;
343 
344  // Increment current index of observation.
345  startIndex += dataIterator->second.first.size( );
346 
347  }
348 
349  int currentObservableSize = startIndex - observableStartIndex;
350  observation_models::checkObservationResidualDiscontinuities(
351  residualsAndPartials.first.block( observableStartIndex, 0, currentObservableSize, 1 ),
352  observablesIterator->first );
353  }

431  std::shared_ptr< PodOutput< ObservationScalarType, TimeType > > estimateParameters(
432  const std::shared_ptr< PodInput< ObservationScalarType, TimeType > > podInput,
433  std::shared_ptr< EstimationConvergenceChecker > convergenceChecker =
434  std::make_shared< EstimationConvergenceChecker >( ) )
435 
436  {
437  currentParameterEstimate_ = parametersToEstimate_->template getFullParameterValues< ObservationScalarType >( );
438 
439  // Get size of parameter vector and number of observations (total and per type)
440  int parameterVectorSize = currentParameterEstimate_.size( );
441  std::pair< std::map< observation_models::ObservableType, int >, int > observationNumberPair =
442  getNumberOfObservationsPerObservable( podInput->getObservationsAndTimes( ) );
443  int totalNumberOfObservations = observationNumberPair.second;
444 
445  // Declare variables to be returned (i.e. results from best iteration)
446  double bestResidual = TUDAT_NAN;
447  ParameterVectorType bestParameterEstimate = ParameterVectorType::Constant( parameterVectorSize, TUDAT_NAN );
448  Eigen::VectorXd bestTransformationData = Eigen::VectorXd::Constant( parameterVectorSize, TUDAT_NAN );
449  Eigen::VectorXd bestResiduals = Eigen::VectorXd::Constant( totalNumberOfObservations, TUDAT_NAN );
450  Eigen::MatrixXd bestInformationMatrix = Eigen::MatrixXd::Constant( totalNumberOfObservations, parameterVectorSize, TUDAT_NAN );
451  Eigen::VectorXd bestWeightsMatrixDiagonal = Eigen::VectorXd::Constant( totalNumberOfObservations, TUDAT_NAN );
452  Eigen::MatrixXd bestInverseNormalizedCovarianceMatrix = Eigen::MatrixXd::Constant( parameterVectorSize, parameterVectorSize, TUDAT_NAN );
453 
454  std::vector< Eigen::VectorXd > residualHistory;
455  std::vector< Eigen::VectorXd > parameterHistory;
456  std::vector< std::vector< std::map< TimeType, Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > > > > dynamicsHistoryPerIteration;
457  std::vector< std::vector< std::map< TimeType, Eigen::VectorXd > > > dependentVariableHistoryPerIteration;
458 
459  // Declare residual bookkeeping variables
460  std::vector< double > rmsResidualHistory;
461  double residualRms;
462 
463  // Declare variables to be used in loop.
464 
465  // Set current parameter estimate as both previous and current estimate
466  ParameterVectorType newParameterEstimate = currentParameterEstimate_ +
467  podInput->getInitialParameterDeviationEstimate( );
468  ParameterVectorType oldParameterEstimate = currentParameterEstimate_;
469 
470  int numberOfEstimatedParameters = parameterVectorSize;
471 
472  bool exceptionDuringPropagation = false, exceptionDuringInversion = false;
473  // Iterate until convergence (at least once)
474  int numberOfIterations = 0;
475  do
476  {
477 // try
478  {
479  // Re-integrate equations of motion and variational equations with new parameter estimate.
480  if( ( numberOfIterations > 0 ) ||( podInput->getReintegrateEquationsOnFirstIteration( ) ) )
481  {
482  resetParameterEstimate( newParameterEstimate, podInput->getReintegrateVariationalEquations( ) );
483  }
484 
485  if( podInput->getSaveStateHistoryForEachIteration( ) )
486  {
487  dynamicsHistoryPerIteration.push_back(
488  variationalEquationsSolver_->getDynamicsSimulatorBase( )->getEquationsOfMotionNumericalSolutionBase( ) );
489  dependentVariableHistoryPerIteration.push_back(
490  variationalEquationsSolver_->getDynamicsSimulatorBase( )->getDependentVariableNumericalSolutionBase( ) );
491  }
492  }
493 // catch( std::runtime_error& error )
494 // {
495 // std::cerr<<"Error when resetting parameters during parameter estimation: "<<std::endl<<
496 // error.what( )<<std::endl<<"Terminating estimation"<<std::endl;
497 // exceptionDuringPropagation = true;
498 // break;
499 // }
500 
501  oldParameterEstimate = newParameterEstimate;
502 
503  if( podInput->getPrintOutput( ) )
504  {
505  std::cout << "Calculating residuals and partials " << totalNumberOfObservations << std::endl;
506  }
507  // Calculate residuals and observation matrix for current parameter estimate.
508  std::pair< Eigen::VectorXd, Eigen::MatrixXd > residualsAndPartials;
509  calculateObservationMatrixAndResiduals(
510  podInput->getObservationsAndTimes( ), parameterVectorSize, totalNumberOfObservations, residualsAndPartials );
511 
512  Eigen::VectorXd transformationData = normalizeObservationMatrix( residualsAndPartials.second );
513 
514  Eigen::MatrixXd normalizedInverseAprioriCovarianceMatrix = Eigen::MatrixXd::Zero(
515  numberOfEstimatedParameters, numberOfEstimatedParameters );
516 
517  Eigen::MatrixXd inverseAPrioriCovariance = podInput->getInverseOfAprioriCovariance( );
518  for( int j = 0; j < numberOfEstimatedParameters; j++ )
519  {
520  for( int k = 0; k < numberOfEstimatedParameters; k++ )
521  {
522  normalizedInverseAprioriCovarianceMatrix( j, k ) = inverseAPrioriCovariance( j, k ) /
523  ( transformationData( j ) * transformationData( k ) );
524  }
525  }
526 
527  // Perform least squares calculation for correction to parameter vector.
528  std::pair< Eigen::VectorXd, Eigen::MatrixXd > leastSquaresOutput;
529  try
530  {
531  Eigen::MatrixXd constraintStateMultiplier;
532  Eigen::VectorXd constraintRightHandSide;
533  parametersToEstimate_->getConstraints( constraintStateMultiplier, constraintRightHandSide );
534  leastSquaresOutput =
535  std::move( linear_algebra::performLeastSquaresAdjustmentFromInformationMatrix(
536  residualsAndPartials.second.block( 0, 0, residualsAndPartials.second.rows( ), numberOfEstimatedParameters ),
537  residualsAndPartials.first, getConcatenatedWeightsVector( podInput->getWeightsMatrixDiagonals( ) ),
538  normalizedInverseAprioriCovarianceMatrix, 1, 1.0E8, constraintStateMultiplier, constraintRightHandSide ) );
539 
540  if( constraintStateMultiplier.rows( ) > 0 )
541  {
542  leastSquaresOutput.first.conservativeResize( parameterVectorSize );
543  }
544  }
545  catch( std::runtime_error& error )
546  {
547  std::cerr<<"Error when solving normal equations during parameter estimation: "<<std::endl<<error.what( )<<
548  std::endl<<"Terminating estimation"<<std::endl;
549  exceptionDuringInversion = true;
550  break;
551  }
552 
553  ParameterVectorType parameterAddition =
554  ( leastSquaresOutput.first.cwiseQuotient( transformationData.segment( 0, numberOfEstimatedParameters ) ) ).
555  template cast< ObservationScalarType >( );
556 
557  // std::cout<<"LSQ: "<<leastSquaresOutput.first<<std::endl<<
558  // transformationData.segment( 0, numberOfEstimatedParameters ).transpose( )<<std::endl;
559 
560 
561 
562  // Update value of parameter vector
563  newParameterEstimate = oldParameterEstimate + parameterAddition;
564  parametersToEstimate_->template resetParameterValues< ObservationScalarType >( newParameterEstimate );
565  newParameterEstimate = parametersToEstimate_->template getFullParameterValues< ObservationScalarType >( );
566 
567  if( podInput->getSaveResidualsAndParametersFromEachIteration( ) )
568  {
569  residualHistory.push_back( residualsAndPartials.first );
570  if( numberOfIterations == 0 )
571  {
572  parameterHistory.push_back( oldParameterEstimate.template cast< double >( ) );
573  }
574  parameterHistory.push_back( newParameterEstimate.template cast< double >( ) );
575  }
576 
577  oldParameterEstimate = newParameterEstimate;
578 
579  if( podInput->getPrintOutput( ) )
580  {
581  std::cout << "Parameter update" << parameterAddition.transpose( ) << std::endl;
582  }
583 
584  // Calculate mean residual for current iteration.
585  residualRms = linear_algebra::getVectorEntryRootMeanSquare( residualsAndPartials.first );
586 
587  rmsResidualHistory.push_back( residualRms );
588  if( podInput->getPrintOutput( ) )
589  {
590  std::cout << "Current residual: " << residualRms << std::endl;
591  }
592 
593  // If current iteration is better than previous one, update 'best' data.
594  if( residualRms < bestResidual || !( bestResidual == bestResidual ) )
595  {
596  bestResidual = residualRms;
597  bestParameterEstimate = std::move( oldParameterEstimate );
598  bestResiduals = std::move( residualsAndPartials.first );
599  if( podInput->getSaveInformationMatrix( ) )
600  {
601  bestInformationMatrix = std::move( residualsAndPartials.second );
602  }
603  bestWeightsMatrixDiagonal = std::move( getConcatenatedWeightsVector( podInput->getWeightsMatrixDiagonals( ) ) );
604  bestTransformationData = std::move( transformationData );
605  bestInverseNormalizedCovarianceMatrix = std::move( leastSquaresOutput.second );
606  }
607 
608 
609  // Increment number of iterations
610  numberOfIterations++;
611 
612  // Check for convergence
613  } while( convergenceChecker->isEstimationConverged( numberOfIterations, rmsResidualHistory ) == false );
614 
615  if( podInput->getPrintOutput( ) )
616  {
617  std::cout << "Final residual: " << bestResidual << std::endl;
618  }
619 
620 
621  std::shared_ptr< PodOutput< ObservationScalarType, TimeType > > podOutput =
622  std::make_shared< PodOutput< ObservationScalarType, TimeType > >(
623  bestParameterEstimate, bestResiduals, bestInformationMatrix, bestWeightsMatrixDiagonal, bestTransformationData,
624  bestInverseNormalizedCovarianceMatrix, bestResidual, residualHistory, parameterHistory, exceptionDuringInversion,
625  exceptionDuringPropagation );
626 
627  if( podInput->getSaveStateHistoryForEachIteration( ) )
628  {
629  podOutput->setStateHistories(
630  dynamicsHistoryPerIteration, dependentVariableHistoryPerIteration );
631  }
632 
633  return podOutput;
634  }

663  static PodInputType convertPodInput( const AlternativePodInputType& alternativePodInput )
664  {
665  // Declare return data structure.
666  PodInputType convertedObservationAndTimes;
667 
668  // Iterate over all observable types.
669  for( typename AlternativePodInputType::const_iterator inputIterator = alternativePodInput.begin( );
670  inputIterator != alternativePodInput.end( ); inputIterator++ )
671  {
672  // Declare data structure for converted measurement data at single observable type.
673  std::map< observation_models::LinkEnds, std::pair< ObservationVectorType, std::pair< std::vector< TimeType >, observation_models::LinkEndType > > > singleTypeObservations;
674 
675  // Iterate over all link ends in current observable types
676  for( typename std::map< observation_models::LinkEnds, std::pair< std::map< TimeType, ObservationScalarType >, observation_models::LinkEndType > >::const_iterator stationIterator =
677  inputIterator->second.begin( ); stationIterator != inputIterator->second.end( ); stationIterator++ )
678  {
679  // Initialize vector of observation times.
680  std::vector< TimeType > times;
681  times.resize( stationIterator->second.first.size( ) );
682 
683  // Initialize vector of observation values.
684  ObservationVectorType observations = ObservationVectorType::Zero( stationIterator->second.first.size( ) );
685 
686  // Iterate over all observations in input map, and split time and observation.
687  int counter = 0;
688  for( typename std::map< TimeType, ObservationScalarType >::const_iterator singleDataSetIterator = stationIterator->second.first.begin( );
689  singleDataSetIterator != stationIterator->second.first.end( ); singleDataSetIterator++ )
690  {
691  times[ counter ] = singleDataSetIterator->first;
692  observations( counter ) = singleDataSetIterator->second;
693  counter++;
