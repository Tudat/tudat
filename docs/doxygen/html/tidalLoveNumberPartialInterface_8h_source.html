290  std::vector< Eigen::Matrix< double, 2, Eigen::Dynamic > >& getCurrentParameterPartial(
291  const std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter,
292  const std::pair< int, int > maximumDegreeAndOrder )
293  {
294  if( currentDoubleParameterPartials_.count( std::make_pair( parameter, maximumDegreeAndOrder ) ) == 0 )
295  {
296  if( parameterDoublePartialFunctions_.count( std::make_pair( parameter, maximumDegreeAndOrder ) ) == 0 )
297  {
298  std::string errorMessage =
299  "Parameter of type " + std::to_string( parameter->getParameterName( ).first ) + ", " +
300  parameter->getParameterName( ).second.first + ", " +
301  parameter->getParameterName( ).second.second + " not found in list of existing partials";
302  throw std::runtime_error( errorMessage );
303  }
304  else
305  {
306  std::cerr << "Warning, double partial should already be calculatedz in Love number interface" << std::endl;
307  currentDoubleParameterPartials_[ std::make_pair( parameter, maximumDegreeAndOrder ) ] =
308  parameterDoublePartialFunctions_.at( std::make_pair( parameter, maximumDegreeAndOrder ) )( );
309  }
310  }
311  return currentDoubleParameterPartials_.at( std::make_pair( parameter, maximumDegreeAndOrder ) );
312  }
