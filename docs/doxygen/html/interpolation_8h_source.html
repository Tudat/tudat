139 template< typename I, typename D >
140 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< DataMapSettings< I, D > >& dataMapSettings )
141 {
142  if ( ! dataMapSettings )
143  {
144  return;
145  }
146  using namespace json_interface;
147  using K = Keys::Interpolation::DataMap;
148 
149  std::shared_ptr< HermiteDataSettings< I, D > > hermiteDataSettings =
150  std::dynamic_pointer_cast< HermiteDataSettings< I, D > >( dataMapSettings );
151  if ( hermiteDataSettings )
152  {
153  jsonObject[ K::map ] = hermiteDataSettings->getDataMap( );
154  jsonObject[ K::dependentVariableFirstDerivativeValues ] =
155  hermiteDataSettings->firstDerivativeOfDependentVariables_;
156  return;
157  }
158 
159  std::shared_ptr< FromFileDataMapSettings< D > > fromFileDataMapSettings =
160  std::dynamic_pointer_cast< FromFileDataMapSettings< D > >( dataMapSettings );
161  if ( fromFileDataMapSettings )
162  {
163  jsonObject[ K::file ] = boost::filesystem::path( fromFileDataMapSettings->relativeFilePath_ );
164  return;
165  }
166 
167  std::shared_ptr< IndependentDependentDataMapSettings< I, D > > independentDependentDataMapSettings =
168  std::dynamic_pointer_cast< IndependentDependentDataMapSettings< I, D > >( dataMapSettings );
169  if ( independentDependentDataMapSettings )
170  {
171  jsonObject[ K::independentVariableValues ] = independentDependentDataMapSettings->independentVariableValues_;
172  jsonObject[ K::dependentVariableValues ] = independentDependentDataMapSettings->dependentVariableValues_;
173  return;
174  }
175 
176  jsonObject[ K::map ] = dataMapSettings->getDataMap( );
177 }

180 template< typename I, typename D >
181 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< DataMapSettings< I, D > >& dataMapSettings )
182 {
183  using namespace json_interface;
184  using K = Keys::Interpolation::DataMap;
185 
186  if ( isDefined( jsonObject, K::dependentVariableFirstDerivativeValues ) )
187  {
188  dataMapSettings = std::make_shared< HermiteDataSettings< I, D > >(
189  getValue< std::map< I, D > >( jsonObject, K::map ),
190  getValue< std::vector< D > >( jsonObject, K::dependentVariableFirstDerivativeValues ) );
191  return;
192  }
193 
194  if ( isDefined( jsonObject, K::file ) )
195  {
196  dataMapSettings = std::make_shared< FromFileDataMapSettings< D > >(
197  getValue< boost::filesystem::path >( jsonObject, K::file ).string( ) );
198  return;
199  }
200 
201  if ( isDefined( jsonObject, K::independentVariableValues ) || isDefined( jsonObject, K::dependentVariableValues ) )
202  {
203  dataMapSettings = std::make_shared< IndependentDependentDataMapSettings< I, D > >(
204  getValue< std::vector< I > >( jsonObject, K::independentVariableValues ),
205  getValue< std::vector< D > >( jsonObject, K::dependentVariableValues ) );
206  return;
207  }
208 
209  dataMapSettings = std::make_shared< DataMapSettings< I, D > >(
210  getValue< std::map< I, D > >( jsonObject, K::map ) );
211  return;
212 }
