201  void checkBoundaryCase(
202  DependentVariableType& dependentVariable, bool& useValue,
203  const IndependentVariableType& targetIndependentVariable )
204  {
205  // If extrapolation outside domain is not allowed
206  if ( boundaryHandling_ != extrapolate_at_boundary )
207  {
208  // If independent variable is out of range
209  int isAtBoundary = checkInterpolationBoundary( targetIndependentVariable );
210  if ( isAtBoundary != 0 )
211  {
212  // Select course of action based on boundary handling method selected
213  switch ( boundaryHandling_ )
214  {
215  case throw_exception_at_boundary:
216  {
217  // Throw exception
218  std::string errorMessage = "Error in interpolator, requesting data point outside of boundaries, requested data at " +
219  boost::lexical_cast< std::string >( targetIndependentVariable ) + " but limit values are " +
220  boost::lexical_cast< std::string >( independentValues_.front( ) ) + " and " +
221  boost::lexical_cast< std::string >( independentValues_.back( ) );
222  throw std::runtime_error( errorMessage );
223  break;
224  }
225  case extrapolate_at_boundary_with_warning:
226  {
227  // Warn user
228  std::string errorMessage = "Warning in interpolator, requesting data point outside of boundaries, requested data at " +
229  boost::lexical_cast< std::string >( targetIndependentVariable ) + " but limit values are " +
230  boost::lexical_cast< std::string >( independentValues_.front( ) ) + " and " +
231  boost::lexical_cast< std::string >( independentValues_.back( ) ) + ", applying extrapolation instead.";
232  std::cerr << errorMessage << std::endl;
233  break;
234  }
235  case use_boundary_value:
236  case use_boundary_value_with_warning:
237  {
238  // Warn user, if requested
239  if ( boundaryHandling_ == use_boundary_value_with_warning )
240  {
241  std::string errorMessage = "Warning in interpolator, requesting data point outside of boundaries, requested data at " +
242  boost::lexical_cast< std::string >( targetIndependentVariable ) + " but limit values are " +
243  boost::lexical_cast< std::string >( independentValues_.front( ) ) + " and " +
244  boost::lexical_cast< std::string >( independentValues_.back( ) ) + ", taking boundary value instead.";
245  std::cerr << errorMessage << std::endl;
246  }
247 
248  // Get boundary value
249  useValue = true;
250  if ( isAtBoundary == -1 )
251  {
252  dependentVariable = dependentValues_.front( );
253  }
254  else if ( isAtBoundary == 1 )
255  {
256  dependentVariable = dependentValues_.back( );
257  }
258  else
259  {
260  throw std::runtime_error( "Error when checking interpolation boundary, inconsistent data encountered" );
261  }
262  break;
263  }
264  case use_default_value:
265  case use_default_value_with_warning:
266  {
267  // Warn user, if requested
268  if ( boundaryHandling_ == use_default_value_with_warning )
269  {
270  std::string errorMessage = "Warning in interpolator, requesting data point outside of boundaries, requested data at " +
271  boost::lexical_cast< std::string >( targetIndependentVariable ) + " but limit values are " +
272  boost::lexical_cast< std::string >( independentValues_.front( ) ) + " and " +
273  boost::lexical_cast< std::string >( independentValues_.back( ) ) + ", taking default value instead.";
274  std::cerr << errorMessage << std::endl;
275  }
276 
277  // Get default value
278  useValue = true;
279  if ( isAtBoundary == -1 )
280  {
281  dependentVariable = defaultExtrapolationValue_.first;
282  }
283  else if ( isAtBoundary == 1 )
284  {
285  dependentVariable = defaultExtrapolationValue_.second;
286  }
287  else
288  {
289  throw std::runtime_error( "Error when checking interpolation boundary, inconsistent data encountered" );
290  }
291  break;
292  }
293  default:
294  throw std::runtime_error( "Error when checking interpolation boundary, boundary handling method not recognized." );
295  }
296  }
297  }

308  void makeLookupScheme( const AvailableLookupScheme selectedScheme )
309  {
310  // Find which type of scheme is used.
311  switch ( selectedScheme )
312  {
313  case binarySearch:
314  {
315  // Create binary search look up scheme.
316  lookUpScheme_ = std::shared_ptr< LookUpScheme< IndependentVariableType > >
317  ( new BinarySearchLookupScheme< IndependentVariableType >
318  ( independentValues_ ) );
319  break;
320  }
321  case huntingAlgorithm:
322  {
323  // Create hunting scheme, which uses an intial guess from previous look-ups.
324  lookUpScheme_ = std::shared_ptr< LookUpScheme< IndependentVariableType > >
325  ( new HuntingAlgorithmLookupScheme< IndependentVariableType >
326  ( independentValues_ ) );
327  break;
328  }
329  default:
330  throw std::runtime_error( "Warning: lookup scheme not found when making scheme for 1-D interpolator" );
331  }
332  }
