120  TimeType getCurrentTime(
121  const basic_astrodynamics::TimeScales inputScale, const basic_astrodynamics::TimeScales outputScale,
122  const TimeType& inputTimeValue, const Eigen::Vector3d& earthFixedPosition = Eigen::Vector3d::Zero( ) )
123  {
124  TimeType convertedTime;
125 
126  // Check if any conversion should take place.
127  if( inputScale == outputScale )
128  {
129  convertedTime = inputTimeValue;

133  // Check if update is required
134  if( !( static_cast< TimeType >( getCurrentTimeList< TimeType >( ).getTimeValue( inputScale ) ) ==
135  static_cast< TimeType >( inputTimeValue ) ) ||
136  !( getPreviousGroundStationPosition< TimeType >( ) == earthFixedPosition ) )
137  {

164  template< typename TimeType >
165  void updateTimes( const basic_astrodynamics::TimeScales inputScale, const TimeType& inputTimeValue,
166  const Eigen::Vector3d& earthFixedPosition )
167  {
168  // Retrieve CurrentTimes object that is to be updated
169  CurrentTimes< TimeType >& timesToUpdate = getCurrentTimeList< TimeType >( );
170 
171  // Convert position to SOFA input valies
172  setCurrentGroundStation< TimeType >( earthFixedPosition );
173 
174  double siteLongitude = std::atan2( earthFixedPosition.y( ), earthFixedPosition.x( ) );
175  double distanceFromSpinAxis = std::sqrt( earthFixedPosition.x( ) * earthFixedPosition.x( ) +
176  earthFixedPosition.y( ) * earthFixedPosition.y( ) );
177  double distanceFromEquatorialPlane = earthFixedPosition.z( );
178 
179  TimeType tdbMinusTt;
180 
181  // Check input type, and call conversion functions accordingly
182  switch( inputScale )
183  {
184  case basic_astrodynamics::tdb_scale:
185  timesToUpdate.tdb = inputTimeValue;
186  tdbMinusTt = static_cast< TimeType >( sofa_interface::getTDBminusTT(
187  inputTimeValue, siteLongitude, distanceFromSpinAxis, distanceFromEquatorialPlane ) );
188  timesToUpdate.tt = timesToUpdate.tdb - tdbMinusTt;
189  timesToUpdate.tai = basic_astrodynamics::convertTTtoTAI< TimeType >( timesToUpdate.tt );
190 
191  calculateUniversalTimes< TimeType >( );
192  break;
193 
194  case basic_astrodynamics::tt_scale:
195  timesToUpdate.tt = inputTimeValue;
196  tdbMinusTt = static_cast< TimeType >( sofa_interface::getTDBminusTT(
197  inputTimeValue, siteLongitude, distanceFromSpinAxis, distanceFromEquatorialPlane ) );
198  timesToUpdate.tdb = timesToUpdate.tt + tdbMinusTt;
199  timesToUpdate.tai = basic_astrodynamics::convertTTtoTAI< TimeType >( timesToUpdate.tt );
200 
201  calculateUniversalTimes< TimeType >( );
202  break;
203 
204  case basic_astrodynamics::tai_scale:
205  timesToUpdate.tai = inputTimeValue;
206  timesToUpdate.tt = basic_astrodynamics::convertTAItoTT< TimeType >( timesToUpdate.tai );
207  tdbMinusTt = static_cast< TimeType >( sofa_interface::getTDBminusTT(
208  timesToUpdate.tt, siteLongitude, distanceFromSpinAxis, distanceFromEquatorialPlane ) );
209  timesToUpdate.tdb = timesToUpdate.tt + tdbMinusTt;
210 
211  calculateUniversalTimes< TimeType >( );
212 
213  break;
214  case basic_astrodynamics::utc_scale:
215  timesToUpdate.utc = inputTimeValue;
216  timesToUpdate.tai = sofa_interface::convertUTCtoTAI< TimeType >( timesToUpdate.utc );
217 
218  timesToUpdate.tt = basic_astrodynamics::convertTAItoTT< TimeType >( timesToUpdate.tai );
219  tdbMinusTt = static_cast< TimeType >( sofa_interface::getTDBminusTT(
220  timesToUpdate.tt, siteLongitude, distanceFromSpinAxis, distanceFromEquatorialPlane ) );
221  timesToUpdate.tdb = timesToUpdate.tt + tdbMinusTt;
222 
223  timesToUpdate.ut1 = static_cast< TimeType >( dailyUtcUt1CorrectionInterpolator_->interpolate( timesToUpdate.utc ) )
224  + timesToUpdate.utc;
225 
226  timesToUpdate.ut1 += static_cast< TimeType >(
227  shortPeriodUt1CorrectionCalculator_->getCorrections( timesToUpdate.tdb ) );
228 
229  break;
230  case basic_astrodynamics::ut1_scale:
231 
232  timesToUpdate.ut1 = inputTimeValue;
233  timesToUpdate.utc = timesToUpdate.ut1 -
234  static_cast< TimeType >( dailyUtcUt1CorrectionInterpolator_->interpolate( timesToUpdate.ut1 ) );
235  timesToUpdate.utc -=
236  static_cast< TimeType >( shortPeriodUt1CorrectionCalculator_->getCorrections( timesToUpdate.utc ) );
237  timesToUpdate.tai = sofa_interface::convertUTCtoTAI< TimeType >( timesToUpdate.utc );
238  timesToUpdate.tt = basic_astrodynamics::convertTAItoTT< TimeType >( timesToUpdate.tai );
239  tdbMinusTt = static_cast< TimeType >( sofa_interface::getTDBminusTT(
240  timesToUpdate.tt, siteLongitude, distanceFromSpinAxis, distanceFromEquatorialPlane ) );
241  timesToUpdate.tdb = timesToUpdate.tt + tdbMinusTt;
242 
243  // Iterate conversion.
244  timesToUpdate.utc = timesToUpdate.ut1 -
245  static_cast< TimeType >( dailyUtcUt1CorrectionInterpolator_->interpolate( timesToUpdate.utc ) );
246  timesToUpdate.utc -= static_cast< TimeType >(
247  shortPeriodUt1CorrectionCalculator_->getCorrections( timesToUpdate.tt ) );
248  timesToUpdate.tai = sofa_interface::convertUTCtoTAI< TimeType >( timesToUpdate.utc );
249  timesToUpdate.tt = basic_astrodynamics::convertTAItoTT< TimeType >( timesToUpdate.tai );
250  tdbMinusTt = sofa_interface::getTDBminusTT(
251  timesToUpdate.tt, siteLongitude, distanceFromSpinAxis, distanceFromEquatorialPlane );
252  timesToUpdate.tdb = timesToUpdate.tt + tdbMinusTt;
253 
254  break;
255 
256  default:
257  throw std::runtime_error( "Error when performing Earth time scales, input time not recognized" );
258  break;
259  }
