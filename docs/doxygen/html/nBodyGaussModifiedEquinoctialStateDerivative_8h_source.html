71  {
72  originalAccelerationModelsPerBody_ = this->accelerationModelsPerBody_ ;
73 
74  // Remove central gravitational acceleration from list of accelerations that is to be evaluated
75  centralBodyGravitationalParameters_ =
76  removeCentralGravityAccelerations(
77  centralBodyData->getCentralBodies( ), this->bodiesToBeIntegratedNumerically_,
78  this->accelerationModelsPerBody_ );
79  this->createAccelerationModelList( );
80 
81  // Check if singularity in equations of motion should be palced at 0 or 180 degrees inclination
82  flipSingularities_.resize( bodiesToIntegrate.size( ) );
83  for( unsigned int i = 0; i < initialKeplerElements.size( ); i++ )
84  {
85  if( initialKeplerElements.at( i )( orbital_element_conversions::inclinationIndex ) > mathematical_constants::PI / 2.0 )
86  {
87  flipSingularities_[ i ] = false;
88  std::cerr << "Warning when using Gauss-MEE propagation, body " << bodiesToIntegrate.at( i ) << " has inclination of "
89  << initialKeplerElements.at( i )( orbital_element_conversions::inclinationIndex ) * 180.0 /
90  mathematical_constants::PI << " degrees, but propagator has singularity at i=180 degrees" << std::endl;
91  }
92  else
93  {
94  flipSingularities_[ i ] = false;
95  }
96 
97  if( initialKeplerElements.at( i )( orbital_element_conversions::inclinationIndex ) >
98  mathematical_constants::PI )
99  {
100  throw std::runtime_error( "Error when setting N Body Gauss MEE propagator, initial inclination of body is larger than pi." );
101  }
102  }

122  void calculateSystemStateDerivative(
123  const TimeType time, const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
124  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
125  {
126  // Get total inertial accelerations acting on bodies
127  stateDerivative.setZero( );
128  this->sumStateDerivativeContributions( stateOfSystemToBeIntegrated, stateDerivative, false );
129 
130  // Compute RSW accelerations for each body, and evaluate MEE Gauss equations.
131  Eigen::Vector3d currentAccelerationInRswFrame;
132  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
133  {
134  currentAccelerationInRswFrame = reference_frames::getInertialToRswSatelliteCenteredFrameRotationMatrix(
135  currentCartesianLocalSolution_.segment( i * 6, 6 ).template cast< double >( ) ) *
136  stateDerivative.block( i * 6 + 3, 0, 3, 1 ).template cast< double >( );
137 
138  stateDerivative.block( i * 6, 0, 6, 1 ) = computeGaussPlanetaryEquationsForModifiedEquinoctialElements(
139  stateOfSystemToBeIntegrated.block( i * 6, 0, 6, 1 ).template cast< double >( ), currentAccelerationInRswFrame,
140  centralBodyGravitationalParameters_.at( i )( ) ).template cast< StateScalarType >( );
141  }
