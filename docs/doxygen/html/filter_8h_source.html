307  void revertToPreviousTimeStep( const double timeToBeRemoved )
308  {
309  // Revert to previous time
310  currentTime_ -= filteringStepSize_;
311 
312  // Erase state estimate corresponding to current time
313  if ( historyOfStateEstimates_.count( timeToBeRemoved ) != 0 )
314  {
315  historyOfStateEstimates_.erase( timeToBeRemoved );
316  }
317  aPosterioriStateEstimate_ = historyOfStateEstimates_.rbegin( )->second;
318 
319  // Erase covariance estimate corresponding to current time
320  if ( historyOfCovarianceEstimates_.count( timeToBeRemoved ) != 0 )
321  {
322  historyOfCovarianceEstimates_.erase( timeToBeRemoved );
323  }
324  aPosterioriCovarianceEstimate_ = historyOfCovarianceEstimates_.rbegin( )->second;
325 
326  // Erase last noise entries
327  if ( !systemNoiseHistory_.empty( ) )
328  {
329  systemNoiseHistory_.pop_back( );
330  }
331  if ( !measurementNoiseHistory_.empty( ) )
332  {
333  measurementNoiseHistory_.pop_back( );
334  }
335 
336  // Revert elements specific to each filter
337  specificRevertToPreviousTimeStep( timeToBeRemoved );
338  }

482  void generateNoiseDistributions( )
483  {
484  using namespace tudat::statistics;
485 
486  // Create system noise
487  for ( unsigned int i = 0; i < systemUncertainty_.rows( ); i++ )
488  {
489  if ( static_cast< double >( systemUncertainty_( i, i ) ) != 0.0 )
490  {
491  systemNoiseDistribution_.push_back(
492  createBoostContinuousRandomVariableGenerator(
493  normal_boost_distribution, { 0.0, static_cast< double >(
494  std::sqrt( systemUncertainty_( i, i ) ) ) },
495  static_cast< double >( i ) ) );
496  }
497  else
498  {
499  systemNoiseDistribution_.push_back( nullptr );
500  }
501  }
502 
503  // Create measurement noise
504  for ( unsigned int i = 0; i < measurementUncertainty_.rows( ); i++ )
505  {
506  if ( static_cast< double >( measurementUncertainty_( i, i ) ) != 0.0 )
507  {
508  measurementNoiseDistribution_.push_back(
509  createBoostContinuousRandomVariableGenerator(
510  normal_boost_distribution, { 0.0, static_cast< double >(
511  std::sqrt( measurementUncertainty_( i, i ) ) ) },
512  static_cast< double >( systemUncertainty_.rows( ) + i ) ) );
513  }
514  else
515  {
516  measurementNoiseDistribution_.push_back( nullptr );
517  }
518  }
519  }

528  void generateNumericalIntegrator( const std::shared_ptr< IntegratorSettings > integratorSettings )
529  {
530  // Check that integration time-step matches filtering time-step
531  if ( filteringStepSize_ != integratorSettings->initialTimeStep_ )
532  {
533  throw std::runtime_error( "Error while setting up filter. The filtering and integration step sizes do not match." );
534  }
535 
536  // Generate integrator
537  switch ( integratorSettings->integratorType_ )
538  {
539  case numerical_integrators::euler:
540  case numerical_integrators::rungeKutta4:
541  {
542  integrator_ = numerical_integrators::createIntegrator< IndependentVariableType, DependentVector >(
543  systemFunction_, aPosterioriStateEstimate_, integratorSettings );
544  break;
545  }
546  case numerical_integrators::rungeKuttaVariableStepSize:
547  {
548  // Warn user of changes that will be made
549  std::cerr << "Warning in setting up filter. Integrator requested is variable step-size, but only constant "
550  "step-size integrators are supported. Step-size control will be turned off." << std::endl;
551 
552  // Create integrator object
553  integrator_ = numerical_integrators::createIntegrator< IndependentVariableType, DependentVector >(
554  systemFunction_, aPosterioriStateEstimate_, integratorSettings );
555 
556  // Turn off step-size control
557  integrator_->setStepSizeControl( false );
558  break;
559  }
560  default:
561  throw std::runtime_error( "Error in setting up filter. Only Euler and Runge-Kutta integrators are supported." );
562  }
563  }
