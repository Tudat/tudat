60 template< typename ParameterType >
61 std::map< observation_models::LinkEnds,
62 std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > >
63 createDifferencedOneWayRangeRatePartials(
64  const std::vector< observation_models::LinkEnds > linkEnds,
65  const simulation_setup::SystemOfBodies bodies,
66  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
67  const PerLinkEndPerLightTimeSolutionCorrections& lightTimeCorrections =
68  PerLinkEndPerLightTimeSolutionCorrections( ) )
69 {
70  using namespace observation_partials;
71 
72 
73  std::map< observation_models::LinkEnds,
74  std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > > rangeRatePartials;
75 
76  // Retrieve separate light-time corrections for arc start and arc end.
77  std::pair< PerLinkEndPerLightTimeSolutionCorrections, PerLinkEndPerLightTimeSolutionCorrections > splitLightTimeCorrections =
78  splitOneWayRangeRateLightTimeCorrectionsBetweenArcs( lightTimeCorrections );
79 
80  // Create one way range partials for link at start of arc.
81  std::map< observation_models::LinkEnds,
82  std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > > arcStartPartials
83  = createOneWayRangePartials( linkEnds, bodies, parametersToEstimate, splitLightTimeCorrections.first );
84 
85  // Create one way range partials for link at end of arc.
86  std::map< observation_models::LinkEnds,
87  std::pair< SingleLinkObservationPartialList , std::shared_ptr< PositionPartialScaling > > > arcEndPartials
88  = createOneWayRangePartials( linkEnds, bodies, parametersToEstimate, splitLightTimeCorrections.second );
89 
90  // Check output consistency
91  if( arcStartPartials.size( ) != arcEndPartials.size( ) )
92  {
93  throw std::runtime_error(
94  "Error when making differenced one way range rate partials, arc start and end partial set size is not consistent" );
95  }
96 
97  // Create iterators over created partial links.
98  std::map< observation_models::LinkEnds,
99  std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > >::iterator
100  arcStartIterator = arcStartPartials.begin( );
101  std::map< observation_models::LinkEnds,
102  std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > >::iterator
103  arcEndIterator = arcEndPartials.begin( );
104 
105  // Pre-create iterators over one-way range partials for the single sets of link ends at the start and end of the arc.
106  SingleLinkObservationPartialList::iterator currentLinkEndArcStartIterator;
107  SingleLinkObservationPartialList::iterator currentLinkEndArcEndIterator;
108 
109  // Iterate over all link ends.
110  for( unsigned int i = 0; i < arcStartPartials.size( ); i++ )
111  {
112  SingleLinkObservationPartialList currentRangeRatePartialList;
113 
114  if( arcEndIterator->first != arcStartIterator->first )
115  {
116  throw std::runtime_error(
117  "Error when making differenced one way range rate partials, arc start and end partial link ends not consistent" );
118  }
119 
120  // Set iterators over set of partials for current link ends.
121  currentLinkEndArcStartIterator = arcStartIterator->second.first.begin( );
122  currentLinkEndArcEndIterator = arcEndIterator->second.first.begin( );
123 
124  // Iterate over all one-way range partials and create one-way range rate partial from them.
125  for( unsigned int j = 0; j < arcStartIterator->second.first.size( ); j++ )
126  {
127  if( currentLinkEndArcStartIterator->second->getParameterIdentifier( ) !=
128  currentLinkEndArcEndIterator->second->getParameterIdentifier( ) )
129  {
130  throw std::runtime_error(
131  "Error when making differenced one way range partial, parameters did not match" );
132  }
133  else
134  {
135  // Create range rate partial.
136  currentRangeRatePartialList[ currentLinkEndArcStartIterator->first ] =
137  std::make_shared< DifferencedOneWayRangeRatePartial >(
138  currentLinkEndArcStartIterator->second->getParameterIdentifier( ),
139  currentLinkEndArcStartIterator->second,
140  currentLinkEndArcEndIterator->second );
141  }
142 
143  // Increment range partial iterators.
144  currentLinkEndArcStartIterator++;
145  currentLinkEndArcEndIterator++;
146  }
147 
148  // Set range partials for current link end.
149  rangeRatePartials[ arcStartIterator->first ] = std::make_pair(
150  currentRangeRatePartialList, std::make_shared< OneWayRangeRateScaling >(
151  std::dynamic_pointer_cast< OneWayRangeScaling >( arcStartIterator->second.second ),
152  std::dynamic_pointer_cast< OneWayRangeScaling >( arcEndIterator->second.second ) ) );
153 
154  // Increment iterators to next link ends.
155  arcStartIterator++;
156  arcEndIterator++;
157  }
158 
159  // Return partials
160  return rangeRatePartials;
161 }
