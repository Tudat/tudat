56 template< typename EigenVectorType = Eigen::VectorXd >
57 std::map< typename EigenVectorType::Scalar, EigenVectorType > readEigenVectorMapFromFile(
58  const std::string& relativePath,
59  const std::string& separators = "\t ;,",
60  const std::string& skipLinesCharacter = "%" )
61 {
62  // Load data into matrix
63  Eigen::Matrix< typename EigenVectorType::Scalar, Eigen::Dynamic, Eigen::Dynamic > matrix =
64  readMatrixFromFile< typename EigenVectorType::Scalar >( relativePath, separators, skipLinesCharacter );
65 
66  // Fill map using Eigen matrix
67  std::map< typename EigenVectorType::Scalar, EigenVectorType > map;
68  for ( int i = 0; i < matrix.rows( ); i++ )
69  {
70  const Eigen::Matrix< typename EigenVectorType::Scalar, Eigen::Dynamic, 1 > dynamicSizeColumnVector =
71  matrix.row( i ).rightCols( matrix.cols( ) - 1 ).transpose( );
72  EigenVectorType columnVector;
73  if ( columnVector.rows( ) == 0 )
74  {
75  columnVector = EigenVectorType( dynamicSizeColumnVector.rows( ) );
76  }
77  else if ( dynamicSizeColumnVector.rows( ) != columnVector.rows( ) )
78  {
79  std::cerr << "The size of the vectors in \"" << relativePath << "\" (" << dynamicSizeColumnVector.rows( )
80  << ") is not consistent with the requested size (" << columnVector.rows( ) << ")." << std::endl;
81  throw;
82  }
83  for ( int j = 0; j < columnVector.rows( ); j++ )
84  {
85  columnVector( j ) = dynamicSizeColumnVector( j );
86  }
87  map[ matrix( i, 0 ) ] = columnVector;
88  }
89  return map;
90 }

104 template< typename KeyType, typename ScalarValueType >
105 std::map< KeyType, std::vector< ScalarValueType > > readStlVectorMapFromFile(
106  const std::string& relativePath, const std::string& separators = "\t ;,",
107  const std::string& skipLinesCharacter = "%" )
108 {
109  // Open input and output.
110  std::fstream file( relativePath.c_str( ), std::ios::in );
111  if ( file.fail( ) )
112  {
113  throw std::runtime_error( "Data file could not be opened: " + relativePath );
114  }
115 
116  std::stringstream filteredStream( std::ios::in | std::ios::out );
117  {
118  // Filter the file stream. This needs to be in its own scope, because filtering_stream::
119  // flush( ) does not work if the underlying end point is a stringstream, so the flush has
120  // to be forced by letting the filtering_stream go out of scope.
121  boost::iostreams::filtering_ostream filterProcessor;
122  for ( unsigned int i = 0; i < skipLinesCharacter.size( ); i++ )
123  {
124  // Remove all comments from the stream.
125  filterProcessor.push( input_output::stream_filters::RemoveComment(
126  skipLinesCharacter[ i ], true ) );
127  }
128 
129  // Add the output to the filter chain.
130  filterProcessor.push( filteredStream );
131 
132  // Copy the input to the filter.
133  boost::iostreams::copy( file, filterProcessor );
134  }
135 
136  // Seek stream back to start.
137  filteredStream.seekg( 0, std::ios::beg );
138 
139  // Read the filtered stream into lines.
140  std::vector< std::string > lines_;
141  while ( !filteredStream.eof( ) )
142  {
143  std::string line_;
144  getline( filteredStream, line_ );
145  if ( !line_.empty( ) )
146  {
147  boost::trim_all( line_ );
148  lines_.push_back( line_ );
149  }
150  }
151 
152  // Initialize the map.
153  std::map< KeyType, std::vector< ScalarValueType > > dataMap_;
154 
155  // If there are no lines, return an empty matrix.
156  if ( lines_.empty( ) )
157  {
158  return dataMap_;
159  }
160 
161  const std::string realSeparators = std::string( separators ) + " ";
162 
163  for ( unsigned int rowIndex = 0; rowIndex < lines_.size( ); rowIndex++ )
164  {
165  // Determine the number of columns from.
166  std::vector< std::string > lineSplit_;
167  boost::algorithm::split( lineSplit_, lines_[ rowIndex ], boost::is_any_of( realSeparators ),
168  boost::algorithm::token_compress_on );
169 
170  // Determine key and put single line entries into vector.
171  KeyType key;
172  std::vector< ScalarValueType > values;
173  for ( unsigned int columnIndex = 0; columnIndex < lineSplit_.size( ); columnIndex++ )
174  {
175  boost::trim( lineSplit_.at( columnIndex ) );
176  if ( columnIndex == 0 )
177  {
178  key = boost::lexical_cast< KeyType >( lineSplit_.at( columnIndex ) );
179  }
180  else
181  {
182  values.push_back( boost::lexical_cast< ScalarValueType >( lineSplit_.at( columnIndex ) ) );
183  }
184  }
185 
186  dataMap_[ key ] = values;
187  }
188 
189  return dataMap_;
190 }
