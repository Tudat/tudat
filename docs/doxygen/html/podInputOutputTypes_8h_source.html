370  EstimationConvergenceChecker(
371  const unsigned int maximumNumberOfIterations = 5,
372  const double minimumResidualChange = 0.0,
373  const double minimumResidual = 1.0E-20,
374  const int numberOfIterationsWithoutImprovement = 2 ):
375  maximumNumberOfIterations_( maximumNumberOfIterations ), minimumResidualChange_( minimumResidualChange ),
376  minimumResidual_( minimumResidual ),
377  numberOfIterationsWithoutImprovement_( numberOfIterationsWithoutImprovement )
378  { }
379 
381 
387  bool isEstimationConverged( const int numberOfIterations, const std::vector< double > rmsResidualHistory )
388  {
389  bool isConverged = 0;
390  if( numberOfIterations >= maximumNumberOfIterations_ )
391  {
392  std::cout << "Maximum number of iterations reached" << std::endl;
393  isConverged = 1;
394  }
395  if( rmsResidualHistory[ rmsResidualHistory.size( ) - 1 ] < minimumResidual_ )
396  {
397  std::cout << "Required residual level achieved" << std::endl;
398  isConverged = 1;
399  }
400  if( ( std::distance( rmsResidualHistory.begin( ), std::max_element(
401  rmsResidualHistory.begin( ), rmsResidualHistory.end( ) ) ) - rmsResidualHistory.size( ) ) <
402  numberOfIterationsWithoutImprovement_ )
403  {
404  std::cout << "Too many iterations without parameter improvement" << std::endl;
405  isConverged = 1;
406  }
407  if( rmsResidualHistory.size( ) > 1 )
408  {
409  if( std::fabs( rmsResidualHistory.at( rmsResidualHistory.size( ) - 1 ) -
410  rmsResidualHistory.at( rmsResidualHistory.size( ) - 2 ) ) < minimumResidualChange_ )
411  {
412  isConverged = 1;
413  }
414  }
415  return isConverged;
416  }
417 protected:
418 
420  int maximumNumberOfIterations_;
421 
423  double minimumResidualChange_;
424 
426  double minimumResidual_;
427 
429  unsigned int numberOfIterationsWithoutImprovement_;
430 };

483  Eigen::MatrixXd getUnnormalizedInverseCovarianceMatrix( )
484  {
485 
486  Eigen::MatrixXd inverseUnnormalizedCovarianceMatrix = inverseNormalizedCovarianceMatrix_;
487 
488  for( int i = 0; i < informationMatrixTransformationDiagonal_.rows( ); i++ )
489  {
490  for( int j = 0; j < informationMatrixTransformationDiagonal_.rows( ); j++ )
491  {
492  inverseUnnormalizedCovarianceMatrix( i, j ) *=
493  informationMatrixTransformationDiagonal_( i ) * informationMatrixTransformationDiagonal_( j );
494  }
495  }
496 
497  return inverseUnnormalizedCovarianceMatrix;
498  }
499 
501 
505  Eigen::MatrixXd getUnnormalizedCovarianceMatrix( )
506  {
507  Eigen::MatrixXd unnormalizedCovarianceMatrix = inverseNormalizedCovarianceMatrix_.inverse( );
508 
509  for( int i = 0; i < informationMatrixTransformationDiagonal_.rows( ); i++ )
510  {
511  for( int j = 0; j < informationMatrixTransformationDiagonal_.rows( ); j++ )
512  {
513  unnormalizedCovarianceMatrix( i, j ) /=
514  informationMatrixTransformationDiagonal_( i ) * informationMatrixTransformationDiagonal_( j );
515  }
516  }
517 
518  return unnormalizedCovarianceMatrix;

526  Eigen::MatrixXd getUnnormalizedPartialDerivatives( )
527  {
528  Eigen::MatrixXd unnormalizedPartialDerivatives = Eigen::MatrixXd::Zero(
529  normalizedInformationMatrix_.rows( ), normalizedInformationMatrix_.cols( ) );
530 
531  for( int i = 0; i < informationMatrixTransformationDiagonal_.rows( ); i++ )
532  {
533  unnormalizedPartialDerivatives.block( 0, i, normalizedInformationMatrix_.rows( ), 1 ) =
534  normalizedInformationMatrix_.block( 0, i, normalizedInformationMatrix_.rows( ), 1 ) *
535  informationMatrixTransformationDiagonal_( i );
