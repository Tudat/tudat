101  static void writeMultiArrayAndIndependentVariablesToFiles( const std::string& fileName,
102  const std::vector< int >& coefficientIndices,
103  const std::vector< std::vector< double > >& independentVariables,
104  const boost::multi_array< Eigen::Matrix< double, NumberOfCoefficients,
105  1 >, 1 >& dependentVariables )
106  {
107  // Get actual number of coefficients
108  // The user may only want to save some coefficients (e.g., (s)he may want to disregard CS)
109  unsigned int actualNumberOfCoefficients = coefficientIndices.size( );
110 
111  // Check consistency of inputs
112  // Check that number of coefficients is not too large
113  if ( actualNumberOfCoefficients > NumberOfCoefficients )
114  {
115  throw std::runtime_error( "Error while saving one-dimensional aerodynamic coefficients to file. "
116  "Number of coefficients does not match number of input files." );
117  }
118  // Check that number of independent variables matches vector size
119  if ( independentVariables.size( ) != 1 )
120  {
121  throw std::runtime_error( "Error while saving one-dimensional aerodynamic coefficients to file. "
122  "Number of independent variables does not match size of independent variables vector." );
123  }
124 
125  // Create directory (if it does not exist)
126  if ( !boost::filesystem::exists( boost::filesystem::path( fileName ).parent_path( ) ) )
127  {
128  boost::filesystem::create_directories( boost::filesystem::path( fileName ).parent_path( ) );
129  }
130 
131  // Open file
132  std::string filePath = fileName;
133  FILE * fileIdentifier = std::fopen( filePath.c_str( ), "w" );
134 
135  // Loop over independent variable rows
136  for ( unsigned int j = 0; j < independentVariables.at( 0 ).size( ); j++ )
137  {
138  // Save independent variable
139  fprintf( fileIdentifier, "%.10f ", independentVariables.at( 0 ).at( j ) );
140 
141  // Save dependent variables
142  for ( unsigned int i = 0; i < actualNumberOfCoefficients; i++ )
143  {
144  fprintf( fileIdentifier, "%.10f ", dependentVariables[ j ][ coefficientIndices.at( i ) ] );
145  }
146  fprintf( fileIdentifier, "\n" );
147  }
148 
149  // Close file
150  std::fclose( fileIdentifier );
151  }
