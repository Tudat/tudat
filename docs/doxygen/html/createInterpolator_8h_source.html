491 template< typename IndependentVariableType, typename DependentVariableType >
492 std::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
493 createOneDimensionalInterpolator(
494  const std::map< IndependentVariableType, DependentVariableType > dataToInterpolate,
495  const std::shared_ptr< InterpolatorSettings > interpolatorSettings,
496  const std::pair< DependentVariableType, DependentVariableType >& defaultExtrapolationValue =
497  std::make_pair( IdentityElement::getAdditionIdentity< DependentVariableType >( ),
498  IdentityElement::getAdditionIdentity< DependentVariableType >( ) ),
499  const std::vector< DependentVariableType > firstDerivativeOfDependentVariables =
500  std::vector< DependentVariableType >( ) )
501 {
502  std::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
503  createdInterpolator;
504 
505  // Check that boundary handling is one-dimensional
506  if ( interpolatorSettings->getBoundaryHandling( ).size( ) != 1 )
507  {
508  throw std::runtime_error( "Error while creating interpolator of type: " +
509  std::to_string( interpolatorSettings->getInterpolatorType( ) ) +
510  ". The interpolator is one-dimensional, but more than one boundary "
511  "handling methods have been defined." );
512  }
513 
514  // Check type of interpolator.
515  switch( interpolatorSettings->getInterpolatorType( ) )
516  {
517  case linear_interpolator:
518  createdInterpolator = std::make_shared< LinearInterpolator
519  < IndependentVariableType, DependentVariableType > >(
520  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
521  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
522  break;
523  case cubic_spline_interpolator:
524  {
525  if( !interpolatorSettings->getUseLongDoubleTimeStep( ) )
526  {
527  createdInterpolator = std::make_shared< CubicSplineInterpolator
528  < IndependentVariableType, DependentVariableType > >(
529  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
530  interpolatorSettings->getBoundaryHandling( ).at( 0 ) );
531  }
532  else
533  {
534  createdInterpolator = std::make_shared< CubicSplineInterpolator
535  < IndependentVariableType, DependentVariableType, long double > >(
536  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
537  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
538  }
539  break;
540  }
541  case lagrange_interpolator:
542  {
543  // Check consistency of input
544  std::shared_ptr< LagrangeInterpolatorSettings > lagrangeInterpolatorSettings =
545  std::dynamic_pointer_cast< LagrangeInterpolatorSettings >( interpolatorSettings );
546  if( lagrangeInterpolatorSettings != nullptr )
547  {
548  // Create Lagrange interpolator with requested time step type
549  if( !lagrangeInterpolatorSettings->getUseLongDoubleTimeStep( ) )
550  {
551  createdInterpolator = std::make_shared< LagrangeInterpolator
552  < IndependentVariableType, DependentVariableType, double > >(
553  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
554  interpolatorSettings->getSelectedLookupScheme( ),
555  lagrangeInterpolatorSettings->getLagrangeBoundaryHandling( ),
556  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
557  }
558  else
559  {
560  createdInterpolator = std::make_shared< LagrangeInterpolator
561  < IndependentVariableType, DependentVariableType, long double > >(
562  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
563  interpolatorSettings->getSelectedLookupScheme( ),
564  lagrangeInterpolatorSettings->getLagrangeBoundaryHandling( ),
565  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
566  }
567  }
568  else
569  {
570  throw std::runtime_error( "Error, did not recognize lagrange interpolator settings" );
571 
572  }
573  break;
574  }
575  case hermite_spline_interpolator:
576  {
577  if( firstDerivativeOfDependentVariables.size( ) != dataToInterpolate.size( ) )
578  {
579  throw std::runtime_error(
580  "Error when creating hermite spline interpolator, derivative size is inconsistent" );
581  }
582  createdInterpolator = std::make_shared< HermiteCubicSplineInterpolator
583  < IndependentVariableType, DependentVariableType > >(
584  dataToInterpolate, firstDerivativeOfDependentVariables,
585  interpolatorSettings->getSelectedLookupScheme( ),
586  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
587  break;
588  }
589  case piecewise_constant_interpolator:
590  createdInterpolator = std::make_shared< PiecewiseConstantInterpolator
591  < IndependentVariableType, DependentVariableType > >(
592  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
593  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
594  break;
595  default:
596  throw std::runtime_error( "Error when making interpolator, function cannot be used to create interplator of type " +
597  std::to_string( interpolatorSettings->getInterpolatorType( ) ) );
598  }
599  return createdInterpolator;
600 }

613  const std::shared_ptr< DataInterpolationSettings< IndependentType, DependentType > > dataInterpolationSettings,
614  const std::pair< DependentType, DependentType >& defaultExtrapolationValue =
615  std::make_pair( IdentityElement::getAdditionIdentity< DependentType >( ),

629  firstDerivativeOfDependentVariables );

645 template< typename IndependentVariableType, typename DependentVariableType, unsigned int NumberOfDimensions >
646 std::shared_ptr< MultiDimensionalInterpolator< IndependentVariableType, DependentVariableType, NumberOfDimensions > >
647 createMultiDimensionalInterpolator(
648  const std::vector< std::vector< IndependentVariableType > >& independentValues,
649  const boost::multi_array< DependentVariableType, static_cast< size_t >( NumberOfDimensions ) >& dependentData,
650  const std::shared_ptr< InterpolatorSettings > interpolatorSettings,
651  const std::vector< std::pair< DependentVariableType, DependentVariableType > >& defaultExtrapolationValue =
652  std::vector< std::pair< DependentVariableType, DependentVariableType > >
653  ( NumberOfDimensions, std::make_pair( IdentityElement::getAdditionIdentity< DependentVariableType >( ),
654  IdentityElement::getAdditionIdentity< DependentVariableType >( ) ) ) )
655 {
656  std::shared_ptr< MultiDimensionalInterpolator< IndependentVariableType, DependentVariableType, NumberOfDimensions > >
657  createdInterpolator;
658 
659  // Assign default values
660  std::vector< BoundaryInterpolationType > boundaryHandlingVector = interpolatorSettings->getBoundaryHandling( );
661  if ( boundaryHandlingVector.empty( ) )
662  {
663  boundaryHandlingVector = std::vector< BoundaryInterpolationType >( NumberOfDimensions, extrapolate_at_boundary );
664  }
665  // Check size of boundary handling methods
666  else if ( boundaryHandlingVector.size( ) != NumberOfDimensions )
667  {
668  throw std::runtime_error( "Error while creating multi-dimensional interpolator. The number of boundary handling methods does not "
669  "match the number of dimensions." );
670  }
671 
672  // Check type of interpolator.
673  switch ( interpolatorSettings->getInterpolatorType( ) )
674  {
675  case multi_linear_interpolator:
676  {
677  createdInterpolator = std::make_shared< MultiLinearInterpolator
678  < IndependentVariableType, DependentVariableType, NumberOfDimensions > >(
679  independentValues, dependentData, interpolatorSettings->getSelectedLookupScheme( ),
680  interpolatorSettings->getBoundaryHandling( ), defaultExtrapolationValue );
681  break;
682  }
683  default:
684  throw std::runtime_error( "Error when making interpolator, function cannot be used to create interplator of type " +
685  std::to_string( interpolatorSettings->getInterpolatorType( ) ) );
686  }
687  return createdInterpolator;
688 }
