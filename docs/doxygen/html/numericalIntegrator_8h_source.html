268 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
269 StateType NumericalIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >::integrateTo(
270  const IndependentVariableType intervalEnd,
271  const TimeStepType initialStepSize,
272  const TimeStepType finalTimeTolerance )
273 {
274  TimeStepType stepSize = initialStepSize;
275 
276  // Flag to indicate that the integration end value of the independent variable has been
277  // reached.
278  bool atIntegrationIntervalEnd = static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) )
279  * stepSize / std::fabs( stepSize )
280  <= finalTimeTolerance;
281 
282  int loopCounter = 0;
283  while ( !atIntegrationIntervalEnd )
284  {
285  // Check if the remaining interval is smaller than the step size.
286  if ( std::fabs( static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) ) )
287  <= std::fabs( stepSize ) *
288  ( 1.0 + finalTimeTolerance ) )
289  {
290  // The next step is beyond the end of the integration interval, so adjust the
291  // step size accordingly.
292  stepSize = intervalEnd - getCurrentIndependentVariable( );
293 
294  // Explicitly flag that the integration interval end is reached. Due to rounding
295  // off errors, it may not be possible to use
296  // ( currentIndependentVariable >= independentVariableEnd ) // in the while condition.
297  atIntegrationIntervalEnd = true;
298  }
299 
300  // Perform the step.
301  performIntegrationStep( stepSize );
302  stepSize = getNextStepSize( );
303 
304  // Only applicable to adaptive step size methods:
305  // Perform additional step(s) to reach intervalEnd exactly, in case the last step was rejected by
306  // the variable step size routine and a new step was used that was too small to get to intervalEnd.
307  if ( atIntegrationIntervalEnd )
308  {
309  // As long as intervalEnd is not reached, perform additional steps with the remaining time
310  // as suggested step size for the variable step size routine.
311  if( std::fabs( static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) ) ) >
312  finalTimeTolerance )
313  {
314  // Ensure that integrateTo function does not get stuck in a loop.
315  if( loopCounter < 1000 )
316  {
317  atIntegrationIntervalEnd = false;
318  loopCounter++;
319  }
320  else
321  {
322  std::cerr << "Warning, integrateTo function has failed to converge to final time to within tolerances, difference between true and requested final time is " <<
323  intervalEnd - getCurrentIndependentVariable( ) << ", final time is: " <<
324  getCurrentIndependentVariable( ) << std::endl;
325  }
326  }
327  }
328  }
329 
330  return getCurrentState( );
331 }
