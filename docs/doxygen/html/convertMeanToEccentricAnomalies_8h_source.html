64 template< typename ScalarType = double >
65 ScalarType computeKeplersFunctionForEllipticalOrbits( const ScalarType eccentricAnomaly,
66  const ScalarType eccentricity,
67  const ScalarType meanAnomaly )
68 {
69  return eccentricAnomaly - eccentricity * std::sin( eccentricAnomaly ) - meanAnomaly;

106 template< typename ScalarType = double >
107 ScalarType computeKeplersFunctionForHyperbolicOrbits( const ScalarType hyperbolicEccentricAnomaly,

154 template< typename ScalarType = double >
155 ScalarType convertMeanAnomalyToEccentricAnomaly(
156  const ScalarType eccentricity, const ScalarType aMeanAnomaly,
157  const bool useDefaultInitialGuess = true,
158  const ScalarType userSpecifiedInitialGuess = TUDAT_NAN,
159  std::shared_ptr< root_finders::RootFinder< ScalarType > > rootFinder =
160  std::shared_ptr< root_finders::RootFinder< ScalarType > >( ) )
161 {
162  using namespace mathematical_constants;
163  using namespace root_finders;
164 
165 
166  // Set mean anomaly to region between 0 and 2 PI.
167  ScalarType meanAnomaly = basic_mathematics::computeModulo< ScalarType >(
168  aMeanAnomaly, getFloatingInteger< ScalarType >( 2 ) *
169  getPi< ScalarType >( ) );
170 
171  // Required because the make_shared in the function definition gives problems for MSVC.
172  if ( !rootFinder.get( ) )
173  {
174  ScalarType tolerance = 10.0 * std::numeric_limits< ScalarType >::epsilon( );
175 
176  // Loosen tolerance for near-parabolic orbits
177  if( std::fabs( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) <
178  1.0E5 * std::numeric_limits< ScalarType >::epsilon( ) )
179  {
180  tolerance *= 2.5;
181  }
182 
183  rootFinder = root_finders::createRootFinder< ScalarType >(
184  root_finders::newtonRaphsonRootFinderSettings(
185  TUDAT_NAN, tolerance, TUDAT_NAN, 20, root_finders::throw_exception ) );
186  }
187 
188  // Declare eccentric anomaly.
189  ScalarType eccentricAnomaly = TUDAT_NAN;
190 
191  // Check if orbit is elliptical, and not near-parabolic.
192  if ( eccentricity < getFloatingInteger< ScalarType >( 1 ) &&
193  eccentricity >= getFloatingInteger< ScalarType >( 0 ) )
194  {
195  // Create an object containing the function of which we whish to obtain the root from.
196  std::shared_ptr< basic_mathematics::FunctionProxy< ScalarType, ScalarType > > rootFunction
197  = std::make_shared< basic_mathematics::FunctionProxy< ScalarType, ScalarType > >(
198  std::bind( &computeKeplersFunctionForEllipticalOrbits< ScalarType >, std::placeholders::_1,
199  eccentricity, meanAnomaly ) );
200 
201  // Add the first derivative of the root function.
202  rootFunction->addBinding(
203  -1, std::bind(
204  &computeFirstDerivativeKeplersFunctionForEllipticalOrbits< ScalarType >, std::placeholders::_1,
205  eccentricity ) );
206 
207  // Declare initial guess.
208  ScalarType initialGuess = TUDAT_NAN;
209 
210  // Set the initial guess. Check if the default scheme is to be used or a user specified
211  // value should be used.
212  // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
213  // If this scheme is changed, please run a very extensive test suite. The root finder
214  // function tends to be chaotic for some very specific combinations of mean anomaly and
215  // eccentricity. Various random tests of 100.000.000 samples were done to verify the
216  // functionality of this one, and of another option for the starter: PI. [Musegaas,2012]
217  if ( useDefaultInitialGuess )
218  {
219  if ( meanAnomaly > getPi< ScalarType >( ) )
220  {
221  initialGuess = meanAnomaly - eccentricity;
222  }
223  else
224  {
225  initialGuess = meanAnomaly + eccentricity;
226  }
227  }
228  else
229  {
230  initialGuess = userSpecifiedInitialGuess;
231  }
232 
233  try
234  {
235  // Set eccentric anomaly based on result of Newton-Raphson root-finding algorithm.
236  eccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
237  }
238  // Use bisection algorithm if root finder fails
239  catch( std::runtime_error )
240  {
241  // Set tolerance
242  ScalarType tolerance = 10.0 * std::numeric_limits< ScalarType >::epsilon( );
243 
244  // Set upper/lower bounds (loosely)
245  ScalarType lowerBound, upperBound;
246  if ( meanAnomaly > getPi< ScalarType >( ) )
247  {
248  lowerBound = 0.5 * getPi< ScalarType >( ) + 10.0 * std::numeric_limits< ScalarType >::epsilon( );
249  upperBound = mathematical_constants::getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( );
250  }
251  else
252  {
253  lowerBound = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
254  upperBound = 1.5 * getPi< ScalarType >( );
255  }
256 
257  // Create root finder
258  std::shared_ptr< RootFinder< ScalarType > > bisectionRootfinder =
259  rootFinder = root_finders::createRootFinder< ScalarType >(
260  root_finders::bisectionRootFinderSettings(
261  TUDAT_NAN, TUDAT_NAN, tolerance, 100, root_finders::accept_result ),
262  lowerBound, upperBound );
263 
264  initialGuess = meanAnomaly;
265  eccentricAnomaly = bisectionRootfinder->execute( rootFunction, initialGuess );
266  }
267  }
268  // Eccentricity is invalid: eccentricity < 0.0 or eccentricity >= 1.0.
269  else
270  {
271  throw std::runtime_error( "Invalid eccentricity. Valid range is 0.0 <= e < 1.0. Eccentricity was: " +
272  std::to_string( eccentricity ) );
273  }
274 
275  // Return eccentric anomaly.
276  return eccentricAnomaly;
277 }

297 template< typename ScalarType = double >
298 ScalarType convertMeanAnomalyToHyperbolicEccentricAnomaly(
299  const ScalarType eccentricity, const ScalarType hyperbolicMeanAnomaly,
300  const bool useDefaultInitialGuess = true,
301  const ScalarType userSpecifiedInitialGuess = TUDAT_NAN,
302  std::shared_ptr< root_finders::RootFinder< ScalarType > > aRootFinder =
303  std::shared_ptr< root_finders::RootFinder< ScalarType > >( ) )
304 {
305  using namespace mathematical_constants;
306  using namespace root_finders;
307 
308 
309  std::shared_ptr< RootFinder< ScalarType > > rootFinder = aRootFinder;
310 
311  // Required because the make_shared in the function definition gives problems for MSVC.
312  if ( !rootFinder.get( ) )
313  {
314  double toleranceMultiplier = ( eccentricity - 1.0 ) < 1.0E-3 ? 10.0 : 1.0;
315  rootFinder = createRootFinder< ScalarType >(
316  newtonRaphsonRootFinderSettings(
317  TUDAT_NAN, toleranceMultiplier * 25.0 * std::numeric_limits< ScalarType >::epsilon( ),
318  TUDAT_NAN, 1000, root_finders::throw_exception ) );
319  }
320  // Declare hyperbolic eccentric anomaly.
321  ScalarType hyperbolicEccentricAnomaly = TUDAT_NAN;
322 
323  // Check if orbit is hyperbolic.
324  if ( eccentricity > getFloatingInteger< ScalarType >( 1 ) )
325  {
326  // Create an object containing the function of which we whish to obtain the root from.
327  std::shared_ptr< basic_mathematics::FunctionProxy< ScalarType, ScalarType > > rootFunction
328  = std::make_shared< basic_mathematics::FunctionProxy< ScalarType, ScalarType > >(
329  std::bind( &computeKeplersFunctionForHyperbolicOrbits< ScalarType >, std::placeholders::_1,
330  eccentricity, hyperbolicMeanAnomaly ) );
331 
332  // Add the first derivative of the root function.
333  rootFunction->addBinding(
334  -1, std::bind(
335  &computeFirstDerivativeKeplersFunctionForHyperbolicOrbits< ScalarType >,
336  std::placeholders::_1, eccentricity ) );
337 
338  // Declare initial guess.
339  ScalarType initialGuess = TUDAT_NAN;
340 
341  // Set the initial guess. Check if the default scheme is to be used or a user specified
342  // value should be used. See [Wakker, 2007] for derivations of the default values.
343  // Note that an error was detected in these starter values, as is discussed in
344  // [Musegaas,2012].
345  // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
346  // If this scheme is changed, please run a very extensive test suite. The root finder
347  // function tends to be chaotic for some very specific combinations of mean anomaly and
348  // eccentricity. Various random tests of 100.000.000 samples were done to verify the
349  // functionality of this one. [Musegaas,2012]
350  if ( useDefaultInitialGuess )
351  {
352  if ( std::abs( hyperbolicMeanAnomaly ) <
353  getFloatingInteger< ScalarType >( 6 ) * eccentricity )
354  {
355  initialGuess =
356  std::sqrt( getFloatingInteger< ScalarType >( 8 ) *
357  ( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) /
358  eccentricity ) *
359  std::sinh( getFloatingFraction< ScalarType >( 1, 3 ) * boost::math::asinh(
360  getFloatingInteger< ScalarType >( 3 ) *
361  hyperbolicMeanAnomaly /
362  ( std::sqrt( getFloatingInteger< ScalarType >( 8 ) *
363  ( eccentricity -
364  getFloatingInteger< ScalarType >( 1 ) ) /
365  eccentricity ) *
366  ( eccentricity - getFloatingInteger< ScalarType >( 1 )
367  ) ) ) );
368  }
369  else if ( hyperbolicMeanAnomaly > getFloatingInteger< ScalarType >( 6 ) * eccentricity )
370  {
371  initialGuess = ( std::log( getFloatingInteger< ScalarType >( 2 ) *
372  hyperbolicMeanAnomaly / eccentricity ) );
373  }
374  else
375  {
376  initialGuess = ( - std::log( -getFloatingInteger< ScalarType >( 2 ) *
377  hyperbolicMeanAnomaly / eccentricity ) );
378  }
379  }
380  else
381  {
382  initialGuess = userSpecifiedInitialGuess;
383  }
384 
385  // Set hyperbolic eccentric anomaly based on result of Newton-Raphson root-finding
386  // algorithm.
387  try
388  {
389  hyperbolicEccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
390 
391  }
392  catch( std::runtime_error )
393  {
394  rootFinder = createRootFinder< ScalarType >(
395  bisectionRootFinderSettings(
396  TUDAT_NAN, 20.0 * std::numeric_limits< ScalarType >::epsilon( ),
397  TUDAT_NAN, 100 ), getFloatingInteger< ScalarType >( 0 ),
398  getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ), root_finders::accept_result );
399 
400  hyperbolicEccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
401  }
402 
403  }
404 
405  // In this case the orbit is not hyperbolic.
406  else
407  {
408  throw std::runtime_error( "Invalid eccentricity. Valid range is e > 1.0. Eccentricity was: " +
409  std::to_string( eccentricity ) );
410  }
411 
412  // Return hyperbolic eccentric anomaly.
413  return hyperbolicEccentricAnomaly;
414 }
