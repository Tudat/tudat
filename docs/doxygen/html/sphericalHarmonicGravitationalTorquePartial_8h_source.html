115  const std::pair< std::string, std::string >& stateReferencePoint,

117  {
118  bool isStateDerivativeDependent = 0;

120  && integratedStateType == propagators::translational_state ) )

122  isStateDerivativeDependent = true;

124  else if( ( ( stateReferencePoint.first == bodyUndergoingTorque_ || ( stateReferencePoint.first == bodyExertingTorque_ ) )
125  && integratedStateType == propagators::body_mass_state ) )
126  {
127  throw std::runtime_error( "Warning, dependency of 2nd degree gravity torques on body masses not yet implemented" );
128  }
129  return isStateDerivativeDependent;
130  }
