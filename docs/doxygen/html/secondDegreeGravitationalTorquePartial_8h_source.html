123  bool isStateDerivativeDependentOnIntegratedAdditionalStateTypes(
124  const std::pair< std::string, std::string >& stateReferencePoint,
125  const propagators::IntegratedStateType integratedStateType )
126  {
127  bool isStateDerivativeDependent = 0;
128  if( ( ( stateReferencePoint.first == bodyUndergoingTorque_ || ( stateReferencePoint.first == bodyExertingTorque_ ) )
129  && integratedStateType == propagators::translational_state ) )
130  {
131  isStateDerivativeDependent = true;
132  }
133  else if( ( ( stateReferencePoint.first == bodyUndergoingTorque_ || ( stateReferencePoint.first == bodyExertingTorque_ ) )
134  && integratedStateType == propagators::body_mass_state ) )
135  {
136  throw std::runtime_error( "Warning, dependency of 2nd degree gravity torques on body masses not yet implemented" );
137  }
138  return isStateDerivativeDependent;
139  }
