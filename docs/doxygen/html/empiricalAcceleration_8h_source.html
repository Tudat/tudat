135  void updateMembers( const double currentTime = TUDAT_NAN )
136  {
137  if( !( this->currentTime_ == currentTime ) )
138  {
139  // Calulate current relative state of accelerated body
140  currentState_ = bodyStateFunction_( ) - centralBodyStateFunction_( );
141 
142  // Calculate current body-fixed state of accelerated body.
143  currentRotationFromRswToInertialFrame_ = Eigen::Quaterniond(
144  reference_frames::getInertialToRswSatelliteCenteredFrameRotationMatrix( currentState_ ) ).inverse( );
145 
146  // Calculate current true anomaly of accelerated body.
147  currentTrueAnomaly_ = orbital_element_conversions::convertCartesianToKeplerianElements(
148  currentState_, centralBodyGravitationalParameterFunction_( ) )( 5 );
149 
150  // Calculate acceleration
151  updateAccelerationComponents( currentTime );
152  currentLocalAcclereration_ = ( currentConstantAcceleration_ +
153  currentSineAcceleration_ * std::sin( currentTrueAnomaly_ ) +
154  currentCosineAcceleration_ * std::cos( currentTrueAnomaly_ ) );
155 
156  // Perform sanity check.
157  if( currentLocalAcclereration_ != currentLocalAcclereration_ )
158  {
159  throw std::runtime_error( "Error when computing emprical acceleration, result is NaN" );
160  }
161 
162  this->currentTime_ = currentTime;
163  }
164  }
