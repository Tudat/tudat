154  DependentVariableType interpolate( const IndependentVariableType independentVariableValue )
155  {
156  // Check whether boundary handling needs to be applied, if independent variable is beyond its defined range.
157  DependentVariableType interpolatedValue;
158  bool useValue = false;
159  this->checkBoundaryCase( interpolatedValue, useValue, independentVariableValue );
160  if( useValue )
161  {
162  return interpolatedValue;
163  }
164 
165  // Lookup nearest lower index.
166  int newNearestLowerIndex = lookUpScheme_->findNearestLowerNeighbour( independentVariableValue );
167 
168 
169  // Check if jump occurs
170  if( std::abs( dependentValues_[ newNearestLowerIndex ] - dependentValues_[ newNearestLowerIndex + 1 ] ) >
171  maximumAllowableVariation_ )
172  {
173  double jumpSign = ( dependentValues_[ newNearestLowerIndex ] - dependentValues_[ newNearestLowerIndex + 1 ] ) /
174  std::abs( dependentValues_[ newNearestLowerIndex ] - dependentValues_[ newNearestLowerIndex + 1 ] );
175  interpolatedValue = dependentValues_[ newNearestLowerIndex ] +
176  ( independentVariableValue - independentValues_[ newNearestLowerIndex ] ) /
177  ( independentValues_[ newNearestLowerIndex + 1 ] - independentValues_[ newNearestLowerIndex ] ) *
178  ( dependentValues_[ newNearestLowerIndex + 1 ] + jumpSign * jumpSize_ - dependentValues_[ newNearestLowerIndex ] );
179  }
180  else
181  {
182 
183  interpolatedValue = dependentValues_[ newNearestLowerIndex ] +
184  ( independentVariableValue - independentValues_[ newNearestLowerIndex ] ) /
185  ( independentValues_[ newNearestLowerIndex + 1 ] - independentValues_[ newNearestLowerIndex ] ) *
186  ( dependentValues_[ newNearestLowerIndex + 1 ] - dependentValues_[ newNearestLowerIndex ] );
187  }
188 
189  return interpolatedValue;
190  }
