198  Eigen::Vector6d getCartesianState(
199  const double secondsSinceEpoch )
200  {
201  Eigen::Vector6d currentState = baseEphemeris_->getCartesianState( secondsSinceEpoch );
202  Eigen::Vector6d stateScaling = stateScalingFunction_( secondsSinceEpoch );
203  if( !isScalingAbsolute_ )
204  {
205  currentState.array( ) *= stateScaling.array( );
206  }

209  currentState += stateScaling;
