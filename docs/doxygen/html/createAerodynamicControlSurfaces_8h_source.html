71  independentVariableNames_( independentVariableNames )
72  {
73  // Check input type consistency
74  if( aerodynamicCoefficientType_ == hypersonic_local_inclincation_coefficients )
75  {
76  throw std::runtime_error( "Error, hypersonic local inclination control surface increments not yet available" );
77  }
78 
79  if( aerodynamicCoefficientType_ == constant_aerodynamic_coefficients )
80  {
81  throw std::runtime_error( "Error, constant control surface increments not available, should depend at least on deflection angle" );

83 
84  if( std::count( independentVariableNames.begin( ), independentVariableNames.end( ),
85  aerodynamics::control_surface_deflection_dependent ) != 1 )
86  {
87  std::cerr << "Warning when creating ControlSurfaceIncrementAerodynamicCoefficientSettings, expected single dependency on control surface deflections, found " <<
88  std::count( independentVariableNames.begin( ), independentVariableNames.end( ),
89  aerodynamics::control_surface_deflection_dependent ) << ", dependencies" << std::endl;
90  }

249 template< unsigned int NumberOfIndependentVariables >
250 std::shared_ptr< ControlSurfaceIncrementAerodynamicCoefficientSettings >
251 readGivenSizeTabulatedControlIncrementAerodynamicCoefficientsFromFiles(
252  const std::map< int, std::string > forceCoefficientFiles,
253  const std::map< int, std::string > momentCoefficientFiles,
254  const std::vector< aerodynamics::AerodynamicCoefficientsIndependentVariables > independentVariableNames )
255 {
256  std::pair< boost::multi_array< Eigen::Vector3d, NumberOfIndependentVariables >, std::vector< std::vector< double > > >
257  aerodynamicForceCoefficients = input_output::readAerodynamicCoefficients< NumberOfIndependentVariables >(
258  forceCoefficientFiles );
259  std::pair< boost::multi_array< Eigen::Vector3d, NumberOfIndependentVariables >, std::vector< std::vector< double > > >
260  aerodynamicMomentCoefficients = input_output::readAerodynamicCoefficients< NumberOfIndependentVariables >(
261  momentCoefficientFiles );
262 
263  if( !input_output::compareIndependentVariables(
264  aerodynamicForceCoefficients.second, aerodynamicMomentCoefficients.second ) )
265  {
266  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, force and moment independent variables are inconsistent" );
267  }
268 
269  if( independentVariableNames.size( ) != NumberOfIndependentVariables )
270  {
271  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, input sizes are inconsistent" );
272  }
273 
274  return std::make_shared< TabulatedControlSurfaceIncrementAerodynamicCoefficientSettings<
275  NumberOfIndependentVariables > >(
276  aerodynamicForceCoefficients.second, aerodynamicForceCoefficients.first, aerodynamicMomentCoefficients.first,
277  independentVariableNames );
278 }

364 template< unsigned int NumberOfDimensions >
365 std::shared_ptr< aerodynamics::ControlSurfaceIncrementAerodynamicInterface >
366 createTabulatedControlSurfaceIncrementAerodynamicCoefficientInterface(
367  const std::vector< std::vector< double > > independentVariables,
368  const boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) > forceCoefficients,
369  const boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) > momentCoefficients,
370  const std::vector< aerodynamics::AerodynamicCoefficientsIndependentVariables >
371  independentVariableNames )
372 {
373  // Check input consistency.
374  if( independentVariables.size( ) != NumberOfDimensions )
375  {
376  throw std::runtime_error( "Error when creating tabulated control surface increment aerodynamic coefficient interface, inconsistent variable vector dimensioning" );
377  }
378 
379  if( independentVariableNames.size( ) != NumberOfDimensions )
380  {
381  throw std::runtime_error( "Error when creating tabulated control surface increment aerodynamic coefficient interface, inconsistent variable name vector dimensioning" );
382 
383  }
384 
385  // Create interpolators for coefficients.
386  std::shared_ptr< interpolators::MultiLinearInterpolator
387  < double, Eigen::Vector3d, NumberOfDimensions > > forceInterpolator =
388  std::make_shared< interpolators::MultiLinearInterpolator
389  < double, Eigen::Vector3d, NumberOfDimensions > >(
390  independentVariables, forceCoefficients );
391  std::shared_ptr< interpolators::MultiLinearInterpolator
392  < double, Eigen::Vector3d, NumberOfDimensions > > momentInterpolator =
393  std::make_shared< interpolators::MultiLinearInterpolator
394  < double, Eigen::Vector3d, NumberOfDimensions > >(
395  independentVariables, momentCoefficients );
396 
397  // Create aerodynamic coefficient interface.
398  return std::make_shared< aerodynamics::CustomControlSurfaceIncrementAerodynamicInterface >(
399  std::bind( &interpolators::MultiLinearInterpolator
400  < double, Eigen::Vector3d, NumberOfDimensions >::interpolate,
401  forceInterpolator, std::placeholders::_1 ),
402  std::bind( &interpolators::MultiLinearInterpolator
403  < double, Eigen::Vector3d, NumberOfDimensions >::interpolate,
404  momentInterpolator, std::placeholders::_1 ),
405  independentVariableNames );
406 }

415 template< unsigned int NumberOfDimensions >
416 std::shared_ptr< aerodynamics::ControlSurfaceIncrementAerodynamicInterface >
417 createTabulatedControlSurfaceIncrementAerodynamicCoefficientInterface(
418  const std::shared_ptr< ControlSurfaceIncrementAerodynamicCoefficientSettings > coefficientSettings,
419  const std::string& body )
420 {
421  // Check consistency of type.
422  std::shared_ptr< TabulatedControlSurfaceIncrementAerodynamicCoefficientSettings< NumberOfDimensions > >
423  tabulatedCoefficientSettings = std::dynamic_pointer_cast<
424  TabulatedControlSurfaceIncrementAerodynamicCoefficientSettings< NumberOfDimensions > >( coefficientSettings );
425  if( tabulatedCoefficientSettings == nullptr )
426  {
427  throw std::runtime_error(
428  "Error, expected tabulated control surface increment aerodynamic coefficients of size " +
429  std::to_string( NumberOfDimensions ) + "for body " + body );
430  }
431  else
432  {
433  return createTabulatedControlSurfaceIncrementAerodynamicCoefficientInterface< NumberOfDimensions >(
434  tabulatedCoefficientSettings->getIndependentVariables( ),
435  tabulatedCoefficientSettings->getForceCoefficients( ),
436  tabulatedCoefficientSettings->getMomentCoefficients( ),
437  tabulatedCoefficientSettings->getIndependentVariableNames( ) );
438  }
