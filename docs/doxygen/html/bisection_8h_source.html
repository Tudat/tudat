120  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess = 0.0 )
121  {
122  // The value of the initialGuess is not used.
123  TUDAT_UNUSED_PARAMETER( initialGuess );
124 
125  // Set the root function.
126  this->rootFunction = rootFunction;
127 
128  // Initialize previous values.
129  DataType previousRootValue = TUDAT_NAN;
130  DataType previousRootFunctionValue = TUDAT_NAN;
131 
132  // Duplicate the interval and use this duplicate to shrink the interval.
133  // Compute the midpoint of the interval, and take this as the current root.
134  DataType currentLowerBound = lowerBound_;
135  DataType currentUpperBound = upperBound_;
136  DataType rootValue = ( currentLowerBound + currentUpperBound ) / 2.0;
137 
138  // Find the corresponding function values at the important interval points (lower bound,
139  // upper bound and midpoint).
140  DataType currentLowerBoundFunctionValue = this->rootFunction->evaluate( currentLowerBound );
141  DataType currentUpperBoundFunctionValue = this->rootFunction->evaluate( currentUpperBound );
142  if( currentLowerBoundFunctionValue == mathematical_constants::getFloatingInteger< DataType >( 0 ) )
143  {
144  rootValue = currentLowerBound;
145  }
146  else if( currentUpperBoundFunctionValue == mathematical_constants::getFloatingInteger< DataType >( 0 ) )
147  {
148  rootValue = currentUpperBound;
149  }
150  else
151  {
152  DataType rootFunctionValue = this->rootFunction->evaluate( rootValue );
153 
154  // Validate that upperbound and lowerbound function values have different signs
155  // (requirement).
156  if( currentLowerBoundFunctionValue * currentUpperBoundFunctionValue > 0.0 )
157  {
158  throw std::runtime_error( "The Bisection algorithm requires that the values at the upper, "
159  "and lower bounds have a different sign." );
160  }
161 
162  // Loop counter.
163  unsigned int counter = 1;
164 
165  // Loop until we have a solution with sufficient accuracy.
166  do
167  {
168  // Sanity check.
169  if( currentLowerBoundFunctionValue * currentUpperBoundFunctionValue > 0.0 )
170  {
171  throw std::runtime_error( "The Bisection algorithm requires that the values at the upper, "
172  "and lower bounds have a different sign, error during iteration." );
173  }
174 
175  // Save old values.
176  previousRootValue = rootValue;
177  previousRootFunctionValue = rootFunctionValue;
178 
179  // Check which subinterval to keep, by maintaining endpoints with opposite function
180  // value signs.
181  if( rootFunctionValue * currentLowerBoundFunctionValue < 0.0 )
182  {
183  // Different sign, hence the upper bound is replaced.
184  currentUpperBound = rootValue;
185  currentUpperBoundFunctionValue = rootFunctionValue;
186  }
187  else
188  {
189  // Same sign, hence the lower bound is replaced.
190  currentLowerBound = rootValue;
191  currentLowerBoundFunctionValue = rootFunctionValue;
192  }
193 
194  // Compute the new midpoint of the interval and its function value.
195  rootValue = ( currentLowerBound + currentUpperBound ) / 2.0;
196  rootFunctionValue = this->rootFunction->evaluate( rootValue );
197 
198  counter++;
199  }
200  while( rootFunctionValue != mathematical_constants::getFloatingInteger< DataType >( 0 ) &&
201  !this->terminationFunction_( rootValue, previousRootValue, rootFunctionValue,
202  previousRootFunctionValue, counter ) );
203  }
204  return rootValue;
205  }
