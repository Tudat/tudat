372  void updateInterface( const double currentTime = TUDAT_NAN )
373  {
374  updateInterfaceBase( currentTime );
375 
376  Eigen::Quaterniond rotationToPropagationFrame = rotationFromLocalToPropagationFrame_( );
377  for( unsigned int i = 0; i < surfaceNormalsInPropagationFrame_.size( ); i++ )
378  {
379  surfaceNormalsInPropagationFrame_[ i ] =
380  rotationToPropagationFrame * localFrameSurfaceNormals_.at( i )( currentTime );
