75 #define TUDAT_CHECK_MATRIX_BASE( L, R ) \
76  boost::test_tools::predicate_result equalRows = \
77  boost::test_tools::tt_detail::equal_impl( L.rows( ), R.rows( ) ); \
78  BOOST_CHECK_MESSAGE( equalRows, boost::str( boost::format( \
79  "Matrix number of rows not equal: %d != %d" ) % L.rows( ) % R.rows( ) ) ); \
80  boost::test_tools::predicate_result equalCols = \
81  boost::test_tools::tt_detail::equal_impl( L.cols( ), R.cols( ) ); \
82  BOOST_CHECK_MESSAGE( equalCols, boost::str( boost::format( \
83  "Matrix number of columns not equal: %d != %d" ) % L.cols( ) % R.cols( ) ) ); \
84  if ( equalRows.p_predicate_value && equalCols.p_predicate_value ) \
85  for ( signed int row = 0; row < L.rows( ); row++ ) \
86  for ( signed int col = 0; col < L.cols( ); col++ )
