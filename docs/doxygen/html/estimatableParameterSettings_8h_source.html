73 inline std::vector< std::pair< int, int > > getSphericalHarmonicBlockIndices(
74  const int minimumDegree,
75  const int minimumOrder,
76  const int maximumDegree,
77  const int maximumOrder )
78 {
79  std::vector< std::pair< int, int > > blockIndices;
80  for( int i = minimumDegree; i <= maximumDegree; i++ )
81  {
82  for( int j = minimumOrder; ( ( j <= i ) && ( j <= maximumOrder ) ); j++ )
83  {
84  blockIndices.push_back( std::make_pair( i, j ) );
85  }
86  }
87  return blockIndices;
88 }
