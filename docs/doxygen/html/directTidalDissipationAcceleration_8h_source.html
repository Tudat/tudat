170  void updateMembers( const double currentTime = TUDAT_NAN )
171  {
172  if( !( this->currentTime_ == currentTime ) )
173  {
174  // Update relative state
175  currentRelativeState_ = stateFunctionOfBodyExertingTide_( ) - stateFunctionOfBodyUndergoingTide_( );
176  double distance = currentRelativeState_.segment( 0, 3 ).norm( );
177  double distanceSquared = distance * distance;
178  double distanceToEighthPower = distanceSquared * distanceSquared * distanceSquared * distanceSquared;
179 
180  // Check if tide on planet or satellite is to be used
181  if( modelTideOnPlanet_ )
182  {
183  // Retrieve/compute planet-specific models
184  currentTidalAccelerationMultiplier_ =
185  - 3.0 * gravitationalParameterFunctionOfBodyUndergoingTide_( ) * equatorialRadiusToFifthPower_ /
186  distanceToEighthPower * k2LoveNumber_;
187  currentAngularVelocityVectorOfBodyUndergoingTide_ = angularVelocityVectorOfBodyUndergoingTide_( );
188 
189  currentAcceleration_ = computeDirectTidalAccelerationDueToTideOnPlanet(
190  currentRelativeState_, currentAngularVelocityVectorOfBodyUndergoingTide_,
191  currentTidalAccelerationMultiplier_, timeLag_, includeDirectRadialComponent_ );
192  }
193  else
194  {
195  // Retrieve/compute satellite-specific models
196  currentTidalAccelerationMultiplier_ =
197  - 3.0 * gravitationalParameterFunctionOfBodyExertingTide_( ) * equatorialRadiusToFifthPower_ /
198  distanceToEighthPower * k2LoveNumber_;
199  currentTidalAccelerationMultiplier_ *= gravitationalParameterFunctionOfBodyExertingTide_( ) /
200  gravitationalParameterFunctionOfBodyUndergoingTide_( );
201 
202  currentAcceleration_ = computeDirectTidalAccelerationDueToTideOnSatellite(
203  currentRelativeState_, currentTidalAccelerationMultiplier_,
204  timeLag_, includeDirectRadialComponent_ );
205  }
206 
207  currentTime_ = currentTime;
208  }
209  }
