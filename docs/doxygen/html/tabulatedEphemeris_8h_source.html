144  std::pair< double, double > getSafeInterpolationInterval( )
145  {
146  std::pair< double, double > safeInterpolationInterval;
147 
148  // Check interpolator type. If interpolator is not a Lagrange interpolator, return full domain
149  if( std::dynamic_pointer_cast< interpolators::LagrangeInterpolator< TimeType, StateType, double > >(
150  interpolator_ ) == nullptr &&
151  std::dynamic_pointer_cast< interpolators::LagrangeInterpolator< TimeType, StateType, long double > >(
152  interpolator_ ) == nullptr )
153  {
154  safeInterpolationInterval.first = interpolator_->getIndependentValues( ).at( 0 );
155  safeInterpolationInterval.second = interpolator_->getIndependentValues( ).at(
156  interpolator_->getIndependentValues( ).size( ) - 1 );
157  }
158  // If interpolator is a Lagrange interpolator, return full domain minus edges where interpolator has reduced accuracy
159  else if( std::dynamic_pointer_cast< interpolators::LagrangeInterpolator< TimeType, StateType, double > >(
160  interpolator_ ) != nullptr )
161  {
162  int numberOfNodes =
163  std::dynamic_pointer_cast< interpolators::LagrangeInterpolator< TimeType, StateType, double > >(
164  interpolator_ )->getNumberOfStages( );
165 
166  safeInterpolationInterval.first = interpolator_->getIndependentValues( ).at( 0 + numberOfNodes / 2 + 1 );
167  safeInterpolationInterval.second = interpolator_->getIndependentValues( ).at(
168  interpolator_->getIndependentValues( ).size( ) - 1 - ( + numberOfNodes / 2 + 1 ) );
169  }
170  else if( std::dynamic_pointer_cast< interpolators::LagrangeInterpolator< TimeType, StateType, long double > >(
171  interpolator_ ) != nullptr )
172  {
173  int numberOfNodes =
174  std::dynamic_pointer_cast< interpolators::LagrangeInterpolator< TimeType, StateType, long double > >(
175  interpolator_ )->getNumberOfStages( );
176  safeInterpolationInterval.first = interpolator_->getIndependentValues( ).at( 0 + numberOfNodes / 2 + 1 );
177  safeInterpolationInterval.second = interpolator_->getIndependentValues( ).at(
178  interpolator_->getIndependentValues( ).size( ) - 1 - ( + numberOfNodes / 2 + 1 ) );
179  }
180  return safeInterpolationInterval;
181  }

243 template< typename StateScalarType = double, typename TimeType = double >
244 std::shared_ptr< Ephemeris > getTabulatedEphemeris(
245  const std::shared_ptr< Ephemeris > ephemerisToInterrogate,
246  const TimeType startTime,
247  const TimeType endTime,
248  const TimeType timeStep,
249  const std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
250  std::make_shared< interpolators::LagrangeInterpolatorSettings >( 8 ) )
251 {
252  typedef Eigen::Matrix< StateScalarType, 6, 1 > StateType;
253 
254  // Create state map that is to be interpolated
255  std::map< TimeType, StateType > stateMap;
256  TimeType currentTime = startTime;
257  while( currentTime <= endTime )
258  {
259  stateMap[ currentTime ] = ephemerisToInterrogate->getTemplatedStateFromEphemeris<
260  StateScalarType, TimeType >( currentTime );
261  currentTime += timeStep;
262  }
263 
264  // Create tabulated ephemeris model
265  return std::make_shared< TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
266  interpolators::createOneDimensionalInterpolator( stateMap, interpolatorSettings ),
267  ephemerisToInterrogate->getReferenceFrameOrigin( ),
268  ephemerisToInterrogate->getReferenceFrameOrientation( ) );
269 
270 }
