146  void updateCurrentControlSurfaceCoefficientsCoefficients(
147  const std::string& currentControlSurface,
148  std::vector< double > controlSurfaceIndependentVariables )
149  {
150  if( controlSurfaceIncrementInterfaces_.count( currentControlSurface ) == 0 )
151  {
152  throw std::runtime_error( "Error when updating coefficients, could not fid control surface " + currentControlSurface );
153  }
154  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->updateCurrentCoefficients(
155  controlSurfaceIndependentVariables );
156  currentForceCoefficients_ +=
157  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentForceCoefficients( );
158  currentMomentCoefficients_ +=
159  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentMomentCoefficients( );
160  }

466  ScaledAerodynamicCoefficientInterface(
467  const std::shared_ptr< AerodynamicCoefficientInterface > baseCoefficientInterface,
468  const std::function< Eigen::Vector3d( const double ) > forceCoefficientScalingFunction,
469  const std::function< Eigen::Vector3d( const double ) > momentCoefficientScalingFunction,
470  const bool isScalingRelative = true ):
471  AerodynamicCoefficientInterface(
472  baseCoefficientInterface->getReferenceLength( ),
473  baseCoefficientInterface->getReferenceArea( ),
474  baseCoefficientInterface->getLateralReferenceLength( ),
475  baseCoefficientInterface->getMomentReferencePoint( ),
476  baseCoefficientInterface->getIndependentVariableNames( ),
477  baseCoefficientInterface->getAreCoefficientsInAerodynamicFrame( ),
478  baseCoefficientInterface->getAreCoefficientsInNegativeAxisDirection( ) ),
479  baseCoefficientInterface_( baseCoefficientInterface ),
480  forceCoefficientScalingFunction_( forceCoefficientScalingFunction ),
481  momentCoefficientScalingFunction_( momentCoefficientScalingFunction ),
482  isScalingRelative_( isScalingRelative){ }
483 
484  void updateCurrentCoefficients(
485  const std::vector< double >& independentVariables,
486  const double currentTime = TUDAT_NAN )
487  {
488  baseCoefficientInterface_->updateCurrentCoefficients( independentVariables, currentTime );
489 
490  currentForceCoefficients_ = baseCoefficientInterface_->getCurrentForceCoefficients( );
491  currentMomentCoefficients_ = baseCoefficientInterface_->getCurrentMomentCoefficients( );
492 
493  if( isScalingRelative_ )
494  {
495  currentForceCoefficients_.array() *= forceCoefficientScalingFunction_( currentTime ).array();
496  currentMomentCoefficients_.array() *= momentCoefficientScalingFunction_( currentTime ).array();
497  }
498  else
499  {
500  currentForceCoefficients_ += forceCoefficientScalingFunction_( currentTime );
501  currentMomentCoefficients_ += momentCoefficientScalingFunction_( currentTime );
502  }
503 
504  }
505 
506 private:
507 
508  std::shared_ptr< AerodynamicCoefficientInterface > baseCoefficientInterface_;
509 
510  std::function< Eigen::Vector3d( const double ) > forceCoefficientScalingFunction_;
511 
512  std::function< Eigen::Vector3d( const double ) > momentCoefficientScalingFunction_;
513 
514  bool isScalingRelative_;
515
