66  virtual DependentVariable computeDerivative( const unsigned int order,
67  const IndependentVariable independentVariable )
68  {
69  if ( order == 0 )
70  {
71  // Zero th order derivative is the function itself.
72  return this->evaluate( independentVariable );
73  }
74 
75  // Define step size to the left and right.
76  double stepSize = sqrt_epsilon_double * independentVariable;
77 
78  // Return derivative according to formula:
79  // f'(x) = [f(x+h)-f(x-h)]/2h
80  return ( computeDerivative( order - 1, independentVariable + stepSize )
81  - computeDerivative( order - 1, independentVariable - stepSize ) )
82  / ( 2.0 * stepSize );
83  }
