79 template< typename StateScalarType = double >
80 Eigen::Matrix< StateScalarType, 6, 1 > convertSphericalOrbitalToCartesianState(
81  const Eigen::Matrix< StateScalarType, 6, 1 >& sphericalOrbitalState )
82 {
83  Eigen::Matrix< StateScalarType, 6, 1 > cartesianState;
84 
85  // Compute Cartesian position
86  Eigen::Matrix< StateScalarType, 3, 1 > sphericalPosition = sphericalOrbitalState.segment( 0, 3 );
87  sphericalPosition( 1 ) = mathematical_constants::getPi< StateScalarType >( ) /
88  mathematical_constants::getFloatingInteger< StateScalarType >( 2 ) - sphericalOrbitalState( 1 );
89  cartesianState.segment( 0, 3 ) = coordinate_conversions::convertSphericalToCartesian< StateScalarType >(
90  sphericalPosition );
91 
92  // Check whether flight path angle and heading angle are valid (corresponding velocity components are zero otherwise)
93  bool isFlightPathAngleValid = ( sphericalOrbitalState( flightPathIndex ) == sphericalOrbitalState( flightPathIndex ) );
94  bool isHeadingAngleValid = ( sphericalOrbitalState( headingAngleIndex ) == sphericalOrbitalState( headingAngleIndex ) );
95 
96  // Compute velocity in vertical frame.
97  Eigen::Matrix< StateScalarType, 3, 1 > velocityInVerticalFrame = Eigen::Matrix< StateScalarType, 3, 1 >::Zero( );
98  if( isFlightPathAngleValid && isHeadingAngleValid )
99  {
100  velocityInVerticalFrame( 0 ) = sphericalOrbitalState( speedIndex ) *
101  std::cos( sphericalOrbitalState( flightPathIndex ) ) *
102  std::cos( sphericalOrbitalState( headingAngleIndex ) );
103  velocityInVerticalFrame( 1 ) = sphericalOrbitalState( speedIndex ) *
104  std::cos( sphericalOrbitalState( flightPathIndex ) ) *
105  std::sin( sphericalOrbitalState( headingAngleIndex ) );
106  }
107 
108  if( isFlightPathAngleValid )
109  {
110  velocityInVerticalFrame( 2 ) = -sphericalOrbitalState( speedIndex ) *
111  std::sin( sphericalOrbitalState( flightPathIndex ) );
112  }
113 
114  // Set velocity in body-fixed frame.
115  cartesianState.segment( 3, 3 ) = reference_frames::getLocalVerticalToRotatingPlanetocentricFrameTransformationQuaternion(
116  sphericalOrbitalState( longitudeIndex ), sphericalOrbitalState( latitudeIndex ) ) * velocityInVerticalFrame;
117 
118  return cartesianState;
119 }
