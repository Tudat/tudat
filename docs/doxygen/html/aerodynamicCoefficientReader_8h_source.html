31 template< unsigned int NumberOfDimensions >
32 boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) > mergeNDimensionalCoefficients(
33  boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) > xComponents,
34  boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) > yComponents,
35  boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) > zComponents )
36 {
37  boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) > vectorArray;
38 
39  // Check input consistency
40  for( unsigned int i = 0; i < NumberOfDimensions; i++ )
41  {
42  if( !( xComponents.shape( )[ i ] == yComponents.shape( )[ i ] ) ||
43  !( xComponents.shape( )[ i ] == zComponents.shape( )[ i ] ) )
44  {
45  throw std::runtime_error( "Error when creating N-D merged multi-array, input sizes are inconsistent" );
46  }
47  }
48 
49  // Retrieve multi-array shape/size
50  std::vector< size_t > sizeVector;
51  const size_t* arrayShape = xComponents.shape( );
52  sizeVector.assign( arrayShape, arrayShape + xComponents.num_dimensions( ) );
53 
54  // Resize coefficient multi-array
55  vectorArray.resize( sizeVector );
56 
57  // Iterate over all elements and combine x,y and z-components into vector3d of associated entry in vectorVector
58  int numberOfEntries = xComponents.num_elements( );
59  Eigen::Vector3d* vectorVector = new Eigen::Vector3d[ numberOfEntries ] ;
60 
61  typedef typename boost::multi_array< double, NumberOfDimensions >::index tIndex;
62  typedef boost::array< tIndex, NumberOfDimensions > tIndexArray;
63 
64  double* p = xComponents.data( );
65  tIndexArray index;
66  for( int i = 0; i < numberOfEntries; i++ )
67  {
68  index = utilities::getMultiArrayIndexArray( xComponents, p );
69 
70  vectorVector[ i ] = ( Eigen::Vector3d( ) << xComponents( index ), yComponents( index ), zComponents( index ) ).finished( );
71  ++p;
72  }
73 
74  // Assign array of entries to output multi-array
75  vectorArray.assign( vectorVector, vectorVector + numberOfEntries );
76 
77  delete[ ] vectorVector;
78 
79  return vectorArray;
80 }

131 template< unsigned int NumberOfDimensions >
132 std::pair< boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) >,
133 std::vector< std::vector< double > > >
134 readAerodynamicCoefficients( const std::map< int, std::string >& fileNames )
135 {
136  std::map< int, boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) > > rawCoefficientArrays;
137 
138  std::vector< boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) > > coefficientArrays;
139  std::vector< std::vector< double > > independentVariables;
140 
141  // Iterate over files and read the contents into rawCoefficientArrays/independentVariables.
142  for( std::map< int, std::string >::const_iterator fileIterator = fileNames.begin( ); fileIterator != fileNames.end( );
143  fileIterator++ )
144  {
145  // Read current coefficients/independent variables
146  std::pair< boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) >,
147  std::vector< std::vector< double > > > currentCoefficients =
148  MultiArrayFileReader< NumberOfDimensions >::readMultiArrayAndIndependentVariables( fileIterator->second );
149 
150  // Save/check consistency of independent variables
151  if( rawCoefficientArrays.size( ) == 0 )
152  {
153  independentVariables = currentCoefficients.second;
154  }
155  else
156  {
157  bool areIndependentVariablesEqual = compareIndependentVariables(
158  independentVariables, currentCoefficients.second );
159 
160  if( !areIndependentVariablesEqual )
161  {
162  throw std::runtime_error( "Error when reading 1-Dimensional aeroynamic coefficients, inconsistent independent variables." );
163  }
164  }
165 
166  // Save file contents into rawCoefficientArrays
167  utilities::copyMultiArray< double, NumberOfDimensions >(
168  currentCoefficients.first, rawCoefficientArrays[ fileIterator->first ] );
169 
170  }
171 
172  // Check if anything has been read from files.
173  if( rawCoefficientArrays.size( ) == 0 )
174  {
175  throw std::runtime_error( "Error when reading aerodynamic coefficients, no files read" );
176  }
177  else
178  {
179  coefficientArrays.resize( 3 );
180  boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) > firstMultiArray =
181  rawCoefficientArrays.begin( )->second;
182 
183  // Iterate over all 3 coefficient entries
184  for( unsigned int i = 0; i < 3; i++ )
185  {
186  // Copy contents for current index into coefficientArrays read from file.
187  if( rawCoefficientArrays.count( i ) != 0 )
188  {
189  utilities::copyMultiArray< double, NumberOfDimensions >(
190  rawCoefficientArrays.at( i ), coefficientArrays[ i ] );
191  }
192  // Set zero multi-array for current index.
193  else
194  {
195  std::vector< size_t > sizeVector;
196  const size_t* arrayShape = firstMultiArray.shape( );
197  sizeVector.assign( arrayShape, arrayShape+ firstMultiArray.num_dimensions( ) );
198 
199  coefficientArrays[ i ].resize( sizeVector );
200 
201  std::fill( coefficientArrays[ i ].data( ),
202  coefficientArrays[ i ].data() + coefficientArrays[ i ].num_elements( ), 0.0 );
203  }
204  }
205  }
206 
207  // Merge coefficient entries
208  return std::make_pair(
209  mergeNDimensionalCoefficients< NumberOfDimensions >(
210  coefficientArrays.at( 0 ), coefficientArrays.at( 1 ), coefficientArrays.at( 2 ) ), independentVariables );
211 }
