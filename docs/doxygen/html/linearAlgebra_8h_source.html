147 static inline void flipMatrixRows( Eigen::MatrixXd& matrixToFlip )
148 {
149  // Loop through each row.
150  for ( int i = 0; i < static_cast< int >( matrixToFlip.rows( ) / 2 ); i++ )
151  {
152  // Save top row in a temporary matrix.
153  const Eigen::MatrixXd temporaryRow = matrixToFlip.row( i );
154 
155  // Set top row to bottom row.
156  matrixToFlip.row( i ) = matrixToFlip.row( matrixToFlip.rows( ) - 1 - i );
157 
158  // Set bottom row to temporarily stored matrix.
159  matrixToFlip.row( matrixToFlip.rows( ) - 1 - i ) = temporaryRow;
160  }
161 }

189 template< typename StateScalarType, int NumberOfRows, int NumberOfColumns >
190 bool doesMatrixHaveNanEntries( const Eigen::Matrix< StateScalarType, NumberOfRows, NumberOfColumns > matrixToCheck )
191 {
192  bool areNanEntriesPresent = false;
193  for( int i = 0; i < matrixToCheck.rows( ); i++ )
194  {
195  for( int j = 0; j < matrixToCheck.cols( ); j++ )
196  {
197  if( !( matrixToCheck( i, j ) == matrixToCheck( i, j ) ) )
198  {
199  areNanEntriesPresent = true;
200  }
201  }
202  }
203  return areNanEntriesPresent;
204 }
