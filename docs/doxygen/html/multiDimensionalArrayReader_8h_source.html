156  static std::pair< boost::multi_array< double, 1 >, std::vector< std::vector< double > > >
157  readMultiArrayAndIndependentVariables(
158  const std::string fileName )
159  {
160  std::vector< std::vector< double > > independentVariables;
161  Eigen::MatrixXd coefficientBlock;
162 
163  // Read raw data from file
164  readCoefficientsFile( fileName, independentVariables, coefficientBlock );
165 
166  if ( !( independentVariables.size( ) == 1 ) )
167  {
168  throw std::runtime_error( "Error when reading 3-D multi-array, wrong number of independent variables found"
169  " (expected 1 independent variable, got " +
170  std::to_string( independentVariables.size( ) ) + ")" );
171  }
172 
173  // Set sizes of independent variables
174  std::vector< int > independentVariableSize;
175  independentVariableSize.push_back( independentVariables.at( 0 ).size( ) );
176 
177  // Parse data from file
178  return std::make_pair( parseRawOneDimensionalCoefficientsFromFile(
179  independentVariableSize, coefficientBlock ), independentVariables );
180  }
