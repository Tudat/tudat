35 template< typename StateScalarType = double >
36 StateScalarType calculateEnckeQFunction( const StateScalarType qValue )
37 {
38  StateScalarType powerTerm = mathematical_constants::getFloatingInteger< StateScalarType >( 1 ) +
39  mathematical_constants::getFloatingInteger< StateScalarType >( 2 ) * qValue;
40  return mathematical_constants::getFloatingInteger< StateScalarType >( 1 ) - 1.0 / ( powerTerm * std::sqrt( powerTerm ) );
41 }

125  void calculateSystemStateDerivative(
126  const TimeType time,
127  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
128  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
129  {
130  stateDerivative.setZero( );
131 
132  // Retrieve Keplerian orbit state for each body.
133  calculateKeplerTrajectoryCartesianStates( time );
134 
135  // Get Cartesian state derivative for all bodies of Encke state (excluding central gravitational accelerations).
136  this->sumStateDerivativeContributions(
137  stateOfSystemToBeIntegrated, stateDerivative, true );
138 
139  // Initialize Encke algorithm variables.
140  StateScalarType qValue = 0.0;
141  StateScalarType qFunction = 0.0;
142  StateScalarType keplerianRadius = 0.0;
143  Eigen::Matrix< StateScalarType, 3, 1 > positionPerturbation = Eigen::Matrix< StateScalarType, 3, 1 >::Zero( );
144 
145  // Update state derivative for each body.
146  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
147  {
148  // Get position perturbation.
149  positionPerturbation = stateOfSystemToBeIntegrated.segment( i * 6, 3 );
150 
151  // Get distance from central body, assuming purely Keplerian orbit.
152  keplerianRadius = currentKeplerianOrbitCartesianState_[ i ].segment( 0, 3 ).norm( );
153 
154  // Calculate Encke algorithm variables.
155  qValue = positionPerturbation.dot( currentKeplerianOrbitCartesianState_[ i ].segment( 0, 3 ) +
156  0.5 * positionPerturbation ) / ( keplerianRadius * keplerianRadius );
157  qFunction = calculateEnckeQFunction( qValue );
158 
159  // Update state derivative with Encke term.
160  stateDerivative.block( i * 6 + 3, 0, 3, 1 ) += static_cast< StateScalarType >(
161  centralBodyGravitationalParameters_[ i ]( ) ) /
162  ( keplerianRadius * keplerianRadius * keplerianRadius ) * (
163  ( positionPerturbation + currentKeplerianOrbitCartesianState_[ i ].segment( 0, 3 ) ) * qFunction -
164  positionPerturbation );
165  }
166 
167  }
