248  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
249  {
250  // Loop over each body
251  Eigen::Matrix< StateScalarType, 3, 1 > exponentialMapVector;
252  StateScalarType exponentialMapMagnitude;
253  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
254  {
255  // Convert to/from shadow exponential map (SEM) (transformation is the same either way)
256  exponentialMapVector = unprocessedState.block( i * 7 + 3, 0, 3, 1 );
257  exponentialMapMagnitude = exponentialMapVector.norm( );
258  if ( exponentialMapMagnitude >= mathematical_constants::PI )
259  {
260  // Invert flag
261  unprocessedState.segment( i * 7 + 6, 1 ) = ( unprocessedState.block( i * 7 + 6, 0, 1, 1 ) -
262  Eigen::Matrix< StateScalarType, 1, 1 >::Ones( ) ).cwiseAbs( );
263 
264  // Convert to EM/SEM
265  exponentialMapVector *= ( 1.0 - ( 2.0 * mathematical_constants::PI / exponentialMapMagnitude ) );
266 
267  // Replace EM with SEM, or vice-versa
268  unprocessedState.segment( i * 7 + 3, 3 ) = exponentialMapVector;
269  }
270  }
271  }
