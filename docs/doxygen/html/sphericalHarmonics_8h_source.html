176  void updateSines( const double longitude )
177  {
179  if( !( currentLongitude_ == longitude ) )
180  {
181  currentLongitude_ = longitude;
182  for( unsigned int i = 0; i < sinesOfLongitude_.size( ); i++ )
183  {
184  sinesOfLongitude_[ i ] = std::sin( static_cast< double >( i ) * longitude );
185  cosinesOfLongitude_[ i ] = std::cos( static_cast< double >( i ) * longitude );
186  }
187  }
188  }

195  void updateRadiusPowers( const double referenceRadiusRatio )
196  {
198  if( !( referenceRadiusRatio_ == referenceRadiusRatio ) )
199  {
200  referenceRadiusRatio_ = referenceRadiusRatio;
201  double currentRatioPower = 1.0;
202  for( int i = 0; i <= maximumDegree_ + 1; i++ )
203  {
204  referenceRadiusRatioPowers_[ i ] = currentRatioPower;
205  currentRatioPower *= referenceRadiusRatio_;
206  }
207  }
208  }
