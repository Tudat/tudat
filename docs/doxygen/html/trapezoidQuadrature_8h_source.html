41 template< typename IndependentVariableType, typename DependentVariableType >
42 DependentVariableType performTrapezoidalQuadrature(
43  const std::vector< IndependentVariableType >& independentVariables,
44  const std::vector< DependentVariableType >& dependentVariables )
45 {
46  DependentVariableType integral = dependentVariables.at( 0 ) - dependentVariables.at( 0 );
47  IndependentVariableType timeStep;
48  for( unsigned int i = 0 ; i < independentVariables.size( ) - 1 ; i++ )
49  {
50  timeStep = independentVariables[ i + 1 ] - independentVariables[ i ];
51  integral += timeStep * ( dependentVariables[ i + 1 ] + dependentVariables[ i ] ) / 2.0 ;
52  }
53  return integral;
54 }

65 template< typename IndependentVariableType, typename DependentVariableType >
66 DependentVariableType performExtendedSimpsonsQuadrature(
67  const IndependentVariableType constantIndependentVariableStep,
68  const std::vector< DependentVariableType >& dependentVariables )
69 {
70  // Get initial variables
71  DependentVariableType integral = dependentVariables.at( 0 ) - dependentVariables.at( 0 );
72  unsigned int numberOfVariables = dependentVariables.size( );
73 
74  // Check that there are enough elements in the vector
75  if ( numberOfVariables > 6 )
76  {
77  // Create vector of weights
78  std::vector< IndependentVariableType > vectorOfWeights = std::vector< IndependentVariableType >( numberOfVariables,
79  constantIndependentVariableStep );
80 
81  // Add weights
82  vectorOfWeights.at( 0 ) *= 3.0 / 8.0;
83  vectorOfWeights.at( numberOfVariables - 1 ) *= 3.0 / 8.0;
84 
85  vectorOfWeights.at( 1 ) *= 7.0 / 6.0;
86  vectorOfWeights.at( numberOfVariables - 2 ) *= 7.0 / 6.0;
87 
88  vectorOfWeights.at( 2 ) *= 23.0 / 24.0;
89  vectorOfWeights.at( numberOfVariables - 3 ) *= 23.0 / 24.0;
90 
91  // Loop over each time step and return result
92  for( unsigned int i = 0 ; i < numberOfVariables; i++ )
93  {
94  integral += vectorOfWeights[ i ] * dependentVariables[ i ];
95  }
96  }
97  else if ( numberOfVariables != 1 )
98  {
99  // Perform trapezoidal integration instead
100  for( unsigned int i = 0 ; i < ( numberOfVariables - 1 ); i++ )
101  {
102  integral += constantIndependentVariableStep * (
103  dependentVariables[ i + 1 ] + dependentVariables[ i ] ) / 2.0 ;
104  }
105  }
106  else
107  {
108  integral = IdentityElement::getAdditionIdentity< DependentVariableType >( );
109  }
110  return integral;
111 }
