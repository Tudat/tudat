137  void calculateSystemStateDerivative(
138  const TimeType time, const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
139  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
140  {
141  // Get total inertial accelerations acting on bodies
142  stateDerivative.setZero( );
143  this->sumStateDerivativeContributions( stateOfSystemToBeIntegrated, stateDerivative, false );
144 
145  // Compute RSW accelerations for each body, and evaluate Gauss equations for Keplerian elements.
146  Eigen::Vector3d currentAccelerationInRswFrame;
147  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
148  {
149  currentAccelerationInRswFrame = reference_frames::getInertialToRswSatelliteCenteredFrameRotationMatrix(
150  currentCartesianLocalSolution_.segment( i * 6, 6 ) ) *
151  stateDerivative.block( i * 6 + 3, 0, 3, 1 ).template cast< double >( );

154  ( Eigen::Vector6d( ) << stateOfSystemToBeIntegrated.block( i * 6, 0, 5, 1 ).template cast< double >( ),
155  currentTrueAnomalies_.at( i ) ).finished( ), currentAccelerationInRswFrame,
156  centralBodyGravitationalParameters_.at( i )( ) ).template cast< StateScalarType >( );

168  Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > convertFromOutputSolution(
169  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic >& cartesianSolution,
170  const TimeType& time )
171  {
172  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > currentState =
173  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( cartesianSolution.rows( ) );
174 
175  // Subtract frame origin and Keplerian states from inertial state.
176  Eigen::Matrix< StateScalarType, 6, 1 > currentCartesianState;
177  Eigen::Matrix< StateScalarType, 6, 1 > currentKeplerianState;
178 
179  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
180  {
181  currentCartesianState = cartesianSolution.block( i * 6, 0, 6, 1 );
182  currentKeplerianState = orbital_element_conversions::convertCartesianToKeplerianElements (
183  currentCartesianState, static_cast< StateScalarType >(
184  centralBodyGravitationalParameters_.at( i )( ) ) );
185  StateScalarType eccentricAnomaly = orbital_element_conversions::convertTrueAnomalyToEccentricAnomaly(
186  currentKeplerianState( 5 ), currentKeplerianState( 1 ) );
187  StateScalarType meanAnomaly = orbital_element_conversions::convertEccentricAnomalyToMeanAnomaly(
188  eccentricAnomaly, currentKeplerianState( 1 ) );
189  currentKeplerianState( 5 ) = meanAnomaly;
190  currentState.segment( i * 6, 6 ) = currentKeplerianState;
191  }
192 
193  return currentState;
194 
195  }

210  void convertToOutputSolution(
211  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic >& internalSolution, const TimeType& time,
212  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > currentCartesianLocalSolution )
213  {
214  // Add Keplerian state to perturbation from Encke algorithm to get Cartesian state in local frames.
215  Eigen::Matrix< StateScalarType, 6, 1 > currentKeplerianState;
216  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
217  {
218  try
219  {
220  currentKeplerianState = internalSolution.block( i * 6, 0, 6, 1 );
221  StateScalarType currentTrueAnomaly;
222  if( currentKeplerianState( 1 ) < 1.0 )
223  {
224  StateScalarType currentEccentricAnomaly = orbital_element_conversions::convertMeanAnomalyToEccentricAnomaly(
225  currentKeplerianState( 1 ), currentKeplerianState( 5 ) );
226  currentTrueAnomaly = orbital_element_conversions::convertEccentricAnomalyToTrueAnomaly(
227  currentEccentricAnomaly, currentKeplerianState( 1 ) );
228  currentKeplerianState( 5 ) = currentTrueAnomaly;
229  }
230  else
231  {
232  StateScalarType currentEccentricAnomaly = orbital_element_conversions::convertMeanAnomalyToHyperbolicEccentricAnomaly(
233  currentKeplerianState( 1 ), currentKeplerianState( 5 ) );
234  currentTrueAnomaly = orbital_element_conversions::convertHyperbolicEccentricAnomalyToTrueAnomaly(
235  currentEccentricAnomaly, currentKeplerianState( 1 ) );
236  currentKeplerianState( 5 ) = currentTrueAnomaly;
237  }
238 
239  currentTrueAnomalies_[ i ] = currentTrueAnomaly;
240  currentCartesianLocalSolution.segment( i * 6, 6 ) = orbital_element_conversions::convertKeplerianToCartesianElements(
241  currentKeplerianState, static_cast< StateScalarType >( centralBodyGravitationalParameters_.at( i )( ) ) );
242  }
243  catch( std::runtime_error )
244  {
245  currentTrueAnomalies_[ i ] = TUDAT_NAN;
246  currentCartesianLocalSolution.segment( i * 6, 6 ) = Eigen::Matrix< StateScalarType, 6, 1 >::Constant( TUDAT_NAN );
247  }
248 
249  }
250 
251  currentCartesianLocalSolution_ = currentCartesianLocalSolution.template cast< double >( );
252  }
