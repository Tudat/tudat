49 template< typename TimeType = double, typename StateScalarType = double >
50 Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > getInitialStatesOfBodiesFromFrameManager(
51  const std::vector< std::string >& bodiesToIntegrate,
52  const std::vector< std::string >& centralBodies,
53  const simulation_setup::SystemOfBodies& bodies,
54  const TimeType initialTime,
55  const std::shared_ptr< ephemerides::ReferenceFrameManager > frameManager )
56 {
57  // Set initial states of bodies to integrate.
58  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > systemInitialState =
59  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( bodiesToIntegrate.size( ) * 6, 1 );
60  std::shared_ptr< ephemerides::Ephemeris > ephemerisOfCurrentBody;
61 
62  // Iterate over all bodies.
63  for( unsigned int i = 0; i < bodiesToIntegrate.size( ) ; i++ )
64  {
65  ephemerisOfCurrentBody = bodies.at( bodiesToIntegrate.at( i ) )->getEphemeris( );
66 
67  if ( !ephemerisOfCurrentBody )
68  {
69  throw std::runtime_error( "Could not determine initial state for body " + bodiesToIntegrate.at( i ) +
70  " because it does not have a valid Ephemeris object." );
71  }
72 
73  // Get body initial state from ephemeris
74  systemInitialState.segment( i * 6 , 6 ) = ephemerisOfCurrentBody->getTemplatedStateFromEphemeris<
75  StateScalarType, TimeType >( initialTime );
76 
77  // Correct initial state if integration origin and ephemeris origin are not equal.
78  if( centralBodies.at( i ) != ephemerisOfCurrentBody->getReferenceFrameOrigin( ) )
79  {
80  std::shared_ptr< ephemerides::Ephemeris > correctionEphemeris =
81  frameManager->getEphemeris( ephemerisOfCurrentBody->getReferenceFrameOrigin( ), centralBodies.at( i ) );
82  systemInitialState.segment( i * 6 , 6 ) -= correctionEphemeris->getTemplatedStateFromEphemeris<
83  StateScalarType, TimeType >( initialTime );
84  }
85  }
86  return systemInitialState;
87 }

99 template< typename TimeType = double, typename StateScalarType = double >
100 Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > getInitialRotationalStatesOfBodies(

109  std::shared_ptr< ephemerides::RotationalEphemeris > rotationModelOfCurrentBody;
110 
111  // Iterate over all bodies.
112  for( unsigned int i = 0; i < bodiesToIntegrate.size( ) ; i++ )
113  {
114  rotationModelOfCurrentBody = bodies.at( bodiesToIntegrate.at( i ) )->getRotationalEphemeris( );

118  throw std::runtime_error( "Could not determine initial state for body " + bodiesToIntegrate.at( i ) +

403  integratorSettings_( integratorSettings ),
404  propagatorSettings_(
405  std::dynamic_pointer_cast< SingleArcPropagatorSettings< StateScalarType > >( propagatorSettings ) ),
406  initialPropagationTime_( integratorSettings_->initialTime_ ),
407  printNumberOfFunctionEvaluations_( printNumberOfFunctionEvaluations ), initialClockTime_( initialClockTime ),

416  {
417  throw std::runtime_error( "Error in dynamics simulator, input must be single-arc." );
418  }
419 
420  if( integratorSettings == nullptr )
421  {
422  throw std::runtime_error( "Error in dynamics simulator, integrator settings not defined." );
423  }
424 
425  if( setIntegratedResult_ )

427  frameManager_ = simulation_setup::createFrameManager( bodies.getMap( ) );

429  propagatorSettings_, bodies_, frameManager_ );

431 
432  try
433  {
434  environmentUpdater_ = createEnvironmentUpdaterForDynamicalEquations< StateScalarType, TimeType >(
435  propagatorSettings_, bodies_ );
436  }
437  catch( const std::runtime_error& error )
438  {
439  throw std::runtime_error( "Error when creating environment updater: " + std::string( error.what( ) ) );
440  }
441 
442 
443  if( stateDerivativeModels.size( ) == 0 )
444  {
445  dynamicsStateDerivative_ = std::make_shared< DynamicsStateDerivativeModel< TimeType, StateScalarType > >(
446  createStateDerivativeModels< StateScalarType, TimeType >(
447  propagatorSettings_, bodies_, initialPropagationTime_ ),
448  std::bind( &EnvironmentUpdater< StateScalarType, TimeType >::updateEnvironment,

459  propagationTerminationCondition_ = createPropagationTerminationConditions(

509  void integrateEquationsOfMotion(
510  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic >& initialStates )
511  {
512 
513  // Empty solution maps
514  equationsOfMotionNumericalSolution_.clear( );
515  equationsOfMotionNumericalSolutionRaw_.clear( );
516 
517  // Reset functions
518  dynamicsStateDerivative_->setPropagationSettings( std::vector< IntegratedStateType >( ), 1, 0 );
519  dynamicsStateDerivative_->resetFunctionEvaluationCounter( );
520  dynamicsStateDerivative_->resetCumulativeFunctionEvaluationCounter( );
521 
522  // Reset initial time to ensure consistency with multi-arc propagation.
523  integratorSettings_->initialTime_ = this->initialPropagationTime_;
524 
525  // Integrate equations of motion numerically.
526  resetPropagationTerminationConditions( );
527  simulation_setup::setAreBodiesInPropagation( bodies_, true );
528  propagationTerminationReason_ =
529  EquationIntegrationInterface< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >, TimeType >::integrateEquations(
530  stateDerivativeFunction_, equationsOfMotionNumericalSolutionRaw_,
531  dynamicsStateDerivative_->convertFromOutputSolution(
532  initialStates, this->initialPropagationTime_ ), integratorSettings_,
533  propagationTerminationCondition_,
534  dependentVariableHistory_,
535  cumulativeComputationTimeHistory_,
536  dependentVariablesFunctions_,
537  statePostProcessingFunction_,
538  propagatorSettings_->getPrintInterval( ),
539  initialClockTime_ );
540  simulation_setup::setAreBodiesInPropagation( bodies_, false );
541 
542  // Convert numerical solution to conventional state
543  dynamicsStateDerivative_->convertNumericalStateSolutionsToOutputSolutions(
544  equationsOfMotionNumericalSolution_, equationsOfMotionNumericalSolutionRaw_ );
545 
546  // Retrieve number of cumulative function evaluations
547  cumulativeNumberOfFunctionEvaluations_ = dynamicsStateDerivative_->getCumulativeNumberOfFunctionEvaluations( );
548 
549  // Retrieve and print number of total function evaluations
550  if ( printNumberOfFunctionEvaluations_ )
551  {
552  std::cout << "Total Number of Function Evaluations: "
553  << dynamicsStateDerivative_->getNumberOfFunctionEvaluations( ) << std::endl;
554  }
555 
556  if( this->setIntegratedResult_ )
557  {
558  processNumericalEquationsOfMotionSolution( );
559  }
560  }

987 Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > getArcInitialStateFromPreviousArcResult(
988  const std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >& previousArcDynamicsSolution,
989  const double currentArcInitialTime )

995  {

1003

1005

1007  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >::
1008  const_reverse_iterator previousArcIterator = previousArcDynamicsSolution.rbegin( );
1009  previousArcIterator != previousArcDynamicsSolution.rend( ); previousArcIterator++ )
1010  {
1011  initialStateInterpolationMap[ previousArcIterator->first ] = previousArcIterator->second;
1012  if( initialTimeIndex < 0 )
1013  {
1014  if( previousArcIterator->first < currentArcInitialTime )
1015  {
1016  initialTimeIndex = currentIndex;
1017  }
1018  }
1019  else
1020  {

1022  {
1023  break;
1024  }
1025  }
1026  currentIndex++;
1027  }
1028

1030  currentArcInitialState =
1031  std::make_shared< interpolators::LagrangeInterpolator<
1032  TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >, long double > >(
1033  initialStateInterpolationMap, 8 )->interpolate( currentArcInitialTime );
1034 
1035  }
