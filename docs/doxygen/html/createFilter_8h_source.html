255 template< typename IndependentVariableType = double, typename DependentVariableType = double >
256 std::shared_ptr< filters::FilterBase< IndependentVariableType, DependentVariableType > >
257 createFilter( const std::shared_ptr< FilterSettings< IndependentVariableType, DependentVariableType > > filterSettings,
258  const std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >(
259  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >& systemFunction,
260  const std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >(
261  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >& measurementFunction,
262  const std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
263  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >&
264  stateJacobianFunction = std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
265  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >( ),
266  const std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
267  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >&
268  stateNoiseJacobianFunction = std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
269  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >( ),
270  const std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
271  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >&
272  measurementJacobianFunction = std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
273  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >( ),
274  const std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
275  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >&
276  measurementNoiseJacobianFunction = std::function< Eigen::Matrix< DependentVariableType, Eigen::Dynamic, Eigen::Dynamic >(
277  const IndependentVariableType, const Eigen::Matrix< DependentVariableType, Eigen::Dynamic, 1 >& ) >( ) )
278 {
279  // Create an empty filter object
280  std::shared_ptr< filters::FilterBase< IndependentVariableType, DependentVariableType > > createdFilter;
281 
282  // Check type of filter
283  switch ( filterSettings->filteringTechnique_ )
284  {
285  case extended_kalman_filter:
286  {
287  // Cast filter settings to extended Kalman filter
288  std::shared_ptr< ExtendedKalmanFilterSettings< IndependentVariableType, DependentVariableType > > extendedKalmanFilterSettings =
289  std::dynamic_pointer_cast< ExtendedKalmanFilterSettings< IndependentVariableType, DependentVariableType > >(
290  filterSettings );
291  if ( extendedKalmanFilterSettings == nullptr )
292  {
293  throw std::runtime_error( "Error while creating extended Kalman filter object. Type of filter settings "
294  "(ExtendedKalmanFilter) not compatible with selected filter (derived class of FilterSettings "
295  "must be ExtendedKalmanFilterSettings for this type)." );
296  }
297 
298  // Check that optional inputs are present
299  if ( ( stateJacobianFunction == nullptr ) || ( stateNoiseJacobianFunction == nullptr ) ||
300  ( measurementJacobianFunction == nullptr ) || ( measurementNoiseJacobianFunction == nullptr ) )
301  {
302  throw std::runtime_error( "Error while creating extended Kalman filter object. An ExtendedKalmanFilter object "
303  "requires the input of the four Jacobian functions for state and measurement (including noise)." );
304  }
305 
306  // Create filter
307  createdFilter = std::make_shared< ExtendedKalmanFilter< IndependentVariableType, DependentVariableType > >(
308  systemFunction, measurementFunction, stateJacobianFunction, stateNoiseJacobianFunction,
309  measurementJacobianFunction, measurementNoiseJacobianFunction,
310  extendedKalmanFilterSettings->systemUncertainty_, extendedKalmanFilterSettings->measurementUncertainty_,
311  extendedKalmanFilterSettings->filteringStepSize_, extendedKalmanFilterSettings->initialTime_,
312  extendedKalmanFilterSettings->initialStateEstimate_, extendedKalmanFilterSettings->initialCovarianceEstimate_,
313  extendedKalmanFilterSettings->integratorSettings_ );
314  break;
315  }
316  case unscented_kalman_filter:
317  {
318  // Cast filter settings to unscented Kalman filter
319  std::shared_ptr< UnscentedKalmanFilterSettings< IndependentVariableType, DependentVariableType > > unscentedKalmanFilterSettings =
320  std::dynamic_pointer_cast< UnscentedKalmanFilterSettings< IndependentVariableType, DependentVariableType > >(
321  filterSettings );
322  if ( unscentedKalmanFilterSettings == nullptr )
323  {
324  throw std::runtime_error( "Error while creating unscented Kalman filter object. Type of filter settings "
325  "(UnscentedKalmanFilter) not compatible with selected filter (derived class of FilterSettings "
326  "must be UnscentedKalmanFilterSettings for this type)." );
327  }
328 
329  // Create filter
330  createdFilter = std::make_shared< UnscentedKalmanFilter< IndependentVariableType, DependentVariableType > >(
331  systemFunction, measurementFunction,
332  unscentedKalmanFilterSettings->systemUncertainty_, unscentedKalmanFilterSettings->measurementUncertainty_,
333  unscentedKalmanFilterSettings->filteringStepSize_, unscentedKalmanFilterSettings->initialTime_,
334  unscentedKalmanFilterSettings->initialStateEstimate_, unscentedKalmanFilterSettings->initialCovarianceEstimate_,
335  unscentedKalmanFilterSettings->integratorSettings_, unscentedKalmanFilterSettings->constantValueReference_,
336  unscentedKalmanFilterSettings->customConstantParameters_ );
337  break;
338  }
339  default:
340  throw std::runtime_error( "Error while creating filter obejct. The creation of linear filters is not yet supported." );
341  }
342 
343  // Check that filter was properly created
344  if ( createdFilter == nullptr )
345  {
346  throw std::runtime_error( "Error while creating filter. The resulting filter pointer is null." );
347  }
348 
349  // Give output
350  return createdFilter;
351 }
