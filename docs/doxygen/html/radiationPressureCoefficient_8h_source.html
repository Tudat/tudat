98  ArcWiseRadiationPressureCoefficient(

101  const std::string& associatedBody ):
102  EstimatableParameter< Eigen::VectorXd >( arc_wise_radiation_pressure_coefficient, associatedBody ),
103  radiationPressureInterface_( radiationPressureInterface ), timeLimits_( timeLimits )
104  {
105  double radiationPressureCoefficient = radiationPressureInterface->getRadiationPressureCoefficient( );
106  for( unsigned int i = 0; i < timeLimits.size( ); i++ )
107  {
108  radiationPressureCoefficients_.push_back( radiationPressureCoefficient );
109  }
110 
111  timeLimits_.push_back( std::numeric_limits< double >::max( ) );
112  fullRadiationPressureCoefficients_ = radiationPressureCoefficients_;
113  fullRadiationPressureCoefficients_.push_back( radiationPressureCoefficient );
114 
115 
116  coefficientInterpolator_ = std::make_shared< interpolators::PiecewiseConstantInterpolator< double, double > >(
117  timeLimits_, fullRadiationPressureCoefficients_ );
118 
119  typedef interpolators::OneDimensionalInterpolator< double, double > LocalInterpolator;
120  radiationPressureInterface->resetRadiationPressureCoefficientFunction(
121  std::bind(
122  static_cast< double( LocalInterpolator::* )( const double ) >
123  ( &LocalInterpolator::interpolate ), coefficientInterpolator_, std::placeholders::_1 ) );
124  }

146  if( static_cast< int >( radiationPressureCoefficients_.size( ) ) !=
147  static_cast< int >( parameterValue.rows( ) ) )
148  {
149  throw std::runtime_error( "Error when resetting arc-wise radiation pressure coefficients, sizes are incompatible" );

152  radiationPressureCoefficients_ = utilities::convertEigenVectorToStlVector( parameterValue );
153  for( unsigned int i = 0; i < radiationPressureCoefficients_.size( ); i++ )
154  {
155  fullRadiationPressureCoefficients_[ i ] = radiationPressureCoefficients_[ i ];
156  }
157  fullRadiationPressureCoefficients_[ radiationPressureCoefficients_.size( ) ] =
158  radiationPressureCoefficients_.at( radiationPressureCoefficients_.size( ) - 1 );
159  coefficientInterpolator_->resetDependentValues( fullRadiationPressureCoefficients_ );
160  }
