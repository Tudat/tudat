101  Eigen::Block< Eigen::MatrixXd > partialMatrix,
102  const bool addContribution = 1, const int startRow = 0, const int startColumn = 3 )
103  {
104  if( addContribution )
105  {
106  partialMatrix.block( startRow, startColumn, 3, 3 ) += currentPartialDerivativeWrtAngularVelocity_;
107  }

119  void update( const double currentTime = TUDAT_NAN )
120  {
121  if( !( currentTime_ == currentTime ) )
122  {
123  currentAngularVelocityVector_ = angularVelocityFunction_( );
124  currentAngularVelocityCrossProductMatrix_ = linear_algebra::getCrossProductMatrix(
125  currentAngularVelocityVector_ );
126 
127  currentInertiaTensorNormalizationFactor_ = getInertiaTensorNormalizationFactor_( );
128  currentGravitationalParameter_ = bodyGravitationalParameterFunction_( );
129 
130  currentInertiaTensor_ = inertiaTensorFunction_( );
131  currentInverseInertiaTensor_ = currentInertiaTensor_.inverse( );
132 
133  currentPartialDerivativeWrtAngularVelocity_ =
134  -linear_algebra::getCrossProductMatrix( currentAngularVelocityVector_ ) * currentInertiaTensor_ +
135  linear_algebra::getCrossProductMatrix( currentInertiaTensor_ * currentAngularVelocityVector_ );
136  }
