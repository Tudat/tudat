232  virtual StateType performIntegrationStep( const TimeStepType stepSize )
233  {
234  if( !( stepSize == stepSize) )
235  {
236  throw std::runtime_error( "Error in ABM integrator, step size is NaN" );
237  }
238 
239  // If stepSize is not same as old, clear the step-size dependent histories.
240  if ( stepSize != stepSize_ )
241  {
242  // Pop all values from the history deque (the history is
243  // invalid as it is dependent on the stepSize), except for
244  // the current state and state derivative.
245  while ( stateHistory_.size( ) > 1 )
246  {
247  stateHistory_.pop_back( );
248  }
249  while ( derivHistory_.size( ) > 1 )
250  {
251  derivHistory_.pop_back( );
252  }
253  stepSize_ = stepSize;
254 
255  // Allow single step integrator to determine own stepsize
256  fixedSingleStep_ = fixedStepSize_;
257  }
258  return performIntegrationStep( );
259  }

267  StateType performIntegrationStep( )
268  {
269  // Set last* variables for rollback.
270  lastStepSize_ = stepSize_;
271  lastState_ = stateHistory_.back( );
272  lastDerivative_ = derivHistory_.back( );
273  lastIndependentVariable_ = currentIndependentVariable_;
274 
275  // Remove old elements so enough are left to calculate predicted and corrected.
276  // max twice the order, to facilitatie a doubling, halving, and order change.
277  while ( stateHistory_.size( ) > order_ * 2 )
278  {
279  stateHistory_.pop_back( );
280  }
281  while ( derivHistory_.size( ) > order_ * 2 )
282  {
283  derivHistory_.pop_back( );
284  }
285  unsigned int sizeStateHistory = stateHistory_.size( );
286  unsigned int sizeDerivativeHistory = derivHistory_.size( );
287  unsigned int possibleOrder = std::min( sizeStateHistory, sizeDerivativeHistory );
288 
289  StateType correctedState;
290  StateType predictedState;
291 
292  // Check if enough history steps are available to perform AM
293  // step if not use a single-step method.
294  if ( possibleOrder < minimumOrder_ || possibleOrder < order_ )
295  {
296  correctedState = performSingleStep( );
297  }
298  else
299  {
300  StateType predictedState = performPredictorStep( order_, false );
301  predictedDerivative_ = this->stateDerivativeFunction_( currentIndependentVariable_ +
302  stepSize_, predictedState );
303  correctedState = performCorrectorStep( predictedState, order_, false );
304  absoluteError_ = estimateAbsoluteError( predictedState, correctedState, order_ );
305  relativeError_ = estimateRelativeError( predictedState, correctedState, absoluteError_ );
306  }
307 
308  // Change order to one that gives a higher predicted accuracy
309  // Add tolenaces
310  StateType predictorAbsoluteError;
311  StateType predictorRelativeError;
312 
313  // If order is not fixed, order is not max yet and enough
314  // history is available, then predict the error of an order
315  // more.
316  if ( !fixedOrder_ && order_ < maximumOrder_ && order_ < possibleOrder )
317  {
318  predictedState = performPredictorStep( order_ + 1, false );
319  correctedState = performCorrectorStep( predictedState, order_ + 1, false );
320  predictorAbsoluteError = estimateAbsoluteError( predictedState, correctedState, order_ + 1 );
321  predictorRelativeError = estimateRelativeError( predictedState, correctedState, predictorAbsoluteError );
322 
323  // If the predicted error is less than the current error,
324  // increase the error.
325  if ( errorCompare( predictorAbsoluteError, predictorRelativeError, absoluteError_, relativeError_ ) )
326  {
327 
328  order_++;
329  }
330  // Else if the order is not fixed, the order is not min yet
331  // and there is enough history available, then predict the
332  // error of an order less.
333  }
334  else if ( !fixedOrder_ && order_ > minimumOrder_ && order_ - 1 <= possibleOrder )
335  {
336  predictedState = performPredictorStep( order_ - 1, false );
337  correctedState = performCorrectorStep( predictedState, order_ - 1, false );
338  predictorAbsoluteError = estimateAbsoluteError( predictedState, correctedState, order_ - 1 );
339  predictorRelativeError = estimateRelativeError( predictedState, correctedState, predictorAbsoluteError );
340  // If it is less than the current order, lower the order.
341  if ( errorCompare( predictorAbsoluteError, predictorRelativeError, absoluteError_, relativeError_ ) )
342  {
343  order_--;
344  } else {
345  predictorAbsoluteError = absoluteError_;
346  predictorRelativeError = relativeError_;
347  }
348  }
349  else
350  {
351  predictorAbsoluteError = absoluteError_;
352  predictorRelativeError = relativeError_;
353  }
354 
355  // If the error (after order change) is too big, stepsize
356  // isn't fixed and will not become too small, then halve the
357  // stepsize.
358  if ( errorTooLarge( predictorAbsoluteError, predictorRelativeError )
359  && std::fabs( stepSize_ / 2.0 )> minimumStepSize_ && !fixedStepSize_ )
360  {
361  // Set up new data for halving
362  std::deque< StateType > tempStateHistory;
363  std::deque< StateType > tempDerivativeHistory;
364  StateType midState = currentState_;
365  StateDerivativeType midDerivative = midState;
366  unsigned int interpolationStateIndex;
367  unsigned int interpolationDerivativeIndex;
368 
369  // Only if there is enough historical data, is it possible to create the intermediate
370  // points. Redefine order based on available data, it could drop, but can't be lower
371  // that minimum order
372  unsigned int possibleHalvingOrder = std::min( 2 * possibleOrder - 1, maximumOrder_ );
373  order_ = std::min( possibleHalvingOrder, order_ );
374 
375  // FIXME: make this a setting?
376  // Reduce order. Too much backlog aversly affects the accuracy when halving.
377  if( order_ > minimumOrder_ )
378  {
379  order_ = minimumOrder_;
380  }
381 
382  for( unsigned int i = 0; i < possibleHalvingOrder ; i++ )
383  {
384  // If states are even, they already exist, no need to interpolate
385  if ( i % 2 == 0 )
386  {
387  tempStateHistory.push_back( stateHistory_.at( i / 2 ));
388  tempDerivativeHistory.push_back( derivHistory_.at( i / 2 ));
389  } else {
390  // Reset midpoint state and deriv to zero
391  midState = 0 * midState;
392  midDerivative = midState;
393  interpolationDerivativeIndex = ( order_ - 1 ) * ( order_ - 1 ) + ( i - 1 ) / 2;
394  interpolationStateIndex = interpolationDerivativeIndex - order_ + 1;
395  for( unsigned int j = 0; j < order_; j++ )
396  {
397  midState += interpolationCoefficients[ interpolationStateIndex ][ j ] *
398  stateHistory_.at( j ) + interpolationCoefficients[ interpolationStateIndex ][ order_ + j ] *
399  derivHistory_.at( j ) * stepSize_;
400  midDerivative += interpolationCoefficients[ interpolationDerivativeIndex ][ j ] *
401  stateHistory_.at( j ) / stepSize_
402  + interpolationCoefficients[ interpolationDerivativeIndex ][ order_ + j ] *
403  derivHistory_.at( j );
404  }
405  tempStateHistory.push_back( midState );
406  tempDerivativeHistory.push_back( midDerivative );
407  }
408  }
409 
410  // Set the new history and stepsize
411  stateHistory_ = tempStateHistory;
412  derivHistory_ = tempDerivativeHistory;
413  stepSize_ = stepSize_ / 2.0;
414 
415  // Temporarily turn halving off.
416  fixedStepSize_ = true;
417  StateType halvedState = performIntegrationStep( );
418  fixedStepSize_ = false;
419  return halvedState;
420  } // end if ( errorTooLarge( ...
421 
422  // If the error (after order change ) is too small, the
423  // stepsize isn't fixed and the and will not become too big,
424  // then double the stepsize.
425  if ( errorTooSmall( predictorAbsoluteError, predictorRelativeError )
426  && sizeDerivativeHistory >= 2 * order_
427  && std::fabs( stepSize_ * 2.0 ) <= maximumStepSize_ && !fixedStepSize_ )
428  {
429 
430  // Predict error after doubling, to prevent error from becoming too big
431  // This prevents fluttering and throwing away states from the history that will be
432  // needed again later.
433  // Note that two important assumptions are made here:
434  // 1. The error made at the current step is representative of the error that
435  // would be made if we blindly double and only compute the error next step.
436  // 2. The difference in the derivative of the predicted state (predictedDerivative_)
437  // at the normal stepsize (already computed) with the doubled stepsize (not computed)
438  // is neglibile. This assumption saves one function evaluation.
439  // It's possible to reuse previously defined variables here except for correctedState
440  // which is still used below.
441  predictedState = performPredictorStep( order_ , true );
442  StateType doubleStepCorrectedState = performCorrectorStep( predictedState, order_, true );
443  predictorAbsoluteError = estimateAbsoluteError( predictedState, doubleStepCorrectedState, order_ );
444  predictorRelativeError = estimateRelativeError( predictedState, doubleStepCorrectedState,
445  predictorAbsoluteError );
446 
447  // Only update the history if the error will not be too large
448  if ( !errorTooLarge( predictorAbsoluteError, predictorRelativeError ) )
449  {
450  // Note that the history should be at least 7 to allow successful
451  // continuation of the AM scheme.
452  std::deque< StateType > tempStateHistory;
453  std::deque< StateType > tempDerivativeHistory;
454 
455  // Use old history to fill new history, skipping every other entry starting at 1
456  for( unsigned int i = 1; i < sizeStateHistory; i += 2 )
457  {
458  tempStateHistory.push_back( stateHistory_.at( i ));
459  }
460  for( unsigned int i = 1; i < sizeDerivativeHistory; i += 2 )
461  {
462  tempDerivativeHistory.push_back( derivHistory_.at( i ));
463  }
464 
465  // Set the new history and stepsize
466  stateHistory_ = tempStateHistory;
467  derivHistory_ = tempDerivativeHistory;
468  stepSize_ = stepSize_ * 2.0;
469  }
470  } // end if ( errorTooSmall( ...
471 
472  // Move computed state to history
473  currentIndependentVariable_ += lastStepSize_;
474  currentState_ = correctedState;
475  stateHistory_.push_front( currentState_ );
476  derivHistory_.push_front( this->stateDerivativeFunction_(
477  currentIndependentVariable_, currentState_ ) );
478  return currentState_;
479  }

864  StateType performPredictorStep( unsigned int order, bool doubleStep )
865  {
866  // Calculate predicted state
867  unsigned int stepsToSkip = static_cast< unsigned int>( doubleStep );
868  TimeStepType stepSize = stepSize_ * static_cast< double >( stepsToSkip + 1 );
869  StateType predictedState = stateHistory_.at( stepsToSkip );
870  for ( unsigned int i = 0; i < order; i++ )
871  {
872  predictedState += extrapolationCoefficients[ order * 2 - 2 ][ i ] * stepSize *
873  derivHistory_.at( i * ( stepsToSkip + 1 ) + stepsToSkip );
874  }
875  return predictedState;
876  }

886  StateType performCorrectorStep( StateType predictedState, unsigned int order, bool doubleStep )
887  {
888  unsigned int stepsToSkip = static_cast< unsigned int>( doubleStep );
889  TimeStepType stepSize = stepSize_ * static_cast< double >( stepsToSkip + 1 );
890  StateType correctedState = stateHistory_.at( stepsToSkip ) + extrapolationCoefficients[ order * 2 - 1 ][ 0 ] *
891  stepSize * predictedDerivative_;
892  for ( unsigned int i = 1; i < order; i++ )
893  {
894  correctedState += stepSize * extrapolationCoefficients[ order * 2 - 1 ][ i ] *
895  derivHistory_.at( ( i - 1 ) * ( stepsToSkip + 1 ) + stepsToSkip );
896  }
897  return correctedState;
898  }

938  bool errorCompare( StateType absoluteError1, StateType relativeError1,
939  StateType absoluteError2, StateType relativeError2 )
940  {
941  // Find compound error
942  StateType c1 = absoluteError1.cwiseMin( relativeError1 );
943  StateType c2 = absoluteError2.cwiseMin( relativeError2 );
944  bool oneBetter = true;
945  if( strictCompare_ )
946  {
947  // Needs to be better or equal for each component
948  for( int i = 0; i < c1.size( ); ++i )
949  {
950  oneBetter = oneBetter && ( c1( i ) <= c2( i ) );
951  }
952  } else {
953  // Needs to be overal better
954  oneBetter = ( c1.norm( ) <= c2.norm( ) );
955  }
956  return oneBetter;
957  }
