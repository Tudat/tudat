240  SphericalHarmonicAccelerationTermsDependentVariableSaveSettings(

244  const int maximumOrder,
245  const int componentIndex = -1,
246  const bool useAccelerationNorm = false ):

253  for( int i = 0; i <= maximumDegree; i++ )

255  for( int j = 0; ( j <= i && j <= maximumOrder ); j++ )

257  componentIndices_.push_back( std::make_pair( i, j ) );

259  }
260  }
261 
263  std::vector< std::pair< int, int > > componentIndices_;
