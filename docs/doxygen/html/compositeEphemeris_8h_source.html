64  CompositeEphemeris(
65  const std::map< int, std::function< StateType( const TimeType& ) > >
66  translationalEphemerides,
67  const std::map< int, std::function< StateType( const TimeType, const StateType& ) > >
68  rotationalEphemerides,
69  const std::string referenceFrameOrigin = "SSB",
70  const std::string referenceFrameOrientation = "ECLIPJ2000" ):
71  Ephemeris( referenceFrameOrigin, referenceFrameOrientation )
72  {
73  // Create iterators over ephemeris functions.
74  typename std::map< int, std::function< StateType( const TimeType& ) > >::const_iterator
75  translationIterator = translationalEphemerides.begin( );
76  typename std::map< int, std::function< StateType( const TimeType, const StateType& ) > >::const_iterator
77  rotationIterator = rotationalEphemerides.begin( );
78 
79  // Check whether chain starts with translation.
80  if( translationIterator->first != 0 )
81  {
82  std::string errorMessage = "Error, composite ephemeris must start with translation";
83  throw std::runtime_error( errorMessage );
84  }
85 
86  // Run over all indices and set order.
87  int currentIndex = 0;
88  while( currentIndex < static_cast< int >( translationalEphemerides.size( )
89  + rotationalEphemerides.size( ) ) )
90  {
91  // If current ephemeris is translational, add to translations list (set as addition) and
92  // set translation flag at current index to true.
93  if( translationIterator != translationalEphemerides.end( )
94  && translationIterator->first == currentIndex )
95  {
96  translationalEphemerides_.push_back( std::make_pair( translationIterator->second, 1 ) );
97  isCurrentEphemerisTranslational_.push_back( 1 );
98  translationIterator++;
99  }
100  // If current ephemeris is rotational, add to rotations list and set translation flag at
101  // current index to false.
102  else if( rotationIterator != rotationalEphemerides.end( )
103  && rotationIterator->first == currentIndex )
104  {
105  rotationalEphemerides_.push_back( rotationIterator->second );
106  isCurrentEphemerisTranslational_.push_back( 0 );
107  rotationIterator++;
108  }
109  // If index is not found, display error message.
110  else
111  {
112  std::string errorMessage = "Error when making composite ephemeris, input indices inconsistentn";
113  throw std::runtime_error( errorMessage );
114  }
115  currentIndex++;
116  }
117 
118  }

252  template< typename OutputTimeType, typename OutputStateScalarType >
253  Eigen::Matrix< OutputStateScalarType, 6, 1 > getTemplatedCartesianStateFromCompositeEphemeris(
254  const OutputTimeType& currentTime )
255  {
256 
257  // Initialize state to zero;
258  Eigen::Matrix< StateScalarType, 6, 1 > state = Eigen::Matrix< StateScalarType, 6, 1 >::Zero( );
259 
260  // Initialize current indices of translation and rotation to zero.
261  int currentTranslationIndex = 0;
262  int currentRotationIndex = 0;
263 
264  // Loop over all ephemerides
265  for( unsigned int i = 0; i < isCurrentEphemerisTranslational_.size( ); i++ )
266  {
267  // If current ephemeris is translational, add it and increment currentTranslationIndex
268  if( isCurrentEphemerisTranslational_[ i ] == true )
269  {
270  state += translationalEphemerides_[ currentTranslationIndex ].first( static_cast< TimeType >( currentTime ) )*
271  static_cast< double >( translationalEphemerides_[ currentTranslationIndex ].second );
272  currentTranslationIndex++;
273  }
274  // If current ephemeris is rotational, multiply position and state by rotation.
275  else
276  {
277  state = rotationalEphemerides_[ currentRotationIndex ]( static_cast< TimeType >( currentTime ), state );
278  currentRotationIndex++;
279  }
280  }
281 
282  return state.template cast< OutputStateScalarType >( );
283  }

354 template< typename TimeType = double, typename StateScalarType = double >
355 std::shared_ptr< Ephemeris > createReferencePointEphemeris(
356  std::shared_ptr< Ephemeris > bodyEphemeris,
357  std::shared_ptr< RotationalEphemeris > bodyRotationModel,
358  std::function< Eigen::Vector6d( const double& ) > referencePointRelativeStateFunction )
359 {
360  typedef Eigen::Matrix< StateScalarType, 6, 1 > StateType;
361 
362  // Cast state fucntion of body (global) and reference point (local) into correct form.
363  std::map< int, std::function< StateType( const TimeType& ) > > referencePointEphemerisVector;
364  referencePointEphemerisVector[ 2 ] = std::bind(
365  &Ephemeris::getTemplatedStateFromEphemeris< StateScalarType, TimeType >, bodyEphemeris, std::placeholders::_1 );
366  referencePointEphemerisVector[ 0 ] = std::bind(
367  &convertStateFunctionStateScalarOutput< double, StateScalarType, TimeType, 6 >,
368  referencePointRelativeStateFunction, std::placeholders::_1 );
369 
370 
371  // Crate rotation functions from local to global frame.
372  std::function< Eigen::Quaterniond( const TimeType ) > rotationToFrameFunction =
373  std::bind( &RotationalEphemeris::getRotationToBaseFrameTemplated< TimeType >, bodyRotationModel, std::placeholders::_1 );
374  std::function< Eigen::Matrix3d( const TimeType ) > rotationMatrixToFrameDerivativeFunction =
375  std::bind( &RotationalEphemeris::getDerivativeOfRotationToBaseFrameTemplated< TimeType >, bodyRotationModel, std::placeholders::_1 );
376 
377  // Create ephemeris
378  std::map< int, std::function< StateType( const TimeType, const StateType& ) > > referencePointRotationVector;
379  referencePointRotationVector[ 1 ] = std::bind(
380  transformStateToFrameFromRotationTimeFunctions< StateScalarType, TimeType >,
381  std::placeholders::_2, std::placeholders::_1, rotationToFrameFunction, rotationMatrixToFrameDerivativeFunction );
382 
383  return std::make_shared< CompositeEphemeris< TimeType, StateScalarType > >(
384  referencePointEphemerisVector, referencePointRotationVector, "SSB", "ECLIPJ2000" );
