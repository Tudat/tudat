264  void update( const double time )
265  {
266  if( !( currentTime_ == time ) )
267  {
268 
269  Eigen::VectorXd costates_ = costateFunction_( time );
270 
271  // Get the current state in cartesian coordinates and keplerian elements, and some convenient parameters
272  Eigen::Vector6d currentState = thrustingBodyStateFunction_( ) - centralBodyStateFunction_( );
273  double centralBodyGravitationalParameter = centralBodyGravitationalParameterFunction_( );
274 
275  // Obtain ModifiedEquinoctial elements, flag of 0 indicates that singularity occurs at 180 deg inclination.
276  Eigen::Vector6d modifiedEquinoctialElements =
277  orbital_element_conversions::convertCartesianToModifiedEquinoctialElements(
278  currentState, centralBodyGravitationalParameter, 0 );
279 
280  // Retrieve modified equinoctial elements.
281  double p = modifiedEquinoctialElements[ orbital_element_conversions::semiParameterIndex ];
282  double f = modifiedEquinoctialElements[ orbital_element_conversions::fElementIndex ];
283  double g = modifiedEquinoctialElements[ orbital_element_conversions::gElementIndex ];
284  double h = modifiedEquinoctialElements[ orbital_element_conversions::hElementIndex ];
285  double k = modifiedEquinoctialElements[ orbital_element_conversions::kElementIndex ];
286  double L = modifiedEquinoctialElements[ orbital_element_conversions::trueLongitudeIndex ];
287 
288  double w1 = 1.0 + f * std::cos( L ) + g * std::sin( L );
289  double w2 = 1.0 + h * h + k * k;
290 
291  // Compute all required auxiliary variables to compute optimal angle alpha.
292  double lambdap = costates_[ orbital_element_conversions::semiParameterIndex ] * ( 2.0 * p ) / w1;
293  double lambdaf1 = costates_[ orbital_element_conversions::fElementIndex ] * std::sin( L );
294  double lambdag1 = costates_[ orbital_element_conversions::gElementIndex ] * std::cos( L );
295  double lambdaf2 = costates_[ orbital_element_conversions::fElementIndex ] / w1 *
296  ( ( w1 + 1.0 ) * std::cos( L ) + f );
297  //costates_[ orbital_element_conversions::fElementIndex ] * ( ( w1 + 1.0 ) * std::cos( L ) + f ) / w1;
298  double lambdag2 = costates_[ orbital_element_conversions::gElementIndex ] / w1 * ( ( w1 + 1.0 ) * std::sin( L ) + g );
299 
300  // Compute sinus of the optimal value of angle alpha.
301  double sinOptimalAlpha = - ( lambdaf1 - lambdag1 ) /
302  std::sqrt( ( lambdaf1 - lambdag1 ) * ( lambdaf1 - lambdag1 ) +
303  ( lambdap + lambdaf2 + lambdag2 ) * ( lambdap + lambdaf2 + lambdag2 ) );
304 
305  // Compute cosinus of the optimal value of angle alpha.
306  double cosOptimalAlpha = - ( lambdap + lambdaf2 + lambdag2 ) /
307  std::sqrt( ( lambdaf1 - lambdag1 ) * ( lambdaf1 - lambdag1 ) +
308  ( lambdap + lambdaf2 + lambdag2 ) * ( lambdap + lambdaf2 + lambdag2 ) );
309 
310  // Compute all required auxiliary variables to compute optimal angle beta.
311  lambdap = costates_[ orbital_element_conversions::semiParameterIndex ] * ( 2.0 * p ) / w1 * cosOptimalAlpha;
312  lambdaf1 = costates_[ orbital_element_conversions::fElementIndex ] * std::sin( L ) * sinOptimalAlpha;
313  lambdag1 = costates_[ orbital_element_conversions::gElementIndex ] * std::cos( L ) * sinOptimalAlpha;
314  lambdaf2 = costates_[ orbital_element_conversions::fElementIndex ] * ( ( 1.0 + w1 ) * std::cos( L ) + f ) / w1 * cosOptimalAlpha;
315  lambdag2 = costates_[ orbital_element_conversions::gElementIndex ] * ( ( 1.0 + w1 ) * std::sin( L ) + g ) / w1 * cosOptimalAlpha;
316  double lambdaf3 = costates_[ orbital_element_conversions::fElementIndex ] * ( g / w1 ) * ( h * std::sin( L ) - k * std::cos( L ) );
317  double lambdag3 = costates_[ orbital_element_conversions::gElementIndex ] * ( f / w1 ) * ( h * std::sin( L ) - k * std::cos( L ) );
318  double lambdah = costates_[ orbital_element_conversions::hElementIndex ] * ( w2 * std::cos( L ) ) / ( 2.0 * w1 );
319  double lambdak = costates_[ orbital_element_conversions::kElementIndex ] * ( w2 * std::sin( L ) ) / ( 2.0 * w1 );
320 
321  // Compute sinus of optimal thrust angle beta.
322  double sinOptimalBeta = - ( - lambdaf3 + lambdag3 + lambdah + lambdak ) /
323  std::sqrt( ( - lambdaf3 + lambdag3 + lambdah + lambdak ) * ( - lambdaf3 + lambdag3 + lambdah + lambdak )
324  + ( lambdap + lambdaf1 - lambdag1 + lambdaf2 + lambdag2 )
325  * ( lambdap + lambdaf1 - lambdag1 + lambdaf2 + lambdag2 ) );
326 
327  // Compute cosinus of optimal thrust angle beta.
328  double cosOptimalBeta = - ( lambdap + lambdaf1 - lambdag1 + lambdaf2 + lambdag2 ) /
329  std::sqrt( ( - lambdaf3 + lambdag3 + lambdah + lambdak ) * ( - lambdaf3 + lambdag3 + lambdah + lambdak )
330  + ( lambdap + lambdaf1 - lambdag1 + lambdaf2 + lambdag2 )
331  * ( lambdap + lambdaf1 - lambdag1 + lambdaf2 + lambdag2 ) );
332 
333 
334  // Switching function for the thrust magnitude.
335  double thrustMagnitudeSwitchingCondition = ( 1.0 / thrustingBodyMassFunction_( ) ) *
336  ( lambdap * cosOptimalBeta + lambdah * sinOptimalBeta + lambdak * sinOptimalBeta
337  + lambdaf1 * cosOptimalBeta + lambdaf2 * cosOptimalBeta - lambdaf3 * sinOptimalBeta
338  - lambdag1 * cosOptimalBeta + lambdag2 * cosOptimalBeta + lambdag3 * sinOptimalBeta );
339 
340 
341  // Compute current thrust magnitude and specific impulse.
342  if ( thrustMagnitudeSwitchingCondition <= 0.0 )
343  {
344 // std::cout << "INSIDE THRUST MAGNITUDE FUNCTION, THRUST ON. " << "\n\n";
345  currentThrustMagnitude_ = maximumThrustMagnitude_;
346  currentSpecificImpulse_ = specificImpulseFunction_( time );
347  }
348  else
349  {
350 // std::cout << "INSIDE THRUST MAGNITUDE FUNCTION, THRUST OFF. " << "\n\n";
351  currentThrustMagnitude_ = 0.0;
352  currentSpecificImpulse_ = specificImpulseFunction_( time );
353  }
354 
355  currentTime_ = time;
356  }
357  }
