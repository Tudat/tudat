75  Eigen::Matrix< ObservationScalarType, 1, 1 > computeIdealObservations(
76  const TimeType time,
77  const LinkEndType linkEndAssociatedWithTime )
78 
79  {
80  // Check link end associated with input time.
81  bool isTimeAtReception = -1;
82  if( linkEndAssociatedWithTime == receiver )
83  {
84  isTimeAtReception = 1;
85  }
86  else if( linkEndAssociatedWithTime == transmitter )
87  {
88  isTimeAtReception = 0;
89  }
90  else
91  {
92  throw std::runtime_error(
93  "Error when calculating one way range observation, link end is not transmitter or receiver" );
94  }
95 
96  // Calculate light-time and multiply by speed of light in vacuum.
97  return ( Eigen::Matrix< ObservationScalarType, 1, 1 >( ) <<
98  lightTimeCalculator_->calculateLightTime( time, isTimeAtReception ) *
99  physical_constants::getSpeedOfLight< ObservationScalarType >( ) ).finished( );
100  }

117  Eigen::Matrix< ObservationScalarType, 1, 1 > computeIdealObservationsWithLinkEndData(
118  const TimeType time,
119  const LinkEndType linkEndAssociatedWithTime,
120  std::vector< double >& linkEndTimes,
121  std::vector< Eigen::Matrix< double, 6, 1 > >& linkEndStates )
122  {
123  linkEndTimes.clear( );
124  linkEndStates.clear( );
125 
126  ObservationScalarType observation = TUDAT_NAN;
127  TimeType transmissionTime = TUDAT_NAN, receptionTime = TUDAT_NAN;
128 
129  // Check link end associated with input time and compute observable
130  switch( linkEndAssociatedWithTime )
131  {
132  case receiver:
133  observation = lightTimeCalculator_->calculateLightTimeWithLinkEndsStates(
134  receiverState, transmitterState, time, 1 );
135  transmissionTime = time - observation;
136  receptionTime = time;
137  break;
138 
139  case transmitter:
140  observation = lightTimeCalculator_->calculateLightTimeWithLinkEndsStates(
141  receiverState, transmitterState, time, 0 );
142  transmissionTime = time;
143  receptionTime = time + observation;
144  break;
145  default:
146  std::string errorMessage = "Error, cannot have link end type: " +
147  std::to_string( linkEndAssociatedWithTime ) + "for one-way range";
148  throw std::runtime_error( errorMessage );
149  }
150 
151  // Convert light time to range.
152  observation *= physical_constants::getSpeedOfLight< ObservationScalarType >( );
153 
154  // Set link end states and times.
155  linkEndTimes.push_back( static_cast< double >( transmissionTime ) );
156  linkEndTimes.push_back( static_cast< double >( receptionTime ) );
157 
158  linkEndStates.push_back( transmitterState.template cast< double >( ) );
159  linkEndStates.push_back( receiverState.template cast< double >( ) );
160 
161  return ( Eigen::Matrix< ObservationScalarType, 1, 1 >( ) << observation ).finished( );
162  }
