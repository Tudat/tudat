111  NBodyStateDerivative( const basic_astrodynamics::AccelerationMap& accelerationModelsPerBody,
112  const std::shared_ptr< CentralBodyData< StateScalarType, TimeType > > centralBodyData,
113  const TranslationalPropagatorType propagatorType,
114  const std::vector< std::string >& bodiesToIntegrate ):
115  propagators::SingleStateTypeDerivative< StateScalarType, TimeType >(
116  propagators::translational_state ),
117  accelerationModelsPerBody_( accelerationModelsPerBody ),
118  centralBodyData_( centralBodyData ),
119  propagatorType_( propagatorType ),
120  bodiesToBeIntegratedNumerically_( bodiesToIntegrate )
121  {
122  // Add empty acceleration map if body is to be propagated with no accelerations.
123  for( unsigned int i = 0; i < bodiesToBeIntegratedNumerically_.size( ); i++ )
124  {
125  if( accelerationModelsPerBody_.count( bodiesToBeIntegratedNumerically_.at( i ) ) == 0 )
126  {
127  accelerationModelsPerBody_[ bodiesToBeIntegratedNumerically_.at( i ) ] =
128  basic_astrodynamics::SingleBodyAccelerationMap( );
129  }
130  }
131 
132  // Correct order of propagated bodies.
133  for( outerAccelerationIterator = accelerationModelsPerBody_.begin( );
134  outerAccelerationIterator != accelerationModelsPerBody_.end( );
135  outerAccelerationIterator++ )
136  {
137  std::vector< std::string >::iterator findIterator =
138  std::find( bodiesToBeIntegratedNumerically_.begin( ), bodiesToBeIntegratedNumerically_.end( ),
139  outerAccelerationIterator->first );
140  bodyOrder_.push_back( std::distance( bodiesToBeIntegratedNumerically_.begin( ), findIterator ) );
141  }

283  Eigen::Vector3d getTotalAccelerationForBody(
284  const std::string& bodyName )
285  {
286  // Check if body is propagated.
287  Eigen::Vector3d totalAcceleration = Eigen::Vector3d::Zero( );
288  if( std::find( bodiesToBeIntegratedNumerically_.begin( ),
289  bodiesToBeIntegratedNumerically_.end( ),
290  bodyName ) == bodiesToBeIntegratedNumerically_.end( ) )
291  {
292  std::string errorMessage = "Error when getting total acceleration for body " + bodyName +
293  ", no such acceleration is found";
294  throw std::runtime_error( errorMessage );
295  }
296  else
297  {
298  if( accelerationModelsPerBody_.count( bodyName ) != 0 )
299  {
300  basic_astrodynamics::SingleBodyAccelerationMap accelerationsOnBody =
301  accelerationModelsPerBody_.at( bodyName );
302 
303  // Iterate over all accelerations acting on body
304  for( innerAccelerationIterator = accelerationsOnBody.begin( );
305  innerAccelerationIterator != accelerationsOnBody.end( );
306  innerAccelerationIterator++ )
307  {
308  for( unsigned int j = 0; j < innerAccelerationIterator->second.size( ); j++ )
309  {
310  // Calculate acceleration and add to state derivative.
311  totalAcceleration += innerAccelerationIterator->second[ j ]->getAcceleration( );
312  }
313  }
314  }
315  }
316  return totalAcceleration;
317  }

364  void sumStateDerivativeContributions(
365  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
366  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative,
367  const bool addPositionDerivatives = true )
368  {
369  using namespace basic_astrodynamics;
370 
371  stateDerivative.setZero( );
372 
373  int currentBodyIndex = 0;
374  int currentAccelerationIndex = 0;
375 
376  // Iterate over all bodies with accelerations.
377  for( outerAccelerationIterator = accelerationModelsPerBody_.begin( );
378  outerAccelerationIterator != accelerationModelsPerBody_.end( );
379  outerAccelerationIterator++ )
380  {
381  currentBodyIndex = bodyOrder_[ currentAccelerationIndex ];
382 
383  // Iterate over all accelerations acting on body
384  for( innerAccelerationIterator = outerAccelerationIterator->second.begin( );
385  innerAccelerationIterator != outerAccelerationIterator->second.end( );
386  innerAccelerationIterator++ )
387  {
388  for( unsigned int j = 0; j < innerAccelerationIterator->second.size( ); j++ )
389  {
390  // Calculate acceleration and add to state derivative.
391  stateDerivative.block( currentBodyIndex * 6 + 3, 0, 3, 1 ) += (
392  innerAccelerationIterator->second[ j ]->getAcceleration( ) ).
393  template cast< StateScalarType >( );
394  }
395  }
396 
397  if( addPositionDerivatives )
398  {
399  // Add body velocity as derivative of its position.
400  stateDerivative.block( currentBodyIndex * 6, 0, 3, 1 ) =
401  ( stateOfSystemToBeIntegrated.segment( currentBodyIndex * 6 + 3, 3 ) );
402  }
403  currentAccelerationIndex++;
404  }
405  }
