94  Eigen::MatrixXd wrtParameter(
95  const std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
96  {
97  // Initialize partial
98  Eigen::MatrixXd partial = Eigen::MatrixXd( accelerationSize_, 1 );
99 
100  // Get partial computation function.
101  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunction =
102  getParameterPartialFunction( parameter );
103 
104  // If parameter dependency exists, compute it, otherwise set partial to zero.
105  if( partialFunction.second > 0 )
106  {
107  partialFunction.first( partial );
108  }
109  else
110  {
111  partial = Eigen::MatrixXd::Zero(
112  propagators::getGeneralizedAccelerationSize( integratedStateType_ ), 1 );
113  }
114 
115  return partial;
116  }

233  void getCurrentParameterPartial(
234  const std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter,
235  Eigen::Block< Eigen::MatrixXd > partialMatrix )
236  {
237  // Check if dependecy is computed
238  if( currentDoubleParameterPartials_.count( parameter ) == 0 )
239  {
240  // Check if dependecy exists at all
241  if( parameterDoublePartialFunctions_.count( parameter ) == 0 )
242  {
243  throw std::runtime_error(
244  "Parameter of type " +
245  std::to_string( parameter->getParameterName( ).first ) + ", " +
246  std::string( parameter->getParameterName( ).second.first ) + ", " +
247  std::string( parameter->getParameterName( ).second.second ) + ", " +
248  " not found in list of existing partials" );
249  }
250  else
251  {
252  std::cerr << "Warning, double partial should already be calculated" << std::endl;
253  parameterDoublePartialFunctions_.at( parameter )( currentDoubleParameterPartials_[ parameter ] );
254  }
255  }
256  partialMatrix += currentDoubleParameterPartials_[ parameter ];
257  }

332  virtual int setParameterPartialUpdateFunction(
333  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
334  {
335  // Get partial function.
336  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > parameterPartialFunction =
337  getParameterPartialFunction( parameter );
338 
339  // If partial function found, add function to list of computations to be performed when calling
340  // updateParameterPartials.
341  if( parameterPartialFunction.second > 0 && parameterDoublePartialFunctions_.count( parameter ) == 0 )
342  {
343  if( parameterDoublePartialFunctions_.count( parameter ) == 0 )
344  {
345  parameterDoublePartialFunctions_[ parameter ] = parameterPartialFunction.first;
346  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
347  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
348  }
349  }
350 
351  return parameterPartialFunction.second;
352  }

388  void updateParameterPartials( )
389  {
390  // Update member object partials.
391  updateParameterPartialsOfMemberObjects( );
392 
393  // Update double parameter partials
394  for( parameterDoublePartialFunctionIterator_ = parameterDoublePartialFunctions_.begin( );
395  parameterDoublePartialFunctionIterator_ != parameterDoublePartialFunctions_.end( );
396  parameterDoublePartialFunctionIterator_++ )
397  {
398  if( isCurrentDoubleParameterPartialSet_.at( parameterDoublePartialFunctionIterator_->first ) == 0 )
399  {
400  parameterDoublePartialFunctionIterator_->second( currentDoubleParameterPartials_[ parameterDoublePartialFunctionIterator_->first ] );
401  isCurrentDoubleParameterPartialSet_[ parameterDoublePartialFunctionIterator_->first ] = 1;
402  }
403  }
404 
405  // Update vector parameter partials
406  for( parameterVectorPartialFunctionIterator_ = parameterVectorPartialFunctions_.begin( );
407  parameterVectorPartialFunctionIterator_ != parameterVectorPartialFunctions_.end( );
408  parameterVectorPartialFunctionIterator_++ )
409  {
410  if( isCurrentVectorParameterPartialSet_.at( parameterVectorPartialFunctionIterator_->first ) == 0 )
411  {
412  parameterVectorPartialFunctionIterator_->second( currentVectorParameterPartials_[ parameterVectorPartialFunctionIterator_->first ] );
413  isCurrentVectorParameterPartialSet_[ parameterVectorPartialFunctionIterator_->first ] = 1;
414  }
