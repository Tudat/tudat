165  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
166  {
167  // Loop over each body
168  const double tolerance = 20.0 * std::numeric_limits< double >::epsilon( );
169  Eigen::Matrix< StateScalarType, 4, 1 > quaternionsVector;
170  StateScalarType quaternionsMagnitude;
171  for( unsigned int i = 0; i < this->bodiesToPropagate_.size( ); i++ )
172  {
173  // Normalize quaternions
174  quaternionsVector = unprocessedState.block( i * 7, 0, 4, 1 );
175  quaternionsMagnitude = quaternionsVector.norm( );
176  if ( std::fabs( quaternionsMagnitude - 1.0 ) >= tolerance )
177  {
178  // Normalize
179  quaternionsVector /= quaternionsMagnitude;
180 
181  // Replace old quaternions with normalized quaternions
182  unprocessedState.segment( i * 7, 4 ) = quaternionsVector;
183  }
184  }
185  }
