182  void updateInterpolator( const double time )
183  {
184  if( !( time == currentTime_ ) )
185  {
186  // Retrieve data from interpolator
187  currentRotationalState_ = interpolator_->interpolate( time );
188 
189  // Normalize quaternion and set current rotation quaternion
190  double quaternionNorm = ( currentRotationalState_.block( 0, 0, 4, 1 ) ).norm( );
191  currentRotationalState_.block( 0, 0, 4, 1 ) = currentRotationalState_.block( 0, 0, 4, 1 ) / quaternionNorm;
192  currentRotationToBaseFrame_ = Eigen::Quaternion< StateScalarType >(
193  currentRotationalState_( 0 ), currentRotationalState_( 1 ),
194  currentRotationalState_( 2 ), currentRotationalState_( 3 ) );
195 
196  // Set current angular velocity vector.
197  currentRotationalVelocityVectorInTargetFrame_ = currentRotationalState_.block( 4, 0, 3, 1 );
198 
199  // Set time at which interpolator was evaluated.
200  currentTime_ = time;
201  }
202  }

236 template< typename StateScalarType = double, typename TimeType = double >
237 std::shared_ptr< RotationalEphemeris > getTabulatedRotationalEphemeris(
238  const std::shared_ptr< RotationalEphemeris > ephemerisToInterrogate,
239  const TimeType startTime,
240  const TimeType endTime,
241  const TimeType timeStep,
242  const std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
243  std::make_shared< interpolators::LagrangeInterpolatorSettings >( 8 ) )
244 {
245  typedef Eigen::Matrix< StateScalarType, 7, 1 > StateType;
246 
247  // Create state map that is to be interpolated
248  std::map< TimeType, StateType > stateMap;
249  TimeType currentTime = startTime;
250  while( currentTime <= endTime )
251  {
252  stateMap[ currentTime ] = ephemerisToInterrogate->getRotationStateVector( currentTime );
253  currentTime += timeStep;
254  }
255 
256  // Create tabulated ephemeris model
257  return std::make_shared< TabulatedRotationalEphemeris< StateScalarType, TimeType > >(
258  interpolators::createOneDimensionalInterpolator( stateMap, interpolatorSettings ),
259  ephemerisToInterrogate->getBaseFrameOrientation( ),
260  ephemerisToInterrogate->getTargetFrameOrientation( ) );
261 
262 }
