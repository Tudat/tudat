52 template< typename ScalarType = double >
53 Eigen::Matrix< ScalarType, Eigen::Dynamic, Eigen::Dynamic > readMatrixFromFile(
54  const std::string& relativePath,
55  const std::string& separators = "\t ;,",
56  const std::string& skipLinesCharacter = "%",
57  const int numberOfHeaderLines = 0 )
58 {
59  // Open input and output.
60  std::fstream file( relativePath.c_str( ), std::ios::in );
61  if ( file.fail( ) )
62  {
63  throw std::runtime_error( "Data file could not be opened:" + relativePath );
64  }
65 
66  std::stringstream filteredStream( std::ios::in | std::ios::out );
67  {
68  // Filter the file stream. This needs to be in its own scope, because filtering_stream::
69  // flush( ) does not work if the underlying end point is a stringstream, so the flush has
70  // to be forced by letting the filtering_stream go out of scope.
71  boost::iostreams::filtering_ostream filterProcessor;
72  for ( unsigned int i = 0; i < skipLinesCharacter.size( ); i++ )
73  {
74  // Remove all comments from the stream.
75  filterProcessor.push( input_output::stream_filters::RemoveComment(
76  skipLinesCharacter[ i ], true ) );
77  }
78 
79  // Add the output to the filter chain.
80  filterProcessor.push( filteredStream );
81 
82  // Copy the input to the filter.
83  boost::iostreams::copy( file, filterProcessor );
84  }
85 
86  // Seek stream back to start.
87  filteredStream.seekg( 0, std::ios::beg );
88 
89  // Read the filtered stream into lines.
90  std::vector< std::string > lines_;
91  int numberOfLinesParsed = 0;
92  while ( !filteredStream.eof( ) )
93  {
94  std::string line_;
95  getline( filteredStream, line_ );
96  if ( numberOfLinesParsed >= numberOfHeaderLines )
97  {
98  if ( !line_.empty( ) )
99  {
100  boost::trim_all( line_ );
101  lines_.push_back( line_ );
102  }
103  }
104  numberOfLinesParsed++;
105  }
106 
107  // If there are no lines, return an empty matrix.
108  if ( lines_.empty( ) )
109  {
110  return Eigen::Matrix< ScalarType, Eigen::Dynamic, Eigen::Dynamic >( );
111  }
112 
113  const std::string realSeparators = std::string( separators ) + " ";
114 
115  // Determine the number of columns from.
116  std::vector< std::string > lineSplit_;
117  boost::algorithm::split( lineSplit_, lines_[ 0 ], boost::is_any_of( realSeparators ),
118  boost::algorithm::token_compress_on );
119  const unsigned int numberOfColumns = lineSplit_.size( );
120 
121  // Initialize the matrix with sizes obtained from the number of lines and the entries in the
122  // first line.
123  Eigen::Matrix< ScalarType, Eigen::Dynamic, Eigen::Dynamic > dataMatrix_( lines_.size( ), numberOfColumns );
124  for ( int rowIndex = 0; rowIndex < dataMatrix_.rows( ); rowIndex++ )
125  {
126  lineSplit_.clear( );
127 
128  // Read current line and split into separate entries.
129  boost::algorithm::split( lineSplit_, lines_[ rowIndex ],
130  boost::is_any_of( realSeparators ),
131  boost::algorithm::token_compress_on );
132 
133  // Check if number of column entries in line matches the number of columns in the matrix.
134  // If not, throw a runtime error.
135  if ( lineSplit_.size( ) != numberOfColumns )
136  {
137  throw std::runtime_error( "Number of colums in row " + std::to_string( rowIndex )
138  + " is " + std::to_string( lineSplit_.size( ) ) + "; should be " + std::to_string( numberOfColumns ) );
139  }
140 
141  // Put single line entries into matrix as doubles.
142  for ( int columnIndex = 0; columnIndex < dataMatrix_.cols( ); columnIndex++ )
143  {
144  boost::trim( lineSplit_.at( columnIndex ) );
145  dataMatrix_( rowIndex, columnIndex ) =
146  boost::lexical_cast< ScalarType >( lineSplit_.at( columnIndex ) );
147  }
148  }
149 
150  return dataMatrix_;
151 }
