48 template< typename IndependentVariableType, typename DependentVariableType >
49 std::vector< DependentVariableType > solveTridiagonalMatrixEquation(
50  const std::vector< IndependentVariableType >& subDiagonal,
51  const std::vector< IndependentVariableType >& diagonal,
52  const std::vector< IndependentVariableType >& superDiagonal,
53  const std::vector< DependentVariableType >& rightHandSide )
54 {
55  // Check whether input diagonals are correct size.
56  unsigned int matrixSize = rightHandSide.size( );
57  if ( ( diagonal.size( ) < matrixSize ) || ( superDiagonal.size( ) < matrixSize - 1 ) ||
58  ( rightHandSide.size( ) < matrixSize - 1 ) )
59  {
60  throw std::runtime_error( "Error, input provided for diagonal and sub/super diagonals incorrect." );
61  }
62 
63  // Check whether solution will not be singular.
64  if ( diagonal[ 0 ] == 0.0 )
65  {
66  throw std::runtime_error( "Error when inverting tridiagonal system, first entry of diagonal is zero" );
67  }
68 
69  std::vector< IndependentVariableType > intermediateVector( matrixSize );
70  std::vector< DependentVariableType > solution( matrixSize );
71 
72  // Perform solution algorithm, from (Press W.H., et al., 2002).
73  double scalingFactor = diagonal[ 0 ];
74  solution[ 0 ] = rightHandSide[ 0 ] / scalingFactor;
75 
76  for ( unsigned int j = 1; j < matrixSize; j++ )
77  {
78  intermediateVector[ j ] = superDiagonal[ j - 1 ] / scalingFactor;
79  scalingFactor = diagonal[ j ] - subDiagonal[ j - 1 ] * intermediateVector[ j ];
80 
81  // Check whether solution will not be singular.
82  if ( scalingFactor == 0.0 )
83  {
84  throw std::runtime_error( "Error when inverting tridiagonal system, scaling factor equals zero!" );
85  }
86  solution[ j ] = ( rightHandSide[ j ] - subDiagonal[ j - 1 ] * solution[ j - 1 ] ) /
87  scalingFactor;
88  }
89 
90  for ( int j = ( matrixSize - 2 ); j >= 0 ; j-- )
91  {
92  solution[ j ] -= intermediateVector[ j + 1 ] * solution[ j + 1 ];
93  }
94 
95  return solution;
96 }

282  DependentVariableType interpolate( const IndependentVariableType targetIndependentVariableValue )
283  {
284  // Check whether boundary handling needs to be applied, if independent variable is beyond its defined range.
285  DependentVariableType interpolatedValue;
286  bool useValue = false;
287  this->checkBoundaryCase( interpolatedValue, useValue, targetIndependentVariableValue );
288  if( useValue )
289  {
290  return interpolatedValue;
291  }
292 
293  // Determine the lower entry in the table corresponding to the target independent variable
294  // value.
295  int lowerEntry_ = lookUpScheme_->findNearestLowerNeighbour(
296  targetIndependentVariableValue );
297 
298  // Get independent variable values bounding interval in which requested value lies.
299  IndependentVariableType lowerValue, upperValue;
300  ScalarType squareDifference;
301  lowerValue = independentValues_[ lowerEntry_ ];
302  upperValue = independentValues_[ lowerEntry_ + 1 ];
303 
304  // Calculate coefficients A,B,C,D (see Numerical (Press W.H., et al., 2002))
305  squareDifference = static_cast< ScalarType >( upperValue - lowerValue ) *
306  static_cast< ScalarType >( upperValue - lowerValue );
307  ScalarType coefficientA_ = ( upperValue - targetIndependentVariableValue )
308  / static_cast< ScalarType >( upperValue - lowerValue );
309  ScalarType coefficientB_ = mathematical_constants::getFloatingInteger< ScalarType >( 1.0 ) - coefficientA_;
310  ScalarType coefficientC_ = ( coefficientA_ * coefficientA_ * coefficientA_ - coefficientA_ ) /
311  mathematical_constants::getFloatingInteger< ScalarType >( 6.0 ) * squareDifference;
312  ScalarType coefficientD_ = ( coefficientB_ * coefficientB_ * coefficientB_ - coefficientB_ ) /
313  mathematical_constants::getFloatingInteger< ScalarType >( 6.0 ) * squareDifference;
314 
315  // The interpolated dependent variable value.
316  return coefficientA_ * dependentValues_[ lowerEntry_ ] +
317  coefficientB_ * dependentValues_[ lowerEntry_ + 1 ] +
318  coefficientC_ * secondDerivativeOfCurve_[ lowerEntry_ ] +
319  coefficientD_ * secondDerivativeOfCurve_[ lowerEntry_ + 1 ];
320  }

332  void calculateSecondDerivatives( )
333  {
334  // Get length of vector.
335  numberOfDataPoints_ = independentValues_.size( );
336 
337  // Sub-diagonal of tri-diagonal matrix.
338  std::vector< ScalarType > aCoefficients_;
339  aCoefficients_.resize( numberOfDataPoints_ - 2 );
340 
341  // Diagonal of tri-diagonal matrix.
342  std::vector< ScalarType > bCoefficients_;
343  bCoefficients_.resize( numberOfDataPoints_ - 2 );
344 
345  // Super-diagonal of tri-diagonal matrix.
346  std::vector< ScalarType > cCoefficients_;
347  cCoefficients_.resize( numberOfDataPoints_ - 2 );
348 
349  // Right-hand side of tridiagonal matrix system
350  std::vector< DependentVariableType > rCoefficients_;
351  rCoefficients_.resize( numberOfDataPoints_ - 2 );
352 
353  // Temporary value vector.
354  std::vector< ScalarType > hCoefficients_;
355  hCoefficients_.resize( numberOfDataPoints_ - 1 );
356 
357  // Set second derivatives of curve to zero at endpoints, i.e. impose natural spline
358  // condition.
359  aCoefficients_[ numberOfDataPoints_- 3 ] = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
360  cCoefficients_[ numberOfDataPoints_- 3 ] = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
361 
362  // Compute the vectors h (temporary values),a,c,b,r.
363  for ( unsigned int i = 0; i < ( numberOfDataPoints_ - 1 ); i++ )
364  {
365  hCoefficients_[ i ] = independentValues_[ i + 1 ] - independentValues_[ i ];
366  }
367 
368  // Set tridiagonal matrix equation input.
369  for ( unsigned int i = 0; i < ( numberOfDataPoints_ - 3 ); i++ )
370  {
371  aCoefficients_[ i ] = hCoefficients_[ i + 1 ];
372  cCoefficients_[ i ] = hCoefficients_[ i + 1 ];
373  }
374 
375  for ( unsigned int i = 0; i < ( numberOfDataPoints_ - 2 ); i++ )
376  {
377  bCoefficients_[ i ] = 2.0 * ( hCoefficients_[ i + 1 ] + hCoefficients_[ i ] );
378  rCoefficients_[ i ] = 6.0 * ( ( dependentValues_[ i + 2 ]-
379  dependentValues_[ i + 1 ] ) / hCoefficients_[ i + 1 ] -
380  ( dependentValues_[ i + 1 ] - dependentValues_[ i ] ) /
381  hCoefficients_[ i ] );
382  }
383 
384  // Solve tridiagonal matrix equatuion.
385  std::vector< DependentVariableType > middleSecondDerivativeOfCurvatures =
386  solveTridiagonalMatrixEquation< ScalarType, DependentVariableType >
387  ( aCoefficients_, bCoefficients_, cCoefficients_, rCoefficients_ );
388 
389  // Append zeros to ends of calculated second derivative values (natural spline condition).
390  secondDerivativeOfCurve_.resize( numberOfDataPoints_ );
391  secondDerivativeOfCurve_[ 0 ] = zeroValue_;
392 
393  for ( unsigned int i = 1; i < numberOfDataPoints_ - 1; i++ )
394  {
395  secondDerivativeOfCurve_[ i ] = middleSecondDerivativeOfCurvatures[ i - 1 ];
396  }
397 
398  secondDerivativeOfCurve_[ numberOfDataPoints_ - 1 ] = zeroValue_;
399  }
400 
402  std::vector< DependentVariableType > secondDerivativeOfCurve_;
403 
405  unsigned int numberOfDataPoints_;
406 
408  DependentVariableType zeroValue_;
409
