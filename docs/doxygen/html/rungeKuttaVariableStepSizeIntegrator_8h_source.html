506 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
507 StateType
508 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
509 ::performIntegrationStep( const TimeStepType stepSize )
510 {
511  if( !( stepSize == stepSize ) )
512  {
513  throw std::runtime_error( "Error in RKF integrator, step size is NaN" );
514  }
515 
516  // Define and allocated vector for the number of stages.
517  currentStateDerivatives_.clear( );
518  currentStateDerivatives_.reserve( this->coefficients_.cCoefficients.rows( ) );
519 
520  // Define lower and higher order estimates.
521  StateType lowerOrderEstimate( this->currentState_ ), higherOrderEstimate( this->currentState_ );
522 
523  // Compute the k_i state derivatives per stage.
524  for ( int stage = 0; stage < this->coefficients_.cCoefficients.rows( ); stage++ )
525  {
526  // Compute the intermediate state to pass to the state derivative for this stage.
527  StateType intermediateState( this->currentState_ );
528 
529  // Compute the intermediate state.
530  for ( int column = 0; column < stage; column++ )
531  {
532  intermediateState += stepSize * this->coefficients_.aCoefficients( stage, column ) *
533  currentStateDerivatives_[ column ];
534  }
535 
536  // Compute the state derivative.
537  const IndependentVariableType time = this->currentIndependentVariable_ +
538  this->coefficients_.cCoefficients( stage ) * stepSize;
539  currentStateDerivatives_.push_back( this->stateDerivativeFunction_( time, intermediateState ) );
540 
541  // Check if propagation should terminate because the propagation termination condition has been reached
542  // while computing the intermediate state.
543  // If so, return immediately the current state (not recomputed yet), which will be discarded.
544  if ( this->propagationTerminationFunction_( static_cast< double >( time ), TUDAT_NAN ) )
545  {
546  this->propagationTerminationConditionReachedDuringStep_ = true;
547  return this->currentState_;
548  }
549 
550  // Update the estimate.
551  lowerOrderEstimate += this->coefficients_.bCoefficients( 0, stage ) * stepSize *
552  currentStateDerivatives_[ stage ];
553  higherOrderEstimate += this->coefficients_.bCoefficients( 1, stage ) * stepSize *
554  currentStateDerivatives_[ stage ];
555  }
556 
557  // Determine if the error was within bounds and compute a new step size.
558  if ( computeNextStepSizeAndValidateResult( lowerOrderEstimate,
559  higherOrderEstimate, stepSize ) )
560  {
561  // Accept the current step.
562  this->lastIndependentVariable_ = this->currentIndependentVariable_;
563  this->lastState_ = this->currentState_;
564  this->currentIndependentVariable_ += stepSize;
565 
566  switch ( this->coefficients_.orderEstimateToIntegrate )
567  {
568  case RungeKuttaCoefficients::lower:
569  this->currentState_ = lowerOrderEstimate;
570  return this->currentState_;
571 
572  case RungeKuttaCoefficients::higher:
573  this->currentState_ = higherOrderEstimate;
574  return this->currentState_;
575 
576  default: // The default case will never occur because OrderEstimateToIntegrate is an enum.
577  throw std::runtime_error( "Order estimate to integrate is invalid." );
578  }
579  }
580  else
581  {
582  // Reject current step.
583  return performIntegrationStep( this->stepSize_ );
584  }
585 }

588 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
589 bool
590 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
591 ::computeNextStepSizeAndValidateResult(
592  const StateType& lowerOrderEstimate, const StateType& higherOrderEstimate,
593  const TimeStepType stepSize )
594 {
595  if( useStepSizeControl_ )
596  {
597  // Compute new step size using new step size function, which also returns whether the
598  // relative error is within bounds or not.
599  std::pair< TimeStepType, bool > newStepSizePair = this->newStepSizeFunction_(
600  stepSize, std::make_pair( this->coefficients_.lowerOrder, this->coefficients_.higherOrder ),
601  this->safetyFactorForNextStepSize_, std::make_pair( this->minimumFactorDecreaseForNextStepSize_,
602  this->maximumFactorIncreaseForNextStepSize_ ),
603  this->relativeErrorTolerance_, this->absoluteErrorTolerance_, lowerOrderEstimate, higherOrderEstimate );
604 
605  // Check whether change in stepsize does not exceed bounds.
606  // If the stepsize is reduced to less than the prescibed minimum factor, set to minimum factor.
607  // If the stepsize is increased to more than the prescribed maximum factor, set to maximum
608  // factor. These bounds are necessary to prevent the stepsize changes from aliasing
609  // with the dynamics of the system of ODEs.
610  // Also check if maximum step size is exceeded and step next step size to maximum if necessary.
611  // Typically used bounds can be found in (Burden and Faires, 2001).
612  if ( newStepSizePair.first / stepSize <= this->minimumFactorDecreaseForNextStepSize_ )
613  {
614  this->stepSize_ = stepSize * this->minimumFactorDecreaseForNextStepSize_;
615  }
616 
617  else if ( newStepSizePair.first / stepSize >= maximumFactorIncreaseForNextStepSize_ )
618  {
619  this->stepSize_ = stepSize * this->maximumFactorIncreaseForNextStepSize_;
620  }
621 
622  else
623  {
624  this->stepSize_ = newStepSizePair.first;
625  }
626 
627  // Check if minimum step size is violated and throw exception if necessary.
628  if ( std::fabs( this->stepSize_ ) < std::fabs( this->minimumStepSize_ ) )
629  {
630  throw MinimumStepSizeExceededError( std::fabs( this->minimumStepSize_ ),
631  std::fabs( this->stepSize_ ) );
632  }
633  else if( std::fabs( this->stepSize_ ) > std::fabs( this->maximumStepSize_ ) )
634  {
635  this->stepSize_ = stepSize / std::fabs( stepSize ) * std::fabs( this->maximumStepSize_ );
636  }
637 
638  if( stepSize * this->stepSize_ < 0 )
639  {
640  throw std::runtime_error( "Error during step size control, step size flipped sign" );
641  }
642 
643  // Check if computed error in state is too large and reject step if true.
644  return newStepSizePair.second;
645  }
646  else
647  {
648  this->stepSize_ = stepSize;
649  return true;
650  }
651 }

657 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
658 std::pair< TimeStepType, bool >
659 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
660 ::computeNewStepSize(
661  const TimeStepType stepSize,
662  const std::pair< TimeStepType, TimeStepType >& orders,
663  const TimeStepType safetyFactorForNextStepSize,
664  const std::pair< TimeStepType, TimeStepType >& minimumAndMaximumFactorsForNextStepSize,
665  const StateType& relativeErrorTolerance,
666  const StateType& absoluteErrorTolerance,
667  const StateType& lowerOrderEstimate,
668  const StateType& higherOrderEstimate )
669 {
670  TUDAT_UNUSED_PARAMETER( minimumAndMaximumFactorsForNextStepSize );
671 
672  // Compute the truncation error based on the higher and lower order estimates.
673  const StateType truncationError_ =
674  ( higherOrderEstimate - lowerOrderEstimate ).array( ).abs( );
675 
676  // Compute error tolerance based on relative and absolute error tolerances.
677  const StateType errorTolerance_ =
678  ( higherOrderEstimate.array( ).abs( ) *
679  relativeErrorTolerance.array( ) ).matrix( )
680  + absoluteErrorTolerance;
681 
682  // Compute relative truncation error. This will indicate if the current step satisfies the
683  // required tolerances.
684  const StateType relativeTruncationError_ = truncationError_.array( ) /
685  errorTolerance_.array( );
686 
687  // Compute the maximum error based on the largest coefficient in the relative truncation error
688  // matrix.
689  const typename StateType::Scalar maximumErrorInState_
690  = relativeTruncationError_.array( ).abs( ).maxCoeff( );
691 
692  // Compute the new step size. This is based off of the equation given in
693  // (Montenbruck and Gill, 2005).
694  const TimeStepType newStepSize = safetyFactorForNextStepSize * stepSize
695  * std::pow( 1.0 / maximumErrorInState_, 1.0 / orders.second );
696 
697  // Check if the current state can be accepted.
698  const bool isIntegrationStepAccepted = maximumErrorInState_ <= 1.0;
699 
700  // Return the computed new step size and whether the current step taken is acceptable. If it
701  // isn't, the step will be recomputed with the computed new step size.
702  return std::make_pair( newStepSize, isIntegrationStepAccepted );
703 }
