128  void updateMembers( const double currentTime = TUDAT_NAN )
129  {
130  // Check if update is needed
131  if( !( currentTime_ == currentTime ) )
132  {
133  // Update thrust dependencies if needed
134  if( !( thrustUpdateFunction_ == nullptr ) )
135  {
136  thrustUpdateFunction_( currentTime );
137  }
138 
139  // Retrieve thrust direction.
140  currentAccelerationDirection_ = thrustDirectionFunction_( );
141 
142  if( ( std::fabs( currentAccelerationDirection_.norm( ) ) - 1.0 ) >
143  10.0 * std::numeric_limits< double >::epsilon( ) )
144  {
145  throw std::runtime_error( "Error in thrust acceleration, direction is not a unit vector" );
146  }
147 
148  // Retrieve magnitude of thrust and mass rate.
149  currentThrustMagnitude_ = thrustMagnitudeFunction_( );
150  currentMassRate_ = -massRateFunction_( );
151 
152  // Compute acceleration due to thrust
153  currentAcceleration_ = currentAccelerationDirection_ * currentThrustMagnitude_ / bodyMassFunction_( );
154 
155  // Reset current time.
156  currentTime_ = currentTime;
157 
158  }

278  MomentumWheelDesaturationThrustAcceleration(
279  const std::vector< double > thrustMidTimes,
280  const std::vector< Eigen::Vector3d > deltaVValues,
281  const double totalManeuverTime,
282  const double maneuverRiseTime ):
283  basic_astrodynamics::AccelerationModel< Eigen::Vector3d >( ),
284  totalManeuverTime_( totalManeuverTime ),
285  maneuverRiseTime_( maneuverRiseTime ),
286  deltaVValues_( deltaVValues )
287  {
288  if( thrustMidTimes.size( ) != deltaVValues.size( ) )
289  {
290  throw std::runtime_error( "Error when making momentum wheel desaturation acceleration, input is inconsistent" );
291  }
292 
293  for( unsigned int i = 0; i < deltaVValues.size( ); i++ )
294  {
295  // Compute accelerations from desaturation deltaVs.
296  accelerationValues_.push_back( deltaVValues.at( i ) / ( totalManeuverTime_ - maneuverRiseTime_ ) );
297 
298  // Compute thrust start times.
299  thrustStartTimes_.push_back( thrustMidTimes.at( i ) - totalManeuverTime_ / 2.0 );
300  }
301  thrustStartTimes_.push_back( std::numeric_limits< double >::max( ) );
302 
303  timeLookUpScheme_ = std::make_shared< interpolators::HuntingAlgorithmLookupScheme< double > >(
304  thrustStartTimes_ );
305  }

326  void updateMembers( const double currentTime = TUDAT_NAN )
327  {
328  // Check if update is needed
329  if( !( currentTime_ == currentTime ) )
330  {
331  // Find maneuver start time closest to current time.
332  nearestTimeIndex_ = timeLookUpScheme_->findNearestLowerNeighbour( currentTime );
333  double currentThrustStartTime = thrustStartTimes_.at( nearestTimeIndex_ );
334 
335  // Check if the maneuver is still ongoing.
336  if( std::fabs( currentTime - currentThrustStartTime ) < totalManeuverTime_ && ( currentTime > currentThrustStartTime ) )
337  {
338  // Retrieve desaturation thrust multiplier and compute acceleration.
339  currentThrustMultiplier_ = getDesaturationThrustMultiplier(
340  currentTime - currentThrustStartTime );
341  currentAcceleration_ = currentThrustMultiplier_ * accelerationValues_.at( nearestTimeIndex_ );
342 
343  }
344  else
345  {
346  // Set acceleration to zero if the maneuver is over.
347  currentThrustMultiplier_ = 0.0;
348  currentAcceleration_.setZero( );
349  }
350 
351  // Reset current time.
352  currentTime_ = currentTime;
353 
354  }
355 
356  }

364  double getDesaturationThrustMultiplier( const double timeSinceThrustStart )
365  {
366  // Check if the acceleration is still increasing (peak acceleration not achieved yet).
367  if( timeSinceThrustStart < maneuverRiseTime_ )
368  {
369  double timeRatio = timeSinceThrustStart / maneuverRiseTime_;
370  return timeRatio * timeRatio * ( 3.0 - 2.0 * timeRatio );
371  }
372  // Check if the peak acceleration is achieved.
373  else if( timeSinceThrustStart < totalManeuverTime_ - maneuverRiseTime_ )
374  {
375  return 1.0;
376  }
377  // Check if the acceleration decreases after having reached its maximum.
378  else if( timeSinceThrustStart < totalManeuverTime_ )
379  {
380  return getDesaturationThrustMultiplier( totalManeuverTime_ - timeSinceThrustStart );
381  }
382  // Check if no maneuver is ongoing.
383  else
384  {
385  return 0.0;
386  }
387  }
