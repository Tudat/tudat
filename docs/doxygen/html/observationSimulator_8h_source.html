38 template< int ObservationSize = 1, typename ObservationScalarType = double, typename TimeType = double >
39 std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >, bool > simulateObservationWithCheck(
40  const TimeType& observationTime,
41  const std::shared_ptr< ObservationModel< ObservationSize, ObservationScalarType, TimeType > > observationModel,
42  const LinkEndType linkEndAssociatedWithTime,
43  const std::vector< std::shared_ptr< ObservationViabilityCalculator > > linkViabilityCalculators =
44  std::vector< std::shared_ptr< ObservationViabilityCalculator > >( ) )
45 {
46  // Initialize vector with reception times.
47  bool observationFeasible = 1;
48 
49  std::vector< Eigen::Vector6d > vectorOfStates;
50  std::vector< double > vectorOfTimes;
51 
52  Eigen::Matrix< ObservationScalarType, ObservationSize, 1 > calculatedObservation =
53  observationModel->computeObservationsWithLinkEndData(
54  observationTime, linkEndAssociatedWithTime, vectorOfTimes, vectorOfStates );
55 
56  observationFeasible = isObservationViable( vectorOfStates, vectorOfTimes, linkViabilityCalculators );
57 
58  // Return pair of simulated ranges and reception times.
59  return std::make_pair( calculatedObservation, observationFeasible );
60 }

73 template< int ObservationSize = 1, typename ObservationScalarType = double, typename TimeType = double >
74 std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >, std::vector< TimeType > >
75 simulateObservationsWithCheck(
76  const std::vector< TimeType >& observationTimes,
77  const std::shared_ptr< ObservationModel< ObservationSize, ObservationScalarType, TimeType > > observationModel,
78  const LinkEndType linkEndAssociatedWithTime,
79  const std::vector< std::shared_ptr< ObservationViabilityCalculator > > linkViabilityCalculators =
80  std::vector< std::shared_ptr< ObservationViabilityCalculator > >( ) )
81 {
82  std::map< TimeType, Eigen::Matrix< ObservationScalarType, ObservationSize, 1 > > observations;
83  std::pair< Eigen::Matrix< ObservationScalarType, ObservationSize, 1 >, bool > simulatedObservation;
84 
85  for( unsigned int i = 0; i < observationTimes.size( ); i++ )
86  {
87  simulatedObservation = simulateObservationWithCheck< ObservationSize, ObservationScalarType, TimeType >(
88  observationTimes.at( i ), observationModel, linkEndAssociatedWithTime, linkViabilityCalculators );
89 
90  // Check if receiving station can view transmitting station.
91  if( simulatedObservation.second )
92  {
93  // If viable, add observable and time to vector of simulated data.
94  observations[ observationTimes[ i ] ] = simulatedObservation.first;
95  }
96  }
97 
98  // Return pair of simulated ranges and reception times.
99  return std::make_pair( utilities::createConcatenatedEigenMatrixFromMapValues( observations ),
100  utilities::createVectorFromMapKeys( observations ) );
101 }
