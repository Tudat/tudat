286  template<typename StateScalarType = double, typename TimeType = double>
287  void setStateFromEphemeris(const TimeType &time) {
288  if (!(static_cast<Time>(time) == timeOfCurrentState_)) {
289  // If body is not global frame origin, set state.
290  if (bodyIsGlobalFrameOrigin_ == 0) {
291  if (sizeof(StateScalarType) == 8) {
292  currentState_ =
293  (bodyEphemeris_->getTemplatedStateFromEphemeris<StateScalarType, TimeType>(time) + ephemerisFrameToBaseFrame_->getBaseFrameState<TimeType, StateScalarType>(time)).template cast<double>();
294  currentLongState_ = currentState_.template cast<long double>();
295  } else {
296  currentLongState_ =
297  (bodyEphemeris_->getTemplatedStateFromEphemeris<StateScalarType, TimeType>(time) + ephemerisFrameToBaseFrame_->getBaseFrameState<TimeType, StateScalarType>(time)).template cast<long double>();
298  currentState_ = currentLongState_.template cast<double>();
299  }
300  }
301  // If body is global frame origin, set state to zeroes, and barycentric state value.
302  else if (bodyIsGlobalFrameOrigin_ == 1) {
303  currentState_.setZero();
304  currentLongState_.setZero();
305 
306  if (sizeof(StateScalarType) == 8) {
307  currentBarycentricState_ =
308  ephemerisFrameToBaseFrame_->getBaseFrameState<TimeType, StateScalarType>(time).template cast<double>();
309  currentBarycentricLongState_ = currentBarycentricState_.template cast<long double>();
310  } else {
311  currentBarycentricLongState_ =
312  ephemerisFrameToBaseFrame_->getBaseFrameState<TimeType, StateScalarType>(time).template cast<long double>();
313  currentBarycentricState_ = currentBarycentricLongState_.template cast<double>();
314  }
315  } else {
316  throw std::runtime_error("Error when setting body state, global origin not yet defined.");
317  }
318 
319  timeOfCurrentState_ = static_cast<TimeType>(time);
320  }
321  }

1006  void setScaledMeanMomentOfInertia(const double scaledMeanMomentOfInertia) {
1007  double oldScaledMeanMomentOfInertia = scaledMeanMomentOfInertia_;
1008  double oldMeanMomentOfInertia =
1009  (bodyInertiaTensor_(0, 0) + bodyInertiaTensor_(1, 1) + bodyInertiaTensor_(2, 2)) / 3.0;
1010  scaledMeanMomentOfInertia_ = scaledMeanMomentOfInertia;
1011  double meanMomentOfInertia = scaledMeanMomentOfInertia_ / oldScaledMeanMomentOfInertia * oldMeanMomentOfInertia;
